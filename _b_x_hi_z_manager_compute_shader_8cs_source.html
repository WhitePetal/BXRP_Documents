84 private void Setup(BXMainCameraRenderBase mainRender)
85 {
86 isViewCamera = mainRender.camera.cameraType == CameraType.SceneView;
87 if (isViewCamera) return;
88 this.cs = mainRender.commonSettings.hizCompute;
89 this.projectionMatrix = GL.GetGPUProjectionMatrix(mainRender.camera.projectionMatrix, false) * mainRender.camera.worldToCameraMatrix;
90 mipCount = 0;
91 this.screenSize = math.int2(mainRender.width, mainRender.height);
92 int texW = ((mainRender.width >> 3) + (mainRender.width >> 4)) + 1;

107 private void Render(CommandBuffer commandBuffer)
108 {
109 if (isViewCamera) return;
110 commandBuffer.BeginSample(SampleName);
111 commandBuffer.SetComputeTextureParam(cs, 0, BXShaderPropertyIDs._EncodeDepthBuffer_ID, BXShaderPropertyIDs._EncodeDepthBuffer_TargetID);
112 commandBuffer.SetComputeTextureParam(cs, 0, BaseShaderProperties._HiZMap_ID, hizMap);
113 commandBuffer.DispatchCompute(cs, 0, Mathf.CeilToInt(screenSize.x / 8), Mathf.CeilToInt(screenSize.y / 8), 1);
114
115 float2 mipSize = math.float2(screenSize.x >> 3, texSize.y);
116 float4 mipOffset = math.float4(0, 0, mipSize.x, mipSize.y);
117 mipSizes[mipCount++] = mipOffset;
118 while (mipSize.x > 1 && mipSize.y > 1)
119 {
120 commandBuffer.SetComputeVectorParam(cs, BaseShaderProperties._MipOffset_ID, mipOffset);
121 commandBuffer.SetComputeTextureParam(cs, 1, BaseShaderProperties._HiZMap_ID, hizMap);
122 commandBuffer.DispatchCompute(cs, 1, Mathf.CeilToInt(mipSize.x / 8f), Mathf.CeilToInt(mipSize.y / 8f), 1);
123 mipOffset += math.float4(mipSize.x, 0, 0, 0);
124 mipSize *= 0.5f;
125 mipSizes[mipCount++] = math.float4(mipOffset.xy, mipSize.xy);
126 if (mipSize.x <= 1 || mipSize.y <= 1) break;
127 mipOffset += math.float4(0, mipSize.y, 0, 0);
128 mipSize *= 0.5f;
129 mipSizes[mipCount++] = math.float4(mipOffset.xy, mipSize.xy);
130 if (mipSize.x <= 1 || mipSize.y <= 1) break;
131 mipOffset += math.float4(mipSize.x, 0, 0, 0);
132 mipSize *= 0.5f;
133 mipOffset.z = mipSize.x;
134 mipOffset.w = mipSize.y;
135 mipSizes[mipCount++] = math.float4(mipOffset.xy, mipSize.xy);
136 }
137
138 if (!isReadbacking && cullingObjectCount > 0)
139 {
140 commandBuffer.SetComputeTextureParam(cs, 2, BaseShaderProperties._BoundCentersTex_ID, boundCentersTex);
141 commandBuffer.SetComputeTextureParam(cs, 2, BaseShaderProperties._BoundSizesTex_ID, boundSizesTex);
142 commandBuffer.SetComputeTextureParam(cs, 2, BaseShaderProperties._HizMapInput_ID, hizMap);
143 commandBuffer.SetComputeVectorParam(cs, BaseShaderProperties._HizParams_ID, new Vector4(screenSize.x, screenSize.y, mipCount - 1, cullingObjectCount));
144 commandBuffer.SetComputeVectorParam(cs, BaseShaderProperties._HizTexSize_ID, new Vector4(texSize.x, texSize.y));
145 commandBuffer.SetComputeMatrixParam(cs, BaseShaderProperties._HizProjectionMatrix_ID, projectionMatrix);
146 commandBuffer.SetComputeVectorArrayParam(cs, BaseShaderProperties._HizMipSize_ID, mipSizes);
147 commandBuffer.SetComputeTextureParam(cs, 2, BaseShaderProperties._HizResultRT_ID, hizCullResultRT);
148 float xgroup = cullingObjectCount / 8f;
149 float ygroup = xgroup / 8f;
150 commandBuffer.DispatchCompute(cs, 2, 8, 8, 1);
151
152 hizReadbackDatas = new NativeArray<float>(dataTexWidth * dataTexHeight, Allocator.Persistent);
153 commandBuffer.RequestAsyncReadbackIntoNativeArray(ref hizReadbackDatas, hizCullResultRT, ReadBack);
154 isReadbacking = true;
155 }
156 commandBuffer.EndSample(SampleName);
157 }

210 private void ReadBack(AsyncGPUReadbackRequest obj)
211 {
212 if (obj.done)
213 {
214 if (obj.hasError)
215 {
216 hizReadbackDatas.Dispose();
217 throw new Exception("Hi-Z Readback Error");
218 }
219
220 for (int i = 0; i < cullingObjectCount; ++i)
221 {
222 if (hizReadbackDatas[i] <= 0f)
223 cullingObjects[i].renderer.renderingLayerMask = 0;

264 cullingObjects[cullingObjectCount++] = data;
