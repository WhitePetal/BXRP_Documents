63 void UpdatePorts()
64 {
65 // Get slots
66 List<GeometrySlot> inputSlots = new List<GeometrySlot>();
67 GetInputSlots(inputSlots);
68
69 // Store the edges
70 Dictionary<GeometrySlot, List<IEdge>> edgeDict = new Dictionary<GeometrySlot, List<IEdge>>();
71 foreach (GeometrySlot slot in inputSlots)
72 edgeDict.Add(slot, (List<IEdge>)slot.owner.owner.GetEdges(slot.slotReference));
73
74 // Remove old slots
75 for (int i = 0; i < inputSlots.Count; i++)
76 {
77 RemoveSlot(inputSlots[i].id);
78 }
79
80 // Add output slot
81 AddSlot(new DynamicVectorGeometrySlot(OutputSlotId, "Out", "Out", SlotType.Output, Vector4.zero));
82
83 // Add input slots
84 int[] slotIds = new int[dropdown.entries.Count + 1];
85 slotIds[dropdown.entries.Count] = OutputSlotId;
86 for (int i = 0; i < dropdown.entries.Count; i++)
87 {
88 // Get slot based on entry id
89 GeometrySlot slot = inputSlots.Where(x =>
90 x.id == dropdown.entries[i].id &&
91 x.RawDisplayName() == dropdown.entries[i].displayName &&
92 x.geometryOutputName == dropdown.entries[i].displayName).FirstOrDefault();
93
94 if (slot == null)
95 {
96 slot = new DynamicVectorGeometrySlot(dropdown.entries[i].id, dropdown.entries[i].displayName, dropdown.entries[i].displayName, SlotType.Input, Vector4.zero);
97 }
98
99 AddSlot(slot);
100 slotIds[i] = dropdown.entries[i].id;
101 }
102 RemoveSlotsNameNotMatching(slotIds);
103
104 // Reconnect the edges
105 foreach (KeyValuePair<GeometrySlot, List<IEdge>> entry in edgeDict)
106 {
107 foreach (IEdge edge in entry.Value)
108 {
109 owner.Connect(edge.outputSlot, edge.inputSlot);
110 }
111 }
112
113 ValidateNode();
114 }
