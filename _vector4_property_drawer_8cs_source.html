21 public int mUndoGroup { get; set; } = -1;
22
23 void CreateCallbacks()
24 {
25 m_KeyDownCallback = new EventCallback<KeyDownEvent>(evt =>
26 {
27 // Record Undo for input field edit
28 if (mUndoGroup == -1)
29 {
30 mUndoGroup = Undo.GetCurrentGroup();
31 preValueChangeCallback?.Invoke();
32 }
33 // Handle escaping input field edit
34 if (evt.keyCode == KeyCode.Escape && mUndoGroup > -1)
35 {
36 Undo.RevertAllDownToGroup(mUndoGroup);
37 mUndoGroup = -1;
38 evt.StopPropagation();
39 }
40 // Dont record Undo again until input field is unfocused
41 mUndoGroup++;
42 postValueChangeCallback?.Invoke();
43 });
44
45 m_FocusOutCallback = new EventCallback<FocusOutEvent>(evt =>
46 {
47 // Reset UndoGroup when done editing input field
48 mUndoGroup = -1;
49 });
50 }
51
52 public Vector4PropertyDrawer()
53 {
54 CreateCallbacks();
55 }

57 internal VisualElement CreateGUI(
58 ValueChangedCallback valueChangedCallback,
59 Vector4 fieldToDraw,
60 string labelName,
61 out VisualElement propertyVec4Field,
62 int indentLevel = 0)
63 {
64 var vector4Field = new Vector4Field { value = fieldToDraw };
65
66 var inputFields = vector4Field.Query("unity-text-input").ToList();
67 foreach (var inputField in inputFields)
68 {
69 inputField.RegisterCallback<KeyDownEvent>(m_KeyDownCallback, TrickleDown.TrickleDown);
70 inputField.RegisterCallback<FocusOutEvent>(m_FocusOutCallback, TrickleDown.TrickleDown);
71 }
72
73 vector4Field.RegisterValueChangedCallback(evt =>
74 {
75 // Only true when setting value via FieldMouseDragger
76 // Undo recorded once per dragger release
77 if (mUndoGroup == -1)
78 preValueChangeCallback?.Invoke();
79
80 valueChangedCallback(evt.newValue);
81 postValueChangeCallback?.Invoke();
82 });
83
84 propertyVec4Field = vector4Field;
85
86 var defaultRow = new PropertyRow(PropertyDrawerUtils.CreateLabel(labelName, indentLevel));
87 defaultRow.Add(propertyVec4Field);
88 defaultRow.styleSheets.Add(Resources.Load<StyleSheet>("Styles/PropertyRow"));
89 return defaultRow;
90 }
