57 internal static void CompressSH(ref SphericalHarmonicsL2 shv, float intensityScale, bool clearForDilation)
58 {
59 // Compress the range of all coefficients but the DC component to [0..1]
60 // Upper bounds taken from http://ppsloan.org/publications/Sig20_Advances.pptx
61 // Divide each coefficient by DC*f to get to [-1, 1] where f is from slide 33
62 for(int rgb = 0; rgb < 3; ++rgb)
63 {
64 for (int k = 0; k < 9; ++k)
65 shv[rgb, k] *= intensityScale;
66
67 var l0 = shv[rgb, 0];
68
69 if(l0 == 0.0f)
70 {
71 shv[rgb, 0] = 0.0f;
72 for (int k = 1; k < 9; ++k)
73 shv[rgb, k] = 0.5f;
74 }
75 else if (clearForDilation)
76 {
77 for (int k = 0; k < 9; ++k)
78 shv[rgb, k] = 0.0f;
79 }
80 else
81 {
82 // TODO: We're working on irradiance instead of radiance coefficients
83 // Add safety margin 2 to avoid out-of-bounds values
84 float l1scale = 2.0f; // Should be: 3/(2*sqrt(3)) * 2, but rounding to 2 to issues we are observing.
85 float l2scale = 3.5777088f; // 4/sqrt(5) * 2
86
87 // L_1^m
88 shv[rgb, 1] = shv[rgb, 1] / (l0 * l1scale * 2.0f) + 0.5f;
89 shv[rgb, 2] = shv[rgb, 2] / (l0 * l1scale * 2.0f) + 0.5f;
90 shv[rgb, 3] = shv[rgb, 3] / (l0 * l1scale * 2.0f) + 0.5f;
91
92 // L_2^-2
93 shv[rgb, 4] = shv[rgb, 4] / (l0 * l2scale * 2.0f) + 0.5f;
94 shv[rgb, 5] = shv[rgb, 5] / (l0 * l2scale * 2.0f) + 0.5f;
95 shv[rgb, 6] = shv[rgb, 6] / (l0 * l2scale * 2.0f) + 0.5f;
96 shv[rgb, 7] = shv[rgb, 7] / (l0 * l2scale * 2.0f) + 0.5f;
97 shv[rgb, 8] = shv[rgb, 8] / (l0 * l2scale * 2.0f) + 0.5f;
98
99 for (int coeff = 1; coeff < 9; ++coeff)
100 shv[rgb, coeff] = Mathf.Clamp01(shv[rgb, coeff]);
101 }
102 }
103 }

105 internal static void DecompressSH(ref SphericalHarmonicsL2 shv)
106 {
107 for(int rgb = 0; rgb < 3; ++rgb)
108 {
109 var l0 = shv[rgb, 0];
110
111 // See CompressSH
112 float l1scale = 2.0f;
113 float l2scale = 3.5777088f;
114
115 // L_1^m
116 shv[rgb, 1] = (shv[rgb, 1] - 0.5f) * (l0 * l1scale * 2.0f);
117 shv[rgb, 2] = (shv[rgb, 2] - 0.5f) * (l0 * l1scale * 2.0f);
118 shv[rgb, 3] = (shv[rgb, 3] - 0.5f) * (l0 * l1scale * 2.0f);
119
120 // L_2^-2
121 shv[rgb, 4] = (shv[rgb, 4] - 0.5f) * (l0 * l2scale * 2.0f);
122 shv[rgb, 5] = (shv[rgb, 5] - 0.5f) * (l0 * l2scale * 2.0f);
123 shv[rgb, 6] = (shv[rgb, 6] - 0.5f) * (l0 * l2scale * 2.0f);
124 shv[rgb, 7] = (shv[rgb, 7] - 0.5f) * (l0 * l2scale * 2.0f);
125 shv[rgb, 8] = (shv[rgb, 8] - 0.5f) * (l0 * l2scale * 2.0f);
126 }
127 }
