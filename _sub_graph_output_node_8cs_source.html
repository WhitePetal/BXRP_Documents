68 void ValidateSlotType()
69 {
70 List<GeometrySlot> slots = new List<GeometrySlot>();
71 GetInputSlots(slots);
72
73 if (!slots.Any())
74 {
75 owner.AddValidationError(objectId, s_MissingOutputSlot, GeometryCompilerMessageSeverity.Error);
76 }
77 else if (!s_ValidSlotTypes.Contains(slots.FirstOrDefault().concreteValueType))
78 {
79 IsFirstSlotValid = false;
80 owner.AddValidationError(objectId, "Preview can only compile if the first output slot is a Vector, Matrix, or Boolean type. Please adjust slot types.", GeometryCompilerMessageSeverity.Error);
81 }
