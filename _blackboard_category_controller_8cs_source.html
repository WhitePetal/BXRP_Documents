143 // Called by GraphDataStore.Subscribe after the model has been changed
144 protected override void ModelChanged(GraphData graphData, IGraphDataAction changeAction)
145 {
146 // If categoryData associated with this controller is removed by an operation, destroy controller and views associated
147 if (graphData.ContainsCategory(Model) == false)
148 {
149 Dispose();
150 return;
151 }
152
153 switch (changeAction)
154 {
155 case AddGeometryInputAction addBlackboardItemAction:
156 if (addBlackboardItemAction.geometryInputReference != null && IsInputInCategory(addBlackboardItemAction.geometryInputReference))
157 {
158 var blackboardRow = FindBlackboardRow(addBlackboardItemAction.geometryInputReference);
159 if (blackboardRow == null)
160 blackboardRow = InsertBlackboardRow(addBlackboardItemAction.geometryInputReference);
161 // Rows should auto-expand when an input is first added
162 // blackboardRow.expanded = true;
163 var propertyView = blackboardRow.Q<GGBlackboardField>();
164 if (addBlackboardItemAction.addInputActionType == AddGeometryInputAction.AddActionSource.AddMenu)
165 propertyView.OpenTextEditor();
166 }
167 break;
168
169 case CopyGeometryInputAction copyGeometryInputAction:
170 // In the specific case of only-one keywords like Material Quality and Raytracing, they can get copied, but because only one can exist, the output copied value is null
171 if (copyGeometryInputAction.copiedGeometryInput != null && IsInputInCategory(copyGeometryInputAction.copiedGeometryInput))
172 {
173 var blackboardRow = InsertBlackboardRow(copyGeometryInputAction.copiedGeometryInput, copyGeometryInputAction.insertIndex);
174 if (blackboardRow != null)
175 {
176 var graphView = ViewModel.parentView.GetFirstAncestorOfType<GeometryGraphView>();
177 var propertyView = blackboardRow.Q<GGBlackboardField>();
178 graphView?.AddToSelectionNoUndoRecord(propertyView);
179 }
180 }
181 break;
182
183 case AddItemToCategoryAction addItemToCategoryAction:
184 // If item was added to category that this controller manages, then add blackboard row to represent that item
185 if (addItemToCategoryAction.itemToAdd != null && addItemToCategoryAction.categoryGuid == ViewModel.associatedCategoryGuid)
186 {
187 InsertBlackboardRow(addItemToCategoryAction.itemToAdd, addItemToCategoryAction.indexToAddItemAt);
188 }
189 else
190 {
191 // If the added input has been added to a category other than this one, and it used to belong to this category,
192 // Then cleanup the controller and view that used to represent that input
193 foreach (var key in m_BlackboardItemControllers.Keys)
194 {
195 var blackboardItemController = m_BlackboardItemControllers[key];
196 if (blackboardItemController.Model == addItemToCategoryAction.itemToAdd)
197 {
198 RemoveBlackboardRow(addItemToCategoryAction.itemToAdd);
199 break;
200 }
201 }
202 }
203 break;
204
205 case DeleteCategoryAction deleteCategoryAction:
206 if (deleteCategoryAction.categoriesToRemoveGuids.Contains(ViewModel.associatedCategoryGuid))
207 {
208 this.Dispose();
209 return;
210 }
211
212 break;
213
214 case ChangeCategoryIsExpandedAction changeIsExpandedAction:
215 if (changeIsExpandedAction.categoryGuids.Contains(ViewModel.associatedCategoryGuid))
216 {
217 ViewModel.isExpanded = changeIsExpandedAction.isExpanded;
218 m_BlackboardCategoryView.TryDoFoldout(changeIsExpandedAction.isExpanded);
219 }
220 break;
221
222 case ChangeCategoryNameAction changeCategoryNameAction:
223 if (changeCategoryNameAction.categoryGuid == ViewModel.associatedCategoryGuid)
224 {
225 ViewModel.name = Model.name;
226 m_BlackboardCategoryView.title = ViewModel.name;
227 }
228 break;
229 }

243 // Creates controller, view and view model for a blackboard item and adds the view to the specified index in the category
244 // By default adds it to the end of the list if no insertionIndex specified
245 internal GGBlackboardRow InsertBlackboardRow(GeometryInput geometryInput, int insertionIndex = -1)
246 {
247 var geometryInputViewModel = new GeometryInputViewModel()
248 {
249 model = geometryInput,
250 parentView = blackboardCategoryView,
251 };
252 var blackboardItemController = new GeometryInputViewController(geometryInput, geometryInputViewModel, DataStore);
253
254 m_BlackboardItemControllers.TryGetValue(geometryInput.objectId, out var existingItemController);
255 if (existingItemController == null)
256 {
257 m_BlackboardItemControllers.Add(geometryInput.objectId, blackboardItemController);
258 // If no index specified, or if trying to insert at last index, add to end of category
259 if (insertionIndex == -1 || insertionIndex == m_BlackboardItemControllers.Count() - 1)
260 blackboardCategoryView.Add(blackboardItemController.BlackboardItemView);
261 else
262 blackboardCategoryView.Insert(insertionIndex, blackboardItemController.BlackboardItemView);
263
264 blackboardCategoryView.MarkDirtyRepaint();
265
266 return blackboardItemController.BlackboardItemView;
267 }
268 else
269 {
270 throw new AssertionException("Tried to add blackboard item that already exists to category.", null);
271 }
272 }
