292 m_UnregisterAll += () => field.UnregisterValueChangedCallback(eventCallback);

296 row.styleSheets.Add(Resources.Load<StyleSheet>("Styles/PropertyRow"));

533 void UpdateGeometryPortsForSlots(bool inputSlots, List<GeometrySlot> allSlots, GeometryPort[] slotGeometryPorts)
534 {
535 VisualElement portContainer = inputSlots ? inputContainer : outputContainer;
536 var existingPorts = portContainer.Query<GeometryPort>().ToList();
537 foreach (GeometryPort geometryPort in existingPorts)
538 {
539 var currentSlotId = geometryPort.slot.id;
540 int newSlotIndex = allSlots.FindIndex(s => s.id == currentSlotId);
541 if (newSlotIndex < 0)
542 {
543 // slot doesn't exist anymore, remove it
544 if (inputSlots)
545 portContainer.Remove(geometryPort.parent); // remove parent (includes the InputView)
546 else
547 portContainer.Remove(geometryPort);
548 }
549 else
550 {
551 var newSlot = allSlots[newSlotIndex];
552 slotGeometryPorts[newSlotIndex] = geometryPort;
553
554 // these should probably be in an UpdateShaderPort(geometryPort, newSlot) function
555 geometryPort.slot = newSlot;
556 geometryPort.portName = newSlot.displayName;
557
558 if (inputSlots) // input slots also have to update the InputView
559 UpdatePortInputView(geometryPort);
560 }
561 }

564 public void OnModified(ModificationScope scope)
565 {
566 UpdateTitle();
567 SetActive(node.isActive);
568 if (node.hasPreview)
569 UpdatePreviewExpandedState(node.previewExpanded);
570
571 base.expanded = node.drawState.expanded;
572
573 switch (scope)
574 {
575 // Update slots to match node modification
576 case ModificationScope.Topological:
577 {
578 var slots = node.GetSlots<GeometrySlot>().ToList();
579 // going to record the corresponding ShaderPort to each slot, so we can order them later
580 GeometryPort[] slotGeometryPorts = new GeometryPort[slots.Count];
581
582 // update existing input and output ports
583 UpdateGeometryPortsForSlots(true, slots, slotGeometryPorts);
584 UpdateGeometryPortsForSlots(false, slots, slotGeometryPorts);
585
586 // check if there are any new slots that must create new ports
587 for (int i = 0; i < slots.Count; i++)
588 {
589 if (slotGeometryPorts[i] == null)
590 slotGeometryPorts[i] = AddGeometryPortForSlot(slots[i]);
591 }
592
593 // make sure they are in the right order
594 // by bringing each port to front in declaration order
595 // note that this sorts input and output containers at the same time
596 foreach (var geometryPort in slotGeometryPorts)
597 {
598 if (geometryPort != null)
599 {
600 if (geometryPort.slot.isInputSlot)
601 geometryPort.parent.BringToFront();
602 else
603 geometryPort.BringToFront();
604 }
605 }
606
607 break;
608 }
609 }
610
611 RefreshExpandedState(); // Necessary b/c we can't override enough Node.cs functions to update only what's needed
612
613 foreach (var listener in m_ControlItems.Children().OfType<INodeModificationListener>())
614 {
615 if (listener != null)
616 listener.OnNodeModified(scope);
617 }
618 }

620 GeometryPort AddGeometryPortForSlot(GeometrySlot slot)
621 {
622 if (slot.hidden)
623 return null;
624
625 GeometryPort port = GeometryPort.Create(slot, m_ConnectorListener);
626 if (slot.isOutputSlot)
627 {
628 outputContainer.Add(port);
629 }
630 else
631 {
632 var portContainer = new VisualElement();
633 portContainer.style.flexDirection = FlexDirection.Row;
634 var portInputView = new PortInputView(slot) { style = { position = Position.Absolute } };
635 portContainer.Add(portInputView);
636 portContainer.Add(port);
637 inputContainer.Add(portContainer);
638
639 // Update active state
640 Debug.Log("AddGeometryPortForSlot node.isActive: " + node.isActive);
641 if (node.isActive)
642 {
643 portInputView.RemoveFromClassList("disabled");
644 }
645 else
646 {
647 portInputView.AddToClassList("disabled");
648 }
649 }
650 port.OnDisconnect = OnEdgeDisconnected;
651
652 return port;
653 }
