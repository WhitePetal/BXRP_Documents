489 void BuildHLSLDeclarationOverrideFields(PropertySheet propertySheet, AbstractGeometryProperty property)
490 {
491 if (isSubGraph)
492 return;
493
494 var hlslDecls = Enum.GetValues(typeof(HLSLDeclaration));
495 var allowedDecls = new List<HLSLDeclaration>();
496
497 bool anyAllowed = false;
498 for (int i = 0; i < hlslDecls.Length; i++)
499 {
500 HLSLDeclaration decl = (HLSLDeclaration)hlslDecls.GetValue(i);
501 var allowed = property.AllowHLSLDeclaration(decl);
502 anyAllowed = anyAllowed || allowed;
503 if (allowed)
504 allowedDecls.Add(decl);
505 }
506
507 const string tooltip = "Indicate where the property is expected to be changed.";
508 if (anyAllowed)
509 {
510 var propRow = new PropertyRow(PropertyDrawerUtils.CreateLabel("Scope", 0));
511 propRow.tooltip = tooltip;
512 var popupField = new PopupField<HLSLDeclaration>(
513 allowedDecls,
514 property.GetDefaultHLSLDeclaration(),
515 (h => allHLSLDeclarationStrings[(int)h]),
516 (h => allHLSLDeclarationStrings[(int)h]));
517
518 popupField.RegisterValueChangedCallback(
519 evt =>
520 {
521 this._preChangeValueCallback("Change Scope");
522 if (property.hlslDeclarationOverride == evt.newValue)
523 return;
524 property.hlslDeclarationOverride = evt.newValue;
525 property.overrideHLSLDeclaration = true;
526 UpdateEnableState();
527 this._exposedFieldChangedCallback(evt.newValue != HLSLDeclaration.Global);
528 property.generatePropertyBlock = evt.newValue != HLSLDeclaration.Global;
529 this._postChangeValueCallback(true, ModificationScope.Graph);
530 });
531
532 propRow.Add(popupField);
533 propertySheet.Add(propRow);
534 }

1113 var simpleReferenceName = Regex.Replace(displayName.ToUpper(), m_ReferenceNameDisallowedPattern, "_");
1114 if (referenceName != simpleReferenceName)
1115 displayName = referenceName;
