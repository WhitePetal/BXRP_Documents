171 public void Init(bool forceRefresh)
172 {
173 lightsDict = new Dictionary<GameObject, LightData>();
174 if (uniqueLights == null) uniqueLights = new List<GameObject>();
175 if (lights == null || uniqueLights == null)
176 {
177 forceRefresh = true;
178 }
179 else if (lights.Count != uniqueLights.Count)
180 {
181 forceRefresh = true;
182 }
183
184 if (forceRefresh)
185 {
186 lights = new List<LightData>();
187 uniqueLights = new List<GameObject>();
188 }
189 else
190 {
191 for(int i=0; i<lights.Count; i++)
192 {
193 if (uniqueLights[i] != null)
194 {
195 lightsDict[uniqueLights[i]] = lights[i];
196 }
197 }
198 }
199 }
200
201 public void StoreLight(GameObject u, LightData light)
202 {
203 if (uniqueLights == null) uniqueLights = new List<GameObject>();
204 if (lights == null) lights = new List<LightData>();
205 light.UID = uniqueLights.Count + 1;
206 uniqueLights.Add(u);
207 lights.Add(light);
208 lightsDict[u] = light;
209 }
