31 public static Func<LightCookieMapping, LightCookieMapping, int> s_CompareByCookieSize = (LightCookieMapping a, LightCookieMapping b) =>
32 {
33 var alc = a.light.cookie;
34 var blc = b.light.cookie;
35 int a2 = alc.width * alc.height;
36 int b2 = blc.width * blc.height;
37 int d = b2 - a2;
38 if (d == 0)
39 {
40 // 若按尺寸无法排序则按纹理id排序
41 int ai = alc.GetInstanceID();
42 int bi = blc.GetInstanceID();
43 return ai - bi;
44 }
45 return d;
46 };
47
48 public static Func<LightCookieMapping, LightCookieMapping, int> s_CompareByBufferIndex = (LightCookieMapping a, LightCookieMapping b) =>
49 {
50 return a.lightIndex - b.lightIndex;
51 };

412 private int FilterAndValidateOtherLights(BXLightsBase lights, LightCookieMapping[] validLightMappings)
413 {
414 int validLightCount = 0;
415
416 int otherLightCount = lights.importedOtherLightCount;
417 for (int i = 0; i < otherLightCount; ++i)
418 {
419 ref var visLight = ref lights.otherLights.UnsafeElementAtMutable(i);
420 Light light = visLight.light;
421
422 if (light.cookie == null) continue;
423
424 var lightType = visLight.lightType;
425 if(!(lightType == LightType.Spot || lightType == LightType.Point))
426 {
427 Debug.LogWarning($"Additional {lightType.ToString()} light called '{light.name}' has a light cookie which will not be visible.", light);
428 continue;
429 }
430
431 LightCookieMapping lp;
432 lp.lightIndex = i;
433 lp.light = light;
434
435 if (lp.lightIndex >= validLightMappings.Length || validLightCount + 1 >= validLightMappings.Length)
436 {
437 // TODO: Better error system
438 if (otherLightCount > maxCookieOtherLightCount &&
439 Time.frameCount - m_PrevWarnFrame > 60 * 60) // warn throttling: ~60 FPS * 60 secs ~= 1 min
440 {
441 m_PrevWarnFrame = Time.frameCount;
442 Debug.LogWarning($"Max light cookies ({validLightMappings.Length.ToString()}) reached. Some visible lights ({(otherLightCount - i - 1).ToString()}) might skip light cookie rendering.");
443 }
444
445 // Always break, buffer full.
446 break;
447 }
448
449 validLightMappings[validLightCount++] = lp;
450 }
451
452 return validLightCount;
453 }

455 private int UpdateOtherLightsAtlas(CommandBuffer cmd, ref WorkSlice<LightCookieMapping> validLightMappings, Vector4[] textureAtlasUVRects)
456 {
457 // 按照cookied尺寸排序，可以提高图集分配效率 和 在下面的Cookie打包时的去重
458 validLightMappings.Sort(LightCookieMapping.s_CompareByCookieSize);
459
460 uint cookieRequestPixelCount = ComputeCookieRequestPixelCount(ref validLightMappings);
461 var atlasSize = m_OtherLightsCookieAtlas.AtlasTexture.referenceSize;
462 float requestAtlasRatio = cookieRequestPixelCount / (float)(atlasSize.x * atlasSize.y);
463 int cookieSizeDivisorApprox = ApproximateCookieSizeDivisor(requestAtlasRatio);
464
465 // 尝试恢复分辨率 并把 cookie 缩放回原来大小
466 // 如果 cookie 能够填满 并且
467 // 如果我们请求的像素比上次少，我们就找到了正确的除数(防止每帧都重新尝试)
468 if(cookieSizeDivisorApprox < m_CookieSizeDivisor &&
469 cookieRequestPixelCount < m_PrevCookieRequestPixelCount)
470 {
471 m_OtherLightsCookieAtlas.ResetAllocator();
472 m_CookieSizeDivisor = cookieSizeDivisorApprox;
473 }
474
475 int uvRectCount = 0;
476 while(uvRectCount <= 0)
477 {
478 uvRectCount = FetchUVRects(cmd, ref validLightMappings, textureAtlasUVRects, m_CookieSizeDivisor);
479
480 if(uvRectCount <= 0)
481 {
482 m_OtherLightsCookieAtlas.ResetAllocator();
483
484 m_CookieSizeDivisor = Mathf.Max(m_CookieSizeDivisor + 1, cookieSizeDivisorApprox);
485 m_PrevCookieRequestPixelCount = cookieRequestPixelCount;
486 }
487 }
488
489 return uvRectCount;
490 }

492 private uint ComputeCookieRequestPixelCount(ref WorkSlice<LightCookieMapping> validLightMappings)
493 {
494 uint requestPixelCount = 0;
495 int prevCookieID = 0;
496 for(int i = 0; i < validLightMappings.length; i++)
497 {
498 var lcm = validLightMappings[i];
499 Texture cookie = lcm.light.cookie;
500 int cookieID = cookie.GetInstanceID();
501
502 // 去重Cookie
503 // 依赖于排序方式
504 if(cookieID == prevCookieID)
505 {
506 continue;
507 }
508 prevCookieID = cookieID;
509
510 int pixelCookieCount = cookie.width * cookie.height;
511 requestPixelCount += (uint)pixelCookieCount;
512 }
513
514 return requestPixelCount;
515 }

517 private int ApproximateCookieSizeDivisor(float requestAtlasRatio)
518 {
519 // (Edge / N)^2 == 1/N^2 of area
520 // Ratio/N^2 == 1, sqrt(Ratio) == N, for "1:1" ratio
521 return (int)Mathf.Max(Mathf.Ceil(Mathf.Sqrt(requestAtlasRatio)), 1);
522 }

524 private int FetchUVRects(CommandBuffer cmd, ref WorkSlice<LightCookieMapping> validLightMappings, Vector4[] textureAtlasUVRects, int cookieSizeDivisor)
525 {
526 int uvRectCount = 0;
527 for(int i = 0; i < validLightMappings.length; i++)
528 {
529 var lcm = validLightMappings[i];
530
531 Light light = lcm.light;
532 Texture cookie = light.cookie;
533
534 Vector4 uvScaleOffset = Vector4.zero;
535 if(cookie.dimension == TextureDimension.Cube)
536 {
537 Assert.IsTrue(light.type == LightType.Point);
538 uvScaleOffset = FetchCube(cmd, cookie, cookieSizeDivisor);
539 }
540 else
541 {
542 Assert.IsTrue(light.type == LightType.Spot, "Light type needs 2D texture!");
543 uvScaleOffset = Fetch2D(cmd, cookie, cookieSizeDivisor);
544 }
545
546 bool isCached = uvScaleOffset != Vector4.zero;
547 if (!isCached)
548 {
549 if(cookieSizeDivisor > k_MaxCookieSizeDivisor)
550 {
551 Debug.LogWarning($"Light cookies atlas is extremely full! Some of the light cookies were discarded. Increase light cookie atlas space or reduce the amount of unique light cookies.");
552 // complete fail
553 return uvRectCount;
554 }
555
556 // 未能给每个cookie分配 uv rect
557 return 0;
558 }
559
560 if (!SystemInfo.graphicsUVStartsAtTop)
561 uvScaleOffset.w = 1f - uvScaleOffset.w - uvScaleOffset.y;
562
563 textureAtlasUVRects[uvRectCount++] = uvScaleOffset;
564 }
565
566 return uvRectCount;
567 }

569 private Vector4 FetchCube(CommandBuffer cmd, Texture cookie, int cookieSizeDivisor = 1)
570 {
571 Assert.IsTrue(cookie != null);
572 Assert.IsTrue(cookie.dimension == TextureDimension.Cube);
573
574 Vector4 uvScaleOffset = Vector4.zero;
575
576 int scaledOctCookieSize = Mathf.Max(ComputeOctahedralCookieSize(cookie) / cookieSizeDivisor, 4);
577
578 bool isCached = m_OtherLightsCookieAtlas.IsCached(out uvScaleOffset, cookie);
579 if (isCached)
580 {
581 m_OtherLightsCookieAtlas.UpdateTexture(cmd, cookie, ref uvScaleOffset);
582 }
583 else
584 {
585 m_OtherLightsCookieAtlas.AllocateTexture(cmd, ref uvScaleOffset, cookie, scaledOctCookieSize, scaledOctCookieSize);
586 }
587
588 // 图集中的cookie大小可能与 cookie texture 的 尺寸不一致
589 // 根据图集调整UVRect
590 var scaledCookieSize = Vector2.one * scaledOctCookieSize;
591 AdjustUVRect(ref uvScaleOffset, cookie, ref scaledCookieSize);
592 return uvScaleOffset;
593 }

595 private int ComputeOctahedralCookieSize(Texture cookie)
596 {
597 // Map 6*W*H pixels into 2W*2H pixels, so 4/6 ratio or 66% of cube pixels
598 int octCookieSize = Math.Max(cookie.width, cookie.height);
599 if (commonSettings.cookieAtlas.isPow2)
600 octCookieSize = octCookieSize * Mathf.NextPowerOfTwo((int)commonSettings.cubeOctahedraSizeScale);
601 else
602 octCookieSize = (int)(octCookieSize * commonSettings.cubeOctahedraSizeScale + 0.5f);
603 return octCookieSize;
604 }

606 private void AdjustUVRect(ref Vector4 uvScaleOffset, Texture cookie, ref Vector2 cookieSize)
607 {
608 if(uvScaleOffset != Vector4.one)
609 {
610 // 缩小0.5px以夹紧双线性采样以排除图集邻居（no padding）
611 ShrinkUVRect(ref uvScaleOffset, 0.5f, ref cookieSize);
612 }

615 private void ShrinkUVRect(ref Vector4 uvScaleOffset, float amountPixels, ref Vector2 cookieSize)
616 {
617 var shrinkOffset = Vector2.one * amountPixels / cookieSize;
618 var shrinkScale = (cookieSize - Vector2.one * (amountPixels * 2)) / cookieSize;
619 uvScaleOffset.z += uvScaleOffset.x * shrinkOffset.x;
620 uvScaleOffset.w += uvScaleOffset.y * shrinkOffset.y;
621 uvScaleOffset.x *= shrinkScale.x;
622 uvScaleOffset.y *= shrinkScale.y;
623 }
