620 private static int Partition<T>(Span<T> data, int left, int right) where T : IComparable<T>, new()
621 {
622 var pivot = data[left];
623
624 --left;
625 ++right;
626 while (true)
627 {
628 var c = 0;
629 var lvalue = default(T);
630 do
631 {
632 ++left;
633 lvalue = data[left];
634 c = lvalue.CompareTo(pivot);
635 }
636 while (c < 0);
637
638 var rvalue = default(T);
639 do
640 {
641 --right;
642 rvalue = data[right];
643 c = rvalue.CompareTo(pivot);
644 }
645 while (c > 0);
646
647 if (left < right)
648 {
649 data[right] = lvalue;
650 data[left] = rvalue;
651 }
652 else
653 {
654 return right;
655 }
