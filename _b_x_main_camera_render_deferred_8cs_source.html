163 Vector4 lb = forward - right - up;
164 Vector4 lu = forward - right + up;
165 Vector4 rb = forward + right - up;
166 Vector4 ru = forward + right + up;
167 viewPortRays.SetRow(0, lb);
168 viewPortRays.SetRow(1, lu);
169 viewPortRays.SetRow(2, rb);
