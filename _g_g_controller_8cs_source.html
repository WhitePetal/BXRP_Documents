96 void NotifyEventHandler(IGGControlledElement eventHandler, IGraphDataAction changeAction)
97 {
98 GGControllerChangedEvent e = new GGControllerChangedEvent();
99 e.controller = this;
100 e.target = eventHandler;
101 e.change = changeAction;
102 eventHandler.OnControllerChanged(ref e);
103 if (e.isPropagationStopped)
104 return;
105 if (eventHandler is VisualElement)
106 {
107 var element = eventHandler as VisualElement;
108 eventHandler = element.GetFirstOfType<IGGControlledElement>();
109 while (eventHandler != null)
110 {
111 eventHandler.OnControllerChanged(ref e);
112 if (e.isPropagationStopped)
113 break;
114 eventHandler = (eventHandler as VisualElement).GetFirstAncestorOfType<IGGControlledElement>();
115 }
116 }
