123 private void Render(CommandBuffer commandBuffer)
124 {
125 if (isViewCamera) return;
126 commandBuffer.BeginSample(SampleName);
127 if (!isReadbacking[willReadBackIndex])
128 {
129 commandBuffer.SetComputeTextureParam(cs, 0, BXShaderPropertyIDs._EncodeDepthBuffer_ID, BXShaderPropertyIDs._EncodeDepthBuffer_TargetID);
130 commandBuffer.SetComputeTextureParam(cs, 0, BaseShaderProperties._HiZMap_ID, hizBuffers[willReadBackIndex]);
131 int2 screenSize = screenSizes[willReadBackIndex];
132 commandBuffer.DispatchCompute(cs, 0, Mathf.CeilToInt(screenSize.x / 8), Mathf.CeilToInt(screenSize.y / 8), 1);
133
134 int2 texSize = texSizes[willReadBackIndex];
135 float2 mipSize = math.float2(screenSize.x >> 3, texSize.y);
136 float4 mipOffset = math.float4(0, 0, mipSize.x, mipSize.y);
137 ref var mipSizes = ref mipSizeses[willReadBackIndex];
138 ref int mipCount = ref mipCounts.UnsafeElementAtMutable(willReadBackIndex);
139 mipSizes[mipCount++] = mipOffset;
140 while (mipSize.x > 1 && mipSize.y > 1)
141 {
142 commandBuffer.SetComputeVectorParam(cs, BaseShaderProperties._MipOffset_ID, mipOffset);
143 commandBuffer.SetComputeTextureParam(cs, 1, BaseShaderProperties._HiZMap_ID, hizBuffers[willReadBackIndex]);
144 commandBuffer.DispatchCompute(cs, 1, Mathf.CeilToInt(mipSize.x / 8f), Mathf.CeilToInt(mipSize.y / 8f), 1);
145 mipOffset += math.float4(mipSize.x, 0, 0, 0);
146 mipSize *= 0.5f;
147 mipSizes[mipCount++] = math.float4(mipOffset.xy, mipSize.xy);
148 if (mipSize.x <= 1 || mipSize.y <= 1) break;
149 mipOffset += math.float4(0, mipSize.y, 0, 0);
150 mipSize *= 0.5f;
151 mipSizes[mipCount++] = math.float4(mipOffset.xy, mipSize.xy);
152 if (mipSize.x <= 1 || mipSize.y <= 1) break;
153 mipOffset += math.float4(mipSize.x, 0, 0, 0);
154 mipSize *= 0.5f;
155 mipOffset.z = mipSize.x;
156 mipOffset.w = mipSize.y;
157 mipSizes[mipCount++] = math.float4(mipOffset.xy, mipSize.xy);
158 }
159
160 hizReadbackDatas[willReadBackIndex] = new NativeArray<float>(texSize.x * texSize.y, Allocator.Persistent);
161 isDone[willReadBackIndex] = false;
162 commandBuffer.RequestAsyncReadbackIntoNativeArray<float>(ref hizReadbackDatas[willReadBackIndex], hizBuffers[willReadBackIndex], readBackActions[willReadBackIndex]);
163 isReadbacking[willReadBackIndex] = true;
164 readBackStartTime[willReadBackIndex] = Time.frameCount;
165 willReadBackIndex = willReadBackIndex == 2 ? 0 : (willReadBackIndex + 1);
166 }
167 commandBuffer.EndSample(SampleName);
168 }

185 private void CompleteCull(BXMainCameraRenderBase mainRender)
186 {
187 isViewCamera = mainRender.camera.cameraType == CameraType.SceneView;
188 if (isViewCamera || validReadBackIndex == -1 || isReadbacking[validReadBackIndex] || isDone[validReadBackIndex]) return;
189 HizCullJob cullJob = new HizCullJob()
190 {
191 hizData = hizReadbackDatas[validReadBackIndex],
192 cullingObjectDatas = cullingObjects[validReadBackIndex],
193 projectionMatrix = projectionMatrixs[validReadBackIndex],
194 screenSize = screenSizes[validReadBackIndex],
195 texSize = texSizes[validReadBackIndex],
196 mipCount = mipCounts[validReadBackIndex],
197 mipSizes = mipSizeses[validReadBackIndex]
198 };
199 int count = cullingObjectCounts[validReadBackIndex];
200 JobHandle handle = cullJob.Schedule(count, 8);
201 handle.Complete();
202 hizReadbackDatas[validReadBackIndex].Dispose();
203 for(int i = 0; i < count; ++i)
204 {
205 ref var data = ref cullingObjects[validReadBackIndex].UnsafeElementAt(i);
206 data.renderer.renderingLayerMask = data.renderLayerMask;
207 }
208 --compeleteReadBackCount;
209 isDone[validReadBackIndex] = true;
210 }

320 private float3 GetNDCPos(float3 pos)
321 {
322 float4 p = projectionMatrix * math.float4(pos.x, pos.y, pos.z, 1f);
323 p.xyz /= p.w;
324 p.xy = p.xy * 0.5f + 0.5f;
325 //p.y = 1.0f - p.y;
326 return p.xyz;
327 }

329 public void Execute(int i)
330 {
331 unsafe
332 {
333 ref var data = ref UnsafeUtility.ArrayElementAsRef<CullingObjectData>(cullingObjectDatas.GetUnsafePtr(), i);
334 float3 center = data.boundCenter;
335 float3 size = data.boundSize;
336
337 float3 p0 = center + size;
338 float3 p1 = center - size;
339 float3 p2 = math.float3(p0.xy, p1.z); // + + -
340 float3 p3 = math.float3(p0.x, p1.yz); // + - -
341 float3 p4 = math.float3(p1.xy, p0.z); // - - +
342 float3 p5 = math.float3(p1.x, p0.y, p1.z);// - + -
343 float3 p6 = math.float3(p0.x, p1.y, p0.z);// + - +
344 float3 p7 = math.float3(p1.x, p0.yz); // - + +
345
346 p0 = GetNDCPos(p0);
347 p1 = GetNDCPos(p1);
348 p2 = GetNDCPos(p2);
349 p3 = GetNDCPos(p3);
350 p4 = GetNDCPos(p4);
351 p5 = GetNDCPos(p5);
352 p6 = GetNDCPos(p6);
353 p7 = GetNDCPos(p7);
354
355 float3 aabbMin = math.min(p0, math.min(p1, math.min(p2, math.min(p3, math.min(p4, math.min(p5, math.min(p6, p7)))))));
356 float3 aabbMax = math.max(p0, math.max(p1, math.max(p2, math.max(p3, math.max(p4, math.max(p5, math.max(p6, p7)))))));
357
358 if (math.any(aabbMax < math.float3(0f)))
359 {
360 data.renderLayerMask = 0;
361 return;
362 }
363 if (math.any(aabbMin > math.float3(1f)))
364 {
365 data.renderLayerMask = 0;
366 return;
367 }
368
369 float2 ndcSize = (aabbMax.xy - aabbMin.xy) * screenSize;
370 float radius = math.max(ndcSize.x, ndcSize.y);
371 int mip = (int)math.floor(math.log2(radius));
372 mip = math.clamp(mip - 2, 0, mipCount - 1);
373 float4 mipSize = mipSizes[mip];
374
375 int2 minPx = (int2)math.ceil(aabbMin.xy * mipSize.zw + mipSize.xy);
376 int2 maxPx = (int2)math.ceil(aabbMax.xy * mipSize.zw + mipSize.xy);
377
378 int maxIndex = texSize.x * texSize.y - 1;
379 int index0 = math.clamp(minPx.x + minPx.y * texSize.x, 0, maxIndex);
380 int index1 = math.clamp(minPx.x + maxPx.y * texSize.x, 0, maxIndex);
381 int index2 = math.clamp(maxPx.x + maxPx.y * texSize.x, 0, maxIndex);
382 int index3 = math.clamp(maxPx.x + minPx.y * texSize.x, 0, maxIndex);
383
384
385 float d0 = hizData[index0];
386 float d1 = hizData[index1];
387 float d2 = hizData[index2];
388 float d3 = hizData[index3];
389 float minD = math.min(d0, math.min(d1, math.min(d2, d3)));
390 if (minD > aabbMax.z)
391 {
392 data.renderLayerMask = 0;
393 }
394 else
395 {
396 data.renderLayerMask = 1;
397 }
398 }
399 }

402 private void Cull(int index)
403 {
404 isReadbacking[index] = false;
405 int startTime = readBackStartTime[index];
406 // 有更晚开启的回读任务提前完成了，那就丢弃这个更早开始的任务
407 // 有多个回读任务同时完成了，则使用最早被调用的那个任务
408 if(startTime < lastReadBackStartTime || compeleteReadBackCount > 0)
409 {
410 hizReadbackDatas[index].Dispose();
411 return;
412 }
413 ++compeleteReadBackCount;
414 validReadBackIndex = index;
415 lastReadBackStartTime = startTime;
416 }
