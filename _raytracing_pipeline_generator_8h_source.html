26/*
27Contacts for feedback:
28- pgautron@nvidia.com (Pascal Gautron)
29- mlefrancois@nvidia.com (Martin-Karl Lefrancois)
30
31The raytracing pipeline combines the raytracing shaders into a state object,
32that can be thought of as an executable GPU program. For that, it requires the
33shaders compiled as DXIL libraries, where each library exports symbols in a way
34similar to DLLs. Those symbols are then used to refer to these shaders libraries
35when creating hit groups, associating the shaders to their root signatures and
36declaring the steps of the pipeline. All the calls to this helper class can be
37done in arbitrary order. Some basic sanity checks are also performed when
38compiling in debug mode.
39
40Simple usage of this class:
41
42pipeline.AddLibrary(m_rayGenLibrary.Get(), {L"RayGen"});
43pipeline.AddLibrary(m_missLibrary.Get(), {L"Miss"});
44pipeline.AddLibrary(m_hitLibrary.Get(), {L"ClosestHit"});
45
46pipeline.AddHitGroup(L"HitGroup", L"ClosestHit");
47
48pipeline.AddRootSignatureAssociation(m_rayGenSignature.Get(), {L"RayGen"});
49pipeline.AddRootSignatureAssociation(m_missSignature.Get(), {L"Miss"});
50pipeline.AddRootSignatureAssociation(m_hitSignature.Get(), {L"HitGroup"});
51
52pipeline.SetMaxPayloadSize(4 * sizeof(float)); // RGB + distance
53
54pipeline.SetMaxAttributeSize(2 * sizeof(float)); // barycentric coordinates
55
56pipeline.SetMaxRecursionDepth(1);
57
58rtStateObject = pipeline.Generate();
59
60*/
