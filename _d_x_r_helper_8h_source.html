62// Compile a HLSL file into a DXIL library
63//
64inline IDxcBlob* CompileShaderLibrary(LPCWSTR fileName)
65{
66 static IDxcCompiler* pCompiler = nullptr;
67 static IDxcLibrary* pLibrary = nullptr;
68 static IDxcIncludeHandler* dxcIncludeHandler;
69
70 HRESULT hr;
71
72 // Initialize the DXC compiler and compiler helper
73 if (!pCompiler)
74 {
75 ThrowIfFaild(DxcCreateInstance(CLSID_DxcCompiler, __uuidof(IDxcCompiler), (void **)&pCompiler));
76 ThrowIfFaild(DxcCreateInstance(CLSID_DxcLibrary, __uuidof(IDxcLibrary), (void **)&pLibrary));
77 ThrowIfFaild(pLibrary->CreateIncludeHandler(&dxcIncludeHandler));
78 }
79 // Open and read the file
80 std::ifstream shaderFile(fileName);
81 if (shaderFile.good() == false)
82 {
83 throw std::logic_error("Cannot find shader file");
84 }
85 std::stringstream strStream;
86 strStream << shaderFile.rdbuf();
87 std::string sShader = strStream.str();
88
89 // Create blob from the string
90 IDxcBlobEncoding* pTextBlob;
91 ThrowIfFaild(pLibrary->CreateBlobWithEncodingFromPinned(
92 (LPBYTE)sShader.c_str(), (uint32_t)sShader.size(), 0, &pTextBlob));
93
94 // Compile
95 IDxcOperationResult* pResult;
96 ThrowIfFaild(pCompiler->Compile(pTextBlob, fileName, L"", L"lib_6_3", nullptr, 0, nullptr, 0,
97 dxcIncludeHandler, &pResult));
98
99 // Verify the result
100 HRESULT resultCode;
101 ThrowIfFaild(pResult->GetStatus(&resultCode));
102 if (FAILED(resultCode))
103 {
104 IDxcBlobEncoding* pError;
105 hr = pResult->GetErrorBuffer(&pError);
106 if (FAILED(hr))
107 {
108 throw std::logic_error("Failed to get shader compiler error");
109 }
110
111 // Convert error blob to a string
112 std::vector<char> infoLog(pError->GetBufferSize() + 1);
113 memcpy(infoLog.data(), pError->GetBufferPointer(), pError->GetBufferSize());
114 infoLog[pError->GetBufferSize()] = 0;
115
116 std::string errorMsg = "Shader Compiler Error:\n";
117 errorMsg.append(infoLog.data());
118
119 MessageBoxA(nullptr, errorMsg.c_str(), "Error!", MB_OK);
120 throw std::logic_error("Failed compile shader");
121 }
122
123 IDxcBlob* pBlob;
124 ThrowIfFaild(pResult->GetResult(&pBlob));
125 return pBlob;
126}

148template <class Vertex>
149void GenerateMengerSponge(int32_t level, float probability, std::vector<Vertex>& outputVertices,
150 std::vector<UINT>& outputIndices)
151{
152 struct Cube
153 {
154 Cube(const DirectX::XMVECTOR& tlf, float s) : m_topLeftFront(tlf), m_size(s)
155 {
156 }
157 DirectX::XMVECTOR m_topLeftFront;
158 float m_size;
159
160 void enqueueQuad(std::vector<Vertex>& vertices, std::vector<UINT>& indices,
161 const DirectX::XMVECTOR& bottomLeft4, const DirectX::XMVECTOR& dx, const DirectX::XMVECTOR& dy, bool flip)
162 {
163 UINT currentIndex = static_cast<UINT>(vertices.size());
164 DirectX::XMFLOAT3 bottomLeft(bottomLeft4.m128_f32);
165 DirectX::XMVECTOR normal = XMVector3Cross(XMVector3Normalize(dy), XMVector3Normalize(dx));
166 if (flip)
167 {
168 normal = -normal;
169
170 indices.push_back(currentIndex + 0);
171 indices.push_back(currentIndex + 2);
172 indices.push_back(currentIndex + 1);
173
174 indices.push_back(currentIndex + 3);
175 indices.push_back(currentIndex + 1);
176 indices.push_back(currentIndex + 2);
177 }
178 else
179 {
180
181 indices.push_back(currentIndex + 0);
182 indices.push_back(currentIndex + 1);
183 indices.push_back(currentIndex + 2);
184
185 indices.push_back(currentIndex + 2);
186 indices.push_back(currentIndex + 1);
187 indices.push_back(currentIndex + 3);
188 }
189
190 const DirectX::XMFLOAT4 n = {normal.m128_f32[0], normal.m128_f32[1], normal.m128_f32[2], 0.f};
191 vertices.push_back(
192 {{bottomLeft.x, bottomLeft.y, bottomLeft.z, 1.f}, n, {1.f, 0.f, 0.f, 1.f}});
193 vertices.push_back({{bottomLeft.x + dx.m128_f32[0], bottomLeft.y + dx.m128_f32[1],
194 bottomLeft.z + dx.m128_f32[2], 1.f},
195 n,
196 {0.5f, 1.f, 0.f, 1.f}});
197 vertices.push_back({{bottomLeft.x + dy.m128_f32[0], bottomLeft.y + dy.m128_f32[1],
198 bottomLeft.z + dy.m128_f32[2], 1.f},
199 n,
200 {0.5f, 0.f, 1.f, 1.f}});
201
202 vertices.push_back({{bottomLeft.x + dx.m128_f32[0] + dy.m128_f32[0],
203 bottomLeft.y + dx.m128_f32[1] + dy.m128_f32[1],
204 bottomLeft.z + dx.m128_f32[2] + dy.m128_f32[2], 1.f},
205 n,
206 {0.f, 1.f, 0.f, 1.f}});
207 }
208 void enqueueVertices(std::vector<Vertex>& vertices, std::vector<UINT>& indices)
209 {
210
211 DirectX::XMVECTOR current = m_topLeftFront;
212 enqueueQuad(vertices, indices, current, {m_size, 0, 0}, {0, m_size, 0}, false);
213 enqueueQuad(vertices, indices, current, {m_size, 0, 0}, {0, 0, m_size}, true);
214 enqueueQuad(vertices, indices, current, {0, m_size, 0}, {0, 0, m_size}, false);
215
216 current.m128_f32[0] += m_size;
217 current.m128_f32[1] += m_size;
218 current.m128_f32[2] += m_size;
219 enqueueQuad(vertices, indices, current, {-m_size, 0, 0}, {0, -m_size, 0}, true);
220 enqueueQuad(vertices, indices, current, {-m_size, 0, 0}, {0, 0, -m_size}, false);
221 enqueueQuad(vertices, indices, current, {0, -m_size, 0}, {0, 0, -m_size}, true);
222 }
223 void split(std::vector<Cube>& cubes)
224 {
225 float size = m_size / 3.f;
226 DirectX::XMVECTOR topLeftFront = m_topLeftFront;
227 for (int x = 0; x < 3; x++)
228 {
229 topLeftFront.m128_f32[0] = m_topLeftFront.m128_f32[0] + static_cast<float>(x) * size;
230 for (int y = 0; y < 3; y++)
231 {
232 if (x == 1 && y == 1)
233 continue;
234 topLeftFront.m128_f32[1] = m_topLeftFront.m128_f32[1] + static_cast<float>(y) * size;
235 for (int z = 0; z < 3; z++)
236 {
237 if (x == 1 && z == 1)
238 continue;
239 if (y == 1 && z == 1)
240 continue;
241
242 topLeftFront.m128_f32[2] = m_topLeftFront.m128_f32[2] + static_cast<float>(z) * size;
243 cubes.push_back({topLeftFront, size});
244 }
245 }
246 }
247 }
248
249 void splitProb(std::vector<Cube>& cubes, float prob)
250 {
251
252 float size = m_size / 3.f;
253 DirectX::XMVECTOR topLeftFront = m_topLeftFront;
254 for (int x = 0; x < 3; x++)
255 {
256 topLeftFront.m128_f32[0] = m_topLeftFront.m128_f32[0] + static_cast<float>(x) * size;
257 for (int y = 0; y < 3; y++)
258 {
259 topLeftFront.m128_f32[1] = m_topLeftFront.m128_f32[1] + static_cast<float>(y) * size;
260 for (int z = 0; z < 3; z++)
261 {
262 float sample = rand() / static_cast<float>(RAND_MAX);
263 if (sample > prob)
264 continue;
265 topLeftFront.m128_f32[2] = m_topLeftFront.m128_f32[2] + static_cast<float>(z) * size;
266 cubes.push_back({topLeftFront, size});
267 }
268 }
269 }
270 }
271 };
272
273 DirectX::XMVECTOR orig;
274 orig.m128_f32[0] = -0.5f;
275 orig.m128_f32[1] = -0.5f;
276 orig.m128_f32[2] = -0.5f;
277 orig.m128_f32[3] = 1.f;
278
279 Cube cube(orig, 1.f);
280
281 std::vector<Cube> cubes1 = {cube};
282 std::vector<Cube> cubes2 = {};
283
284 auto previous = &cubes1;
285 auto next = &cubes2;
286
287 for (int i = 0; i < level; i++)
288 {
289 for (Cube& c : *previous)
290 {
291 if (probability < 0.f)
292 c.split(*next);
293 else
294 c.splitProb(*next, 20.f / 27.f);
295 }
296 auto temp = previous;
297 previous = next;
298 next = temp;
299 next->clear();
300 }
301
302 outputVertices.reserve(24 * previous->size());
303 outputIndices.reserve(24 * previous->size());
304 for (Cube& c : *previous)
305 {
306 c.enqueueVertices(outputVertices, outputIndices);
307 }
308}
