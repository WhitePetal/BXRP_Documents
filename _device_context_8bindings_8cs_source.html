164 private uint nextFreeBufferId;

172 public void Dispose()
173 {
174 foreach(var entry in buffers)

176 UnityEngine.Debug.Assert(entry.Value.IsCreated, "A buffer was unexceptedly not created.");
177 entry.Value.Dispose();

181 public BufferID CreateBuffer(ulong count, ulong stride)
182 {
183 UnityEngine.Debug.Assert(count != 0, "Buffer element count cannot be zero.");
184 UnityEngine.Debug.Assert(stride != 0, "Buffer stride cannot be zero.");
185
186 var buffer = new NativeArray<byte>((int)(count * stride), Allocator.Persistent);
187 var idInteger = nextFreeBufferId++;
188 var id = new BufferID(idInteger);
189 buffers[id] = buffer;
190 return id;
191 }
192
193 public void DestroyBuffer(BufferID id)
194 {
195 UnityEngine.Debug.Assert(buffers.ContainsKey(id), "Invalid buffer ID given.");
196 buffers[id].Dispose();
197 buffers.Remove(id);
198 }
199
200 public void WriteBuffer<T>(BufferSlice<T> dst, NativeArray<T> src) where T : struct
201 {
202 UnityEngine.Debug.Assert(buffers.ContainsKey(dst.Id), "Invalid buffer ID given.");
203 var dstBuffer = buffers[dst.Id].Reinterpret<T>(1);
204 dstBuffer.GetSubArray((int)dst.Offset, dstBuffer.Length - (int)dst.Offset).CopyFrom(src);
205 }
206
207 public void ReadBuffer<T>(BufferSlice<T> src, NativeArray<T> dst) where T : struct
208 {
209 UnityEngine.Debug.Assert(buffers.ContainsKey(src.Id), "Invalid buffer ID given");
210 var srcBuffer = buffers[src.Id].Reinterpret<T>(1);
211 dst.CopyFrom(srcBuffer.GetSubArray((int)src.Offset, srcBuffer.Length - (int)src.Offset));
212 }
