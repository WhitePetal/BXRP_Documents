26 private void CollectLightDatas()
27 {
28 NativeArray<VisibleLight> visibleLights = cullingResults.visibleLights;
29 dirLightCount = 0;
30 clusterLightCount = 0;
31 stencilLightCount = 0;
32 otherLightCount = 0;
33 importedOtherLightCount = 0;
34
35 for (int visbileLightIndex = 0; visbileLightIndex < visibleLights.Length; ++visbileLightIndex)
36 {
37 if (dirLightCount >= maxDirLightCount && otherLightCount >= maxOtherLightCount) break;
38 ref var visibleLight = ref visibleLights.UnsafeElementAtMutable(visbileLightIndex);
39 LightBakingOutput lightBaking = visibleLight.light.bakingOutput;
40 if (lightBaking.lightmapBakeType == LightmapBakeType.Baked) continue;
41 switch (visibleLight.lightType)
42 {
43 case LightType.Directional:
44 if (dirLightCount < maxDirLightCount)
45 {
46 SetupDirectionalLight(dirLightCount++, visbileLightIndex, ref visibleLight);
47 }
48 break;
49 case LightType.Point:
50 if (stencilLightCount < maxStencilLightCount)
51 {
52 SetupStencilPointLight(otherLightCount++, visbileLightIndex, ref visibleLight);
53 ++stencilLightCount;
54 ++importedOtherLightCount;
55 }
56 else if (clusterLightCount < maxClusterLightCount)
57 {
58 SetupClusterPointLight(otherLightCount++, clusterLightCount++, visbileLightIndex, ref visibleLight);
59 }
60 break;
61 case LightType.Spot:
62 if (stencilLightCount < maxStencilLightCount)
63 {
64 SetupStencilSpotLight(otherLightCount++, visbileLightIndex, ref visibleLight);
65 ++stencilLightCount;
66 ++importedOtherLightCount;
67 }
68 else if (clusterLightCount < maxClusterLightCount)
69 {
70 SetupClusterSpotLight(otherLightCount++, clusterLightCount++, visbileLightIndex, ref visibleLight);
71 }
72 break;
73 }
74 }

151 private void SetupClusterSpotLight(int otherLightIndex, int clusterLightIndex, int visibleLightIndex, ref VisibleLight visibleLight)
152 {
153 Matrix4x4 localToWorld = visibleLight.localToWorldMatrix;
154 float range = visibleLight.range;
155 Vector4 lightSphere = localToWorld.GetColumn(3);
156 float threshold = range * range;
157 lightSphere.w = 1f / threshold;
158 threshold = lightSphere.w;
159
160 Vector4 lightDir = -localToWorld.GetColumn(2);
161 Vector4 lightRight = localToWorld.GetColumn(0);
162 Vector4 lightUp = localToWorld.GetColumn(1);
163 float outerRad = Mathf.Deg2Rad * 0.5f * visibleLight.spotAngle;
164 float outerCos = Mathf.Cos(outerRad);
165 float outerSin = Mathf.Sin(outerRad);
166 float angleRangeInv = 1f / Mathf.Max(1f - outerCos, 0.001f);
167
168 float sinRange = outerSin * range;
169 Vector4 upward = lightUp * sinRange;
170 Vector4 rtward = lightRight * sinRange;
171 Vector4 p0 = lightSphere;
172 Vector4 pf = lightSphere - lightDir * range;
173 Vector4 p1 = pf + upward + rtward;
174 Vector4 p2 = pf + upward - rtward;
175 Vector4 p3 = pf - upward + rtward;
176 Vector4 p4 = pf - upward - rtward;
177
178 Vector4 maxBound = Vector4.Max(p4, Vector4.Max(p3, Vector4.Max(p2, Vector4.Max(p1, p0))));
179 Vector4 minBound = Vector4.Min(p4, Vector4.Min(p3, Vector4.Min(p2, Vector4.Min(p1, p0))));
180
181 otherLightSpheres[otherLightIndex] = lightSphere;
182 clusterLightMaxBounds[clusterLightIndex] = maxBound;
183 clusterLightMinBounds[clusterLightIndex] = minBound;
184 otherLightDirections[otherLightIndex] = lightDir;
185 otherLightThresholds[otherLightIndex] = new Vector4(1f / (1f - threshold), threshold, angleRangeInv, -outerCos * angleRangeInv);
186 otherLightColors[otherLightIndex] = visibleLight.finalColor.gamma;
187 otherLights[otherLightIndex] = visibleLight;
188 }
