266 internal float ComputeVolume(in ProbeReferenceVolume.Volume touchupOBB)
267 {
268 if (shape == Shape.Box)
269 return touchupOBB.X.magnitude * touchupOBB.Y.magnitude * touchupOBB.Z.magnitude;
270 else
271 return (4.0f / 3.0f) * Mathf.PI * radius * radius * radius;
272 }

274 internal bool IntersectsVolume(in ProbeReferenceVolume.Volume touchupOBB, in Bounds touchupBounds, Bounds volumeBounds)
275 {
276 if (shape == Shape.Box)
277 return ProbeVolumePositioning.OBBAABBIntersect(touchupOBB, volumeBounds, touchupBounds);
278 else
279 return volumeBounds.SqrDistance(touchupBounds.center) < radius * radius;
280 }
281
282 internal bool ContainsPoint(in ProbeReferenceVolume.Volume touchupOBB, in Vector3 touchupCenter, in Vector3 position)
283 {
284 if (shape == Shape.Box)
285 return ProbeVolumePositioning.OBBContains(touchupOBB, position);
286 else
287 return (touchupCenter - position).sqrMagnitude < radius * radius;
288 }
