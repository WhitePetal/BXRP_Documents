141 parameters.skyOcclusionShadingDirection = skyOcclusion && skyOcclusionShadingDirection;
142 parameters.regionCount = m_CurrentBakingSet.bakedMaskCount;
143 parameters.regionLayerMasks = supportRenderingLayers ? m_CurrentBakingSet.bakedLayerMasks : 0xFFFFFFFF;
144 parameters.worldOffset = probeVolumesOptions.worldOffset_runtime;
145 UpdateConstantBuffer(cmd, parameters);
