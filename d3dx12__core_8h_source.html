74 D3D12_VIEWPORT( o )
75 {}
76 explicit CD3DX12_VIEWPORT(
77 FLOAT topLeftX,
78 FLOAT topLeftY,
79 FLOAT width,
80 FLOAT height,
81 FLOAT minDepth = D3D12_MIN_DEPTH,
82 FLOAT maxDepth = D3D12_MAX_DEPTH ) noexcept
83 {
84 TopLeftX = topLeftX;
85 TopLeftY = topLeftY;
86 Width = width;
87 Height = height;
88 MinDepth = minDepth;
89 MaxDepth = maxDepth;
90 }
91 explicit CD3DX12_VIEWPORT(
92 _In_ ID3D12Resource* pResource,
93 UINT mipSlice = 0,
94 FLOAT topLeftX = 0.0f,
95 FLOAT topLeftY = 0.0f,
96 FLOAT minDepth = D3D12_MIN_DEPTH,
97 FLOAT maxDepth = D3D12_MAX_DEPTH ) noexcept
98 {
99#if defined(_MSC_VER) || !defined(_WIN32)
100 const auto Desc = pResource->GetDesc();
101#else
102 D3D12_RESOURCE_DESC tmpDesc;
103 const auto& Desc = *pResource->GetDesc(&tmpDesc);
104#endif
105 const UINT64 SubresourceWidth = Desc.Width >> mipSlice;
106 const UINT64 SubresourceHeight = Desc.Height >> mipSlice;
107 switch (Desc.Dimension)
108 {
109 case D3D12_RESOURCE_DIMENSION_BUFFER:
110 TopLeftX = topLeftX;
111 TopLeftY = 0.0f;
112 Width = float(Desc.Width) - topLeftX;
113 Height = 1.0f;
114 break;
115 case D3D12_RESOURCE_DIMENSION_TEXTURE1D:
116 TopLeftX = topLeftX;
117 TopLeftY = 0.0f;
118 Width = (SubresourceWidth ? float(SubresourceWidth) : 1.0f) - topLeftX;
119 Height = 1.0f;
120 break;
121 case D3D12_RESOURCE_DIMENSION_TEXTURE2D:
122 case D3D12_RESOURCE_DIMENSION_TEXTURE3D:
123 TopLeftX = topLeftX;
124 TopLeftY = topLeftY;
125 Width = (SubresourceWidth ? float(SubresourceWidth) : 1.0f) - topLeftX;
126 Height = (SubresourceHeight ? float(SubresourceHeight) : 1.0f) - topLeftY;
127 break;
128 default: break;
129 }
130
131 MinDepth = minDepth;
132 MaxDepth = maxDepth;
133 }
134};

1444// Fills in the mipmap and alignment values of pDesc when either members are zero
1445// Used to replace an implicit field to an explicit (0 mip map = max mip map level)
1446// If expansion has occured, returns LclDesc, else returns the original pDesc
1447inline const CD3DX12_RESOURCE_DESC1* D3DX12ConditionallyExpandAPIDesc(
1448 CD3DX12_RESOURCE_DESC1& LclDesc,
1449 const CD3DX12_RESOURCE_DESC1* pDesc)
1450{
1451 // Expand mip levels:
1452 if (pDesc->MipLevels == 0 || pDesc->Alignment == 0)
1453 {
1454 LclDesc = *pDesc;
1455 if (pDesc->MipLevels == 0)
1456 {
1457 auto MaxMipLevels = [](UINT64 uiMaxDimension) -> UINT16
1458 {
1459 UINT16 uiRet = 0;
1460 while (uiMaxDimension > 0)
1461 {
1462 uiRet++;
1463 uiMaxDimension >>= 1;
1464 }
1465 return uiRet;
1466 };
1467 auto Max = [](UINT64 const& a, UINT64 const& b)
1468 {
1469 return (a < b) ? b : a;
1470 };
1471
1472 LclDesc.MipLevels = MaxMipLevels(
1473 Max(LclDesc.Dimension == D3D12_RESOURCE_DIMENSION_TEXTURE3D ? LclDesc.DepthOrArraySize : 1,
1474 Max(LclDesc.Width, LclDesc.Height)));
1475 }
1476 if (pDesc->Alignment == 0)
1477 {
1478 if (pDesc->Layout == D3D12_TEXTURE_LAYOUT_64KB_UNDEFINED_SWIZZLE
1479 || pDesc->Layout == D3D12_TEXTURE_LAYOUT_64KB_STANDARD_SWIZZLE
1480 )
1481 {
1482 LclDesc.Alignment = D3D12_DEFAULT_RESOURCE_PLACEMENT_ALIGNMENT;
1483 }
1484 else
1485 {
1486 LclDesc.Alignment =
1487 (pDesc->SampleDesc.Count > 1 ? D3D12_DEFAULT_MSAA_RESOURCE_PLACEMENT_ALIGNMENT : D3D12_DEFAULT_RESOURCE_PLACEMENT_ALIGNMENT);
1488 }
1489 }
1490 return &LclDesc;
1491 }
