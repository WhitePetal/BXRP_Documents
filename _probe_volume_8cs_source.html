109 internal Bounds ComputeBounds(GIContributor.ContributorFilter filter, Scene? scene = null)

114 void ExpandBounds(Bounds bound)
115 {
116 if (!foundABound)

118 bounds = bound;

128 foreach (var renderer in contributors.renderers)

226 internal bool ShouldCullCell(Vector3 cellPosition)
227 {
228 var cellSizeInMeters = ProbeReferenceVolume.instance.MaxBrickSize();
229 var probeOffset = ProbeReferenceVolume.instance.ProbeOffset() + ProbeVolumeDebug.currentOffset;
230 var debugDisplay = ProbeReferenceVolume.instance.probeVolumeDebug;
231 if (debugDisplay.realtimeSubdivision)
232 {
233 var bakingSet = ProbeVolumeBakingSet.GetBakingSetForScene(gameObject.scene);
234 if (bakingSet == null)
235 return true;
236
237 // Use the non-backed data to display real-time info
238 cellSizeInMeters = ProbeVolumeBakingSet.GetMinBrickSize(bakingSet.minDistanceBetweenProbes) * ProbeVolumeBakingSet.GetCellSizeInBricks(bakingSet.simplificationLevels);
239 probeOffset = bakingSet.probeOffset + ProbeVolumeDebug.currentOffset;
240 }
241 Camera activeCamera = Camera.current;
242#if UNITY_EDITOR
243 if (activeCamera == null)
244 activeCamera = UnityEditor.SceneView.lastActiveSceneView.camera;
245#endif
246
247 if (activeCamera == null)
248 return true;
249
250 var cameraTransform = activeCamera.transform;
251
252 Vector3 cellCenterWS = probeOffset + cellPosition * cellSizeInMeters + Vector3.one * (cellSizeInMeters / 2.0f);
253
254 // Round down to cell size distance
255 float roundedDownDist = Mathf.Floor(Vector3.Distance(cameraTransform.position, cellCenterWS) / cellSizeInMeters) * cellSizeInMeters;
256
257 if (roundedDownDist > debugDisplay.subdivisionViewCullingDistance)
258 return true;
259
260 var frustumPlanes = GeometryUtility.CalculateFrustumPlanes(activeCamera);
261 var volumeAABB = new Bounds(cellCenterWS, cellSizeInMeters * Vector3.one);
262
263 return !GeometryUtility.TestPlanesAABB(frustumPlanes, volumeAABB);
264 }
