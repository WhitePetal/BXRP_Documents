69 //--------------------------------------------------------------------------------------------------
70 // Add a vertex buffer along with its index buffer in GPU memory into the
71 // acceleration structure. The vertices are supposed to be represented by 3
72 // float32 value. This implementation limits the original flexibility of the
73 // API:
74 // - triangles (no custom intersector support)
75 // - 3xfloat32 format
76 // - 32-bit indices
77 void BottomLevelASGenerator::AddVertexBuffer(
78 ID3D12Resource* vertexBuffer, // Buffer containing the vertex coordinates,
79 // possibly interleaved with other vertex data
80 UINT64
81 vertexOffsetInBytes, // Offset of the first vertex in the vertex buffer
82 uint32_t vertexCount, // Number of vertices to consider in the buffer
83 UINT vertexSizeInBytes, // Size of a vertex including all its other data,
84 // used to stride in the buffer
85 ID3D12Resource* indexBuffer, // Buffer containing the vertex indices
86 // describing the triangles
87 UINT64 indexOffsetInBytes, // Offset of the first index in the index buffer
88 uint32_t indexCount, // Number of indices to consider in the buffer
89 ID3D12Resource* transformBuffer, // Buffer containing a 4x4 transform matrix
90 // in GPU memory, to be applied to the
91 // vertices. This buffer cannot be nullptr
92 UINT64 transformOffsetInBytes, // Offset of the transform matrix in the
93 // transform buffer
94 bool isOpaque /* = true */ // If true, the geometry is considered opaque,
95 // optimizing the search for a closest hit
96 )
97 {
98 // Create the DX12 descriptor representing the input data, assumed to be
99 // opaque triangles, with 3xf32 vertex coordinates and 32-bit indices
100 D3D12_RAYTRACING_GEOMETRY_DESC descriptor = {};
101 descriptor.Type = D3D12_RAYTRACING_GEOMETRY_TYPE_TRIANGLES;
102 descriptor.Triangles.VertexBuffer.StartAddress =
103 vertexBuffer->GetGPUVirtualAddress() + vertexOffsetInBytes;
104 descriptor.Triangles.VertexBuffer.StrideInBytes = vertexSizeInBytes;
105 descriptor.Triangles.VertexCount = vertexCount;
106 descriptor.Triangles.VertexFormat = DXGI_FORMAT_R32G32B32_FLOAT;
107 descriptor.Triangles.IndexBuffer =
108 indexBuffer ? (indexBuffer->GetGPUVirtualAddress() + indexOffsetInBytes)
109 : 0;
110 descriptor.Triangles.IndexFormat =
111 indexBuffer ? DXGI_FORMAT_R32_UINT : DXGI_FORMAT_UNKNOWN;
112 descriptor.Triangles.IndexCount = indexCount;
113 descriptor.Triangles.Transform3x4 =
114 transformBuffer
115 ? (transformBuffer->GetGPUVirtualAddress() + transformOffsetInBytes)
116 : 0;
117 descriptor.Flags = isOpaque ? D3D12_RAYTRACING_GEOMETRY_FLAG_OPAQUE
118 : D3D12_RAYTRACING_GEOMETRY_FLAG_NONE;
119
120 m_vertexBuffers.push_back(descriptor);
121 }

124 // Compute the size of the scratch space required to build the acceleration
125 // structure, as well as the size of the resulting structure. The allocation of
126 // the buffers is then left to the application
127 void BottomLevelASGenerator::ComputeASBufferSizes(
128 ID3D12Device5* device, // Device on which the build will be performed
129 bool allowUpdate, // If true, the resulting acceleration structure will
130 // allow iterative updates
131 UINT64* scratchSizeInBytes, // Required scratch memory on the GPU to build
132 // the acceleration structure
133 UINT64* resultSizeInBytes // Required GPU memory to store the acceleration
134 // structure
135 )
136 {
137 // The generated AS can support iterative updates. This may change the final
138 // size of the AS as well as the temporary memory requirements, and hence has
139 // to be set before the actual build
140 m_flags =
141 allowUpdate
142 ? D3D12_RAYTRACING_ACCELERATION_STRUCTURE_BUILD_FLAG_ALLOW_UPDATE
143 : D3D12_RAYTRACING_ACCELERATION_STRUCTURE_BUILD_FLAG_NONE;
144
145 // Describe the work being requested, in this case the construction of a
146 // (possibly dynamic) bottom-level hierarchy, with the given vertex buffers
147
148 D3D12_BUILD_RAYTRACING_ACCELERATION_STRUCTURE_INPUTS prebuildDesc;
149 prebuildDesc.Type = D3D12_RAYTRACING_ACCELERATION_STRUCTURE_TYPE_BOTTOM_LEVEL;
150 prebuildDesc.DescsLayout = D3D12_ELEMENTS_LAYOUT_ARRAY;
151 prebuildDesc.NumDescs = static_cast<UINT>(m_vertexBuffers.size());
152 prebuildDesc.pGeometryDescs = m_vertexBuffers.data();
153 prebuildDesc.Flags = m_flags;
154
155 // This structure is used to hold the sizes of the required scratch memory and
156 // resulting AS
157 D3D12_RAYTRACING_ACCELERATION_STRUCTURE_PREBUILD_INFO info = {};
158
159 // Building the acceleration structure (AS) requires some scratch space, as
160 // well as space to store the resulting structure This function computes a
161 // conservative estimate of the memory requirements for both, based on the
162 // geometry size.
163 device->GetRaytracingAccelerationStructurePrebuildInfo(&prebuildDesc, &info);
164
165 // Buffer sizes need to be 256-byte-aligned
166 *scratchSizeInBytes =
167 ROUND_UP(info.ScratchDataSizeInBytes,
168 D3D12_CONSTANT_BUFFER_DATA_PLACEMENT_ALIGNMENT);
169 *resultSizeInBytes = ROUND_UP(info.ResultDataMaxSizeInBytes,
170 D3D12_CONSTANT_BUFFER_DATA_PLACEMENT_ALIGNMENT);
171
172 // Store the memory requirements for use during build
173 m_scratchSizeInBytes = *scratchSizeInBytes;
174 m_resultSizeInBytes = *resultSizeInBytes;
175 }
176
177 //--------------------------------------------------------------------------------------------------
178 // Enqueue the construction of the acceleration structure on a command list,
179 // using application-provided buffers and possibly a pointer to the previous
180 // acceleration structure in case of iterative updates. Note that the update can
181 // be done in place: the result and previousResult pointers can be the same.
182 void BottomLevelASGenerator::Generate(
183 ID3D12GraphicsCommandList4
184 * commandList, // Command list on which the build will be enqueued
185 ID3D12Resource* scratchBuffer, // Scratch buffer used by the builder to
186 // store temporary data
187 ID3D12Resource
188 * resultBuffer, // Result buffer storing the acceleration structure
189 bool updateOnly, // If true, simply refit the existing
190 // acceleration structure
191 ID3D12Resource* previousResult // Optional previous acceleration
192 // structure, used if an iterative update
193 // is requested
194 )
195 {
196
197 D3D12_RAYTRACING_ACCELERATION_STRUCTURE_BUILD_FLAGS flags = m_flags;
198 // The stored flags represent whether the AS has been built for updates or
199 // not. If yes and an update is requested, the builder is told to only update
200 // the AS instead of fully rebuilding it
201 if (flags ==
202 D3D12_RAYTRACING_ACCELERATION_STRUCTURE_BUILD_FLAG_ALLOW_UPDATE &&
203 updateOnly)
204 {
205 flags = D3D12_RAYTRACING_ACCELERATION_STRUCTURE_BUILD_FLAG_PERFORM_UPDATE;
206 }
207
208 // Sanity checks
209 if (m_flags !=
210 D3D12_RAYTRACING_ACCELERATION_STRUCTURE_BUILD_FLAG_ALLOW_UPDATE &&
211 updateOnly)
212 {
213 throw std::logic_error(
214 "Cannot update a bottom-level AS not originally built for updates");
215 }
216 if (updateOnly && previousResult == nullptr)
217 {
218 throw std::logic_error(
219 "Bottom-level hierarchy update requires the previous hierarchy");
220 }
221
222 if (m_resultSizeInBytes == 0 || m_scratchSizeInBytes == 0)
223 {
224 throw std::logic_error(
225 "Invalid scratch and result buffer sizes - ComputeASBufferSizes needs "
226 "to be called before Build");
227 }
228 // Create a descriptor of the requested builder work, to generate a
229 // bottom-level AS from the input parameters
230 D3D12_BUILD_RAYTRACING_ACCELERATION_STRUCTURE_DESC buildDesc;
231 buildDesc.Inputs.Type = D3D12_RAYTRACING_ACCELERATION_STRUCTURE_TYPE_BOTTOM_LEVEL;
232 buildDesc.Inputs.DescsLayout = D3D12_ELEMENTS_LAYOUT_ARRAY;
233 buildDesc.Inputs.NumDescs = static_cast<UINT>(m_vertexBuffers.size());
234 buildDesc.Inputs.pGeometryDescs = m_vertexBuffers.data();
235 buildDesc.DestAccelerationStructureData = {
236 resultBuffer->GetGPUVirtualAddress() };
237 buildDesc.ScratchAccelerationStructureData = {
238 scratchBuffer->GetGPUVirtualAddress() };
239 buildDesc.SourceAccelerationStructureData =
240 previousResult ? previousResult->GetGPUVirtualAddress() : 0;
241 buildDesc.Inputs.Flags = flags;
242
243 // Build the AS
244 commandList->BuildRaytracingAccelerationStructure(&buildDesc, 0, nullptr);
245
246 // Wait for the builder to complete by setting a barrier on the resulting
247 // buffer. This is particularly important as the construction of the top-level
248 // hierarchy may be called right afterwards, before executing the command
249 // list.
250 D3D12_RESOURCE_BARRIER uavBarrier;
251 uavBarrier.Type = D3D12_RESOURCE_BARRIER_TYPE_UAV;
252 uavBarrier.UAV.pResource = resultBuffer;
253 uavBarrier.Flags = D3D12_RESOURCE_BARRIER_FLAG_NONE;
254 commandList->ResourceBarrier(1, &uavBarrier);
255 }
