106 private void GenerateGraphicsBuffe()
107 {
108 var renderSettings = BXVolumeManager.instance.renderSettings;
109 var expourseComponent = renderSettings.GetComponent<BXExpourseComponent>();
110
111 // EV-Expourse
112 commandBuffer.SetGlobalFloat(BXShaderPropertyIDs._ReleateExpourse_ID, expourseComponent.expourseRuntime / renderSettings.standard_expourse);
113
114 float aspec = camera.aspect;
115 float h_half;
116 if (camera.orthographic || camera.fieldOfView == 0f)
117 {
118 h_half = camera.orthographicSize;
119 }
120 else
121 {
122 float fov = camera.fieldOfView;
123 h_half = Mathf.Tan(0.5f * fov * Mathf.Deg2Rad);
124 }
125 float w_half = h_half * aspec;
126
127 Matrix4x4 viewPortRays = Matrix4x4.identity;
128 Vector4 forward = new Vector4(0f, 0f, -1f);
129 Vector4 up = new Vector4(0f, h_half, 0f);
130 Vector4 right = new Vector4(w_half, 0f, 0f);
131
132 Vector4 lb = forward - right - up;
133 Vector4 lu = forward - right + up;
134 Vector4 rb = forward + right - up;
135 Vector4 ru = forward + right + up;
136 viewPortRays.SetRow(0, lb);
137 viewPortRays.SetRow(1, lu);
138 viewPortRays.SetRow(2, rb);
139 viewPortRays.SetRow(3, ru);
140
141 commandBuffer.SetGlobalMatrix(BXShaderPropertyIDs._ViewPortRaysID, viewPortRays);
142
143 ExecuteCommand();
144 }

252 }

254 case LightType.Spot:
255 {
256 float range = visibleLight.range;
257 Vector3 lightSphere = lights.otherLightSpheres[i];

260 Vector3 toLight = (camera.transform.position - lightPos).normalized;
261 float cosAngle = Vector3.Dot(toLight, visibleLight.light.transform.forward);
262 float angleDst = (1 - cosAngle) * angleRangeInv;
263 Vector3 scale = Vector3.one * range;
264 scale.x *= visibleLight.spotAngle / 30f;
265 scale.y *= visibleLight.spotAngle / 30f;
266 Matrix4x4 localToWorld = Matrix4x4.TRS(lightPos, visibleLight.light.transform.rotation, scale);
267 if (Vector3.SqrMagnitude(lightSphere) <= (range * range - camera.nearClipPlane) && angleDst < 1f)
268 {
269 otherLightMat.SetInt(BXShaderPropertyIDs._StencilComp_ID, (int)CompareFunction.Always);
270 otherLightMat.SetInt(BXShaderPropertyIDs._StencilOp_ID, (int)StencilOp.Replace);
271 commandBuffer.DrawMesh(commonSettings.spotLightMesh, localToWorld, otherLightMat, 0, 1);
272 }
273 else
274 {
275 otherLightMat.SetInt(BXShaderPropertyIDs._StencilComp_ID, (int)CompareFunction.Equal);
276 otherLightMat.SetInt(BXShaderPropertyIDs._StencilOp_ID, (int)StencilOp.Keep);

278 commandBuffer.DrawMesh(commonSettings.spotLightMesh, localToWorld, otherLightMat, 0, 1);
