22// Uses STL

26// defining subobjects using the helper classes for each subobject (CD3DX12_DXIL_LIBRARY_SUBOBJECT
27// etc.). The subobject helpers each have methods specific to the subobject for configuring its
28// contents.

50class CD3DX12_STATE_OBJECT_DESC

52public:
53 CD3DX12_STATE_OBJECT_DESC() noexcept
54 {
55 Init(D3D12_STATE_OBJECT_TYPE_COLLECTION);
56 }
57 CD3DX12_STATE_OBJECT_DESC(D3D12_STATE_OBJECT_TYPE Type) noexcept
58 {
59 Init(Type);
60 }
61 void SetStateObjectType(D3D12_STATE_OBJECT_TYPE Type) noexcept { m_Desc.Type = Type; }
62 CD3DX12_STATE_OBJECT_DESC(const CD3DX12_STATE_OBJECT_DESC& other) = delete;
63 CD3DX12_STATE_OBJECT_DESC& operator=(const CD3DX12_STATE_OBJECT_DESC& other) = delete;
64 CD3DX12_STATE_OBJECT_DESC(CD3DX12_STATE_OBJECT_DESC&& other) = default;
65 CD3DX12_STATE_OBJECT_DESC& operator=(CD3DX12_STATE_OBJECT_DESC&& other) = default;
66 operator const D3D12_STATE_OBJECT_DESC& ()
67 {
68 // Do final preparation work
69 for (auto& ownedSubobject : m_OwnedSubobjectHelpers)
70 {
71 ownedSubobject->Finalize();
72 }
73
74#if defined(D3D12_SDK_VERSION) && (D3D12_SDK_VERSION >= 612)
75 m_RepointedSubobjectVectors.clear();
76 m_RepointedPrograms.clear();
77#endif
78 m_RepointedAssociations.clear();
79 m_SubobjectArray.clear();
80 m_SubobjectArray.reserve(m_Desc.NumSubobjects);
81 // Flatten subobjects into an array (each flattened subobject still has a
82 // member that's a pointer to its desc that's not flattened)
83 for (auto Iter = m_SubobjectList.begin();
84 Iter != m_SubobjectList.end(); Iter++)
85 {
86 m_SubobjectArray.push_back(*Iter);
87 // Store new location in array so we can redirect pointers contained in subobjects
88 Iter->pSubobjectArrayLocation = &m_SubobjectArray.back();
89 }
90 // For subobjects with pointer fields, create a new copy of those subobject definitions
91 // with fixed pointers
92 for (UINT i = 0; i < m_Desc.NumSubobjects; i++)
93 {
94 if (m_SubobjectArray[i].Type == D3D12_STATE_SUBOBJECT_TYPE_SUBOBJECT_TO_EXPORTS_ASSOCIATION)
95 {
96 auto pOriginalSubobjectAssociation =
97 static_cast<const D3D12_SUBOBJECT_TO_EXPORTS_ASSOCIATION*>(m_SubobjectArray[i].pDesc);
98 D3D12_SUBOBJECT_TO_EXPORTS_ASSOCIATION Repointed = *pOriginalSubobjectAssociation;
99 auto pWrapper =
100 static_cast<const SUBOBJECT_WRAPPER*>(pOriginalSubobjectAssociation->pSubobjectToAssociate);
101 Repointed.pSubobjectToAssociate = pWrapper->pSubobjectArrayLocation;
102 m_RepointedAssociations.push_back(Repointed);
103 m_SubobjectArray[i].pDesc = &m_RepointedAssociations.back();
104 }
105#if defined(D3D12_SDK_VERSION) && (D3D12_SDK_VERSION >= 612)
106 else if (m_SubobjectArray[i].Type == D3D12_STATE_SUBOBJECT_TYPE_GENERIC_PROGRAM)
107 {
108 auto originalGenericProgramDesc =
109 static_cast<const D3D12_GENERIC_PROGRAM_DESC*>(m_SubobjectArray[i].pDesc);
110 D3D12_GENERIC_PROGRAM_DESC Repointed = *originalGenericProgramDesc;
111 if (originalGenericProgramDesc->NumSubobjects > 0)
112 {
113 m_RepointedSubobjectVectors.emplace_back(std::vector<const D3D12_STATE_SUBOBJECT*>());
114 std::vector<D3D12_STATE_SUBOBJECT const*>& repointedGenericProgramSubobjects = m_RepointedSubobjectVectors.back();
115 repointedGenericProgramSubobjects.resize(originalGenericProgramDesc->NumSubobjects);
116 for (UINT s = 0; s < originalGenericProgramDesc->NumSubobjects; s++)
117 {
118 auto pWrapper =
119 static_cast<const SUBOBJECT_WRAPPER*>(originalGenericProgramDesc->ppSubobjects[s]);
120 repointedGenericProgramSubobjects[s] = pWrapper->pSubobjectArrayLocation;
121 }
122 // Below: using ugly way to get pointer in case .data() is not defined
123 Repointed.ppSubobjects = &repointedGenericProgramSubobjects[0];
124 }
125 m_RepointedPrograms.push_back(Repointed);
126 m_SubobjectArray[i].pDesc = &m_RepointedPrograms.back();
127 }
128#endif
129 }
130 // Below: using ugly way to get pointer in case .data() is not defined
131 m_Desc.pSubobjects = m_Desc.NumSubobjects ? &m_SubobjectArray[0] : nullptr;
132 return m_Desc;
133 }
134 operator const D3D12_STATE_OBJECT_DESC* ()
135 {
136 // Cast calls the above final preparation work
137 return &static_cast<const D3D12_STATE_OBJECT_DESC&>(*this);
138 }

140 // CreateSubobject creates a sububject helper (e.g. CD3DX12_HIT_GROUP_SUBOBJECT)
141 // whose lifetime is owned by this class.
142 // e.g.
143 //
144 // CD3DX12_STATE_OBJECT_DESC Collection1(D3D12_STATE_OBJECT_TYPE_COLLECTION);
145 // auto Lib0 = Collection1.CreateSubobject<CD3DX12_DXIL_LIBRARY_SUBOBJECT>();
146 // Lib0->SetDXILLibrary(&pMyAppDxilLibs[0]);
147 // Lib0->DefineExport(L"rayGenShader0"); // in practice these export listings might be
148 // // data/engine driven
149 // etc.
150 //
151 // Alternatively, users can instantiate sububject helpers explicitly, such as via local
152 // variables instead, passing the state object desc that should point to it into the helper
153 // constructor (or call mySubobjectHelper.AddToStateObject(Collection1)).
154 // In this alternative scenario, the user must keep the subobject alive as long as the state
155 // object it is associated with is alive, else its pointer references will be stale.
156 // e.g.
157 //
158 // CD3DX12_STATE_OBJECT_DESC RaytracingState2(D3D12_STATE_OBJECT_TYPE_RAYTRACING_PIPELINE);
159 // CD3DX12_DXIL_LIBRARY_SUBOBJECT LibA(RaytracingState2);
160 // LibA.SetDXILLibrary(&pMyAppDxilLibs[4]); // not manually specifying exports
161 // // - meaning all exports in the libraries
162 // // are exported
163 // etc.
164
165 template<typename T>
166 T* CreateSubobject()
167 {
168 T* pSubobject = new T(*this);
169 m_OwnedSubobjectHelpers.emplace_back(pSubobject);
170 return pSubobject;
171 }

174 D3D12_STATE_SUBOBJECT* TrackSubobject(D3D12_STATE_SUBOBJECT_TYPE Type, void* pDesc)
175 {
176 SUBOBJECT_WRAPPER Subobject;
177 Subobject.pSubobjectArrayLocation = nullptr;
178 Subobject.Type = Type;
179 Subobject.pDesc = pDesc;
180 m_SubobjectList.push_back(Subobject);
181 m_Desc.NumSubobjects++;
182 return &m_SubobjectList.back();
183 }
184 void Init(D3D12_STATE_OBJECT_TYPE Type) noexcept
185 {
186 SetStateObjectType(Type);
187 m_Desc.pSubobjects = nullptr;
188 m_Desc.NumSubobjects = 0;
189 m_SubobjectList.clear();
190 m_SubobjectArray.clear();
191 m_RepointedAssociations.clear();
192#if defined(D3D12_SDK_VERSION) && (D3D12_SDK_VERSION >= 612)
193 m_RepointedSubobjectVectors.clear();
194 m_RepointedPrograms.clear();
195#endif

202 D3D12_STATE_OBJECT_DESC m_Desc;

204 // these can be edited live

206
207 std::list<D3D12_SUBOBJECT_TO_EXPORTS_ASSOCIATION>
