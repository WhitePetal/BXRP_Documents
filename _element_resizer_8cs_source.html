86 void OnMouseMove(MouseMoveEvent e)
87 {
88 if (!isEnabled)
89 return;
90
91 VisualElement resizedTarget = resizedElement.parent;
92 VisualElement resizedBase = resizedTarget.parent;
93
94 // Top left position of the parent visual element
95 var parentRootPosition = resizedBase.worldBound;
96 // Top left of the target visual element for resizing
97 var targetRootPosition = resizedTarget.worldBound;
98 var canResizePastParentBounds = ((IGGResizable)resizedTarget).CanResizePastParentBounds();
99
100 Vector2 mousePos = resizedBase.WorldToLocal(e.mousePosition);
101
102 if (!m_DragStarted)
103 {
104 if (resizedTarget is IGGResizable resizable)
105 resizable.OnStartResize();
106 m_DragStarted = true;
107 }
108
109 if ((direction & ResizableElement.Resizer.Right) != 0)
110 {
111 var newWidth = m_StartSize.x + mousePos.x - m_StartMouse.x;
112 var parentRightBoundary = parentRootPosition.x + resizedBase.layout.width;
113 // Also ensure resizing does not happen past edge of parent views boundaries if the target does not allow it
114 if (!canResizePastParentBounds)
115 {
116 if ((targetRootPosition.x + newWidth) > parentRightBoundary)
117 {
118 var targetToRightBoundaryDelta = parentRightBoundary - targetRootPosition.x;
119 newWidth = targetToRightBoundaryDelta;
120 }
121 var newLayoutLeft = targetRootPosition.x - parentRootPosition.x;
122 // When resizing to right, make sure to calculate and set the target elements Style.left before resizing to ensure correct resizing behavior
123 // If Style.left is NaNpx it results in scaling towards the left
124 // This is due to how the WindowDockingLayout code affects GraphSubWindows
125 resizedTarget.style.left = newLayoutLeft;
126 }
127
128 resizedTarget.style.width = Mathf.Clamp(newWidth, m_MinSize.x, m_MaxSize.x);
129 }
130 else if ((direction & ResizableElement.Resizer.Left) != 0)
131 {
132 float delta = mousePos.x - m_StartMouse.x;
133
134 if (m_StartSize.x - delta < m_MinSize.x)
135 {
136 delta = -m_MinSize.x + m_StartSize.x;
137 }
138 else if (m_StartSize.x - delta > m_MaxSize.x)
139 {
140 delta = -m_MaxSize.x + m_StartSize.x;
141 }
142
143 var newWidth = -delta + m_StartSize.x;
144 var targetToLeftBoundaryDelta = delta + m_StartPosition.x;
145
146 if (!canResizePastParentBounds)
147 {
148 // This ensures that the left side of the resizing target never can get pushed past the parent boundary even if mouse is moving really fast
149 targetToLeftBoundaryDelta = Mathf.Clamp(targetToLeftBoundaryDelta, 2.5f, targetToLeftBoundaryDelta);
150
151 // Clamps width to max out at left edge of parent window
152 if (Mathf.Approximately(targetToLeftBoundaryDelta, 2.5f))
153 newWidth = (m_StartPosition.x + m_StartSize.x);
154
155 newWidth = Mathf.Clamp(newWidth, m_MinSize.x, m_MaxSize.x);
156 }
157
158 resizedTarget.style.left = targetToLeftBoundaryDelta;
159 resizedTarget.style.width = newWidth;
160 }
161

163 {

176 // When resizing to bottom, make sure to calculate and set the target elements Style.top before resizing to ensure correct resizing behavior

178 // This is due to how the WindowDockingLayout code affects GraphSubWindows
