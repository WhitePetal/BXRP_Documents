168 // Used to avoid gcalloc in m_Pools.FindIndex
169 private static int s_ChunkCount;
170
171 // Will return a the smallest GraphicsBuffer with at least enough space for chunkCount chunks.
172 public bool AllocateScratchBuffer(int chunkCount, out CellStreamingScratchBuffer scratchBuffer, out CellStreamingScratchBufferLayout layout, bool allocateGraphicsBuffers)
173 {
174 s_ChunkCount = chunkCount;
175 int index = m_Pools.FindIndex(0, (o) => o.chunkCount == s_ChunkCount);
176 // The size of buffer we return may not be the exact requested size (can be bigger).
177 // So we need to make sure the layout is the right one for the number of requested chunks.
178 layout = GetOrCreateScratchBufferLayout(chunkCount);
179
180 if(index != -1)
181 {
182 var pool = m_Pools[index].pool;
183 // If one is available, return it.
184 if(pool.Count > 0)
185 {
186 scratchBuffer = pool.Pop();
187 scratchBuffer.Swap();
188 return true;
189 }
190 else
191 {
192 // If none are available of the exact size, try to find a bigger buffer instead.
193 // We won't go above twice the size to avoid "wasting" bigger buffers for requests that are too small.
194 for(int i = index; i < m_Pools.Count; ++i)
195 {
196 var biggerPool = m_Pools[index];
197 // Next existing pools scratch size are too big. Break out of the loop.
198 // We don't want to hog a big buffer for a small request.
199 if(biggerPool.chunkCount >= (chunkCount * 2))
200 {
201 break;
202 }
203 else if(biggerPool.pool.Count > 0)
204 {
205 scratchBuffer = pool.Pop();
206 scratchBuffer.Swap();
207 return true;
208 }
209 }
210
211 // We did not find any suitable buffer. Create a new one with the exact requested size.
212 if((m_CurrentlyAllocatedChunkCount + chunkCount) < maxChunkCount)
213 {
214 scratchBuffer = CreateScratchBuffer(chunkCount, allocateGraphicsBuffers);
215 return true;
216 }
217 // If we are above the maximum, we don't allocate more buffers.
218 // It's ok since we know at least one buffer will exist for any given chunkCount so we won't get impossible to fulfill requests.
219 else
220 {
221 scratchBuffer = null;
222 return false;
223 }
224 }
225 }
226 else
227 {
228 // No pool of this size exists. Create a new pool of that size and return a new buffer;
229 ScratchBufferPool newPool = new ScratchBufferPool(chunkCount);
230 m_Pools.Add(newPool);
231 m_Pools.Sort();
232
233 scratchBuffer = CreateScratchBuffer(chunkCount, allocateGraphicsBuffers);
234 return true;
235 }
236 }

238 public void ReleaseScratchBuffer(CellStreamingScratchBuffer scratchBuffer)
239 {
240 if(scratchBuffer.chunkSize != chunkSize)
241 {
242 scratchBuffer.Dispose();
243 return;
244 }
245

247 var pool = m_Pools.Find((o) => o.chunkCount == s_ChunkCount);

249 pool.pool.Push(scratchBuffer);
