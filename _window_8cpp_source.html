89void Window::OnResize(ResizeEventArgs& e)
90{
91 if (m_ClientWidth != e.Width || m_ClientHeight != e.Height)
92 {
93 // Don't allow 0 size swap chain back buffers.
94 m_ClientWidth = std::max(1, e.Width);
95 m_ClientHeight = std::max(1, e.Height);
96
97 Application& app = Application::Get();
98
99 // Flush the GPU queue to make sure the swap chain's back buffers
100 // are not being referenced by an in-flight command list.
101 app.Flush();
102
103 for (int i = 0; i < BackBufferCount; ++i)
104 {
105 // Any references to the back buffers must be released
106 // before the swap chain can be resized
107 m_d3d12BackBuffers[i].Reset();
108 }
109
110 DXGI_SWAP_CHAIN_DESC swapChainDesc = {};
111 ThrowIfFaild(m_dxgiSwapChain->GetDesc(&swapChainDesc));
112 ThrowIfFaild(m_dxgiSwapChain->ResizeBuffers(BackBufferCount, m_ClientWidth, m_ClientHeight,
113 swapChainDesc.BufferDesc.Format, swapChainDesc.Flags));
114
115 m_CurrentBackBufferIndex = m_dxgiSwapChain->GetCurrentBackBufferIndex();
116
117 UpdateRenderTargetViews(app);
118 }
119
120 if (auto pEngine = m_pEngine.lock())
121 {
122 pEngine->OnResize(e);
123 }
124}

214void Window::SetFullScreen(bool fullscreen)
215{
216 if (m_Fullscreen != fullscreen)
217 {
218 m_Fullscreen = fullscreen;
219
220 if (m_Fullscreen) // Switching to fullscreen
221 {
222 // Store the current window dimensions so they can be restored
223 // when switching out of fullscreen state.
224 ::GetWindowRect(m_hWnd, &m_WindowRect);
225
226 // Set the window style to a borderless window so the client area fills
227 // the enter screen.
228 UINT windowStyle = WS_OVERLAPPEDWINDOW & ~(WS_CAPTION | WS_SYSMENU | WS_THICKFRAME | WS_MINIMIZEBOX | WS_MAXIMIZEBOX);
229 ::SetWindowLongW(m_hWnd, GWL_STYLE, windowStyle);
230
231 // Query the name of the nearest display device for the window.
232 // This is required to set the fullscreen dimensions of the window
233 // when using a multi-monitor steup.
234 HMONITOR hMonitor = ::MonitorFromWindow(m_hWnd, MONITOR_DEFAULTTONEAREST);
235 MONITORINFOEX monitorInfo = {};
236 monitorInfo.cbSize = sizeof(MONITORINFOEX);
237 ::GetMonitorInfo(hMonitor, &monitorInfo);
238
239 ::SetWindowPos(m_hWnd, HWND_TOP,
240 monitorInfo.rcMonitor.left,
241 monitorInfo.rcMonitor.top,
242 monitorInfo.rcMonitor.right - monitorInfo.rcMonitor.left,
243 monitorInfo.rcMonitor.bottom - monitorInfo.rcMonitor.top,
244 SWP_FRAMECHANGED | SWP_NOACTIVATE);
245
246 ::ShowWindow(m_hWnd, SW_MAXIMIZE);
247 }
248 else
249 {
250 // Restore all the window decorators.
251 ::SetWindowLong(m_hWnd, GWL_STYLE, WS_OVERLAPPEDWINDOW);
252
253 ::SetWindowPos(m_hWnd, HWND_NOTOPMOST,
254 m_WindowRect.left,
255 m_WindowRect.top,
256 m_WindowRect.right - m_WindowRect.left,
257 m_WindowRect.bottom - m_WindowRect.top,
258 SWP_FRAMECHANGED | SWP_NOACTIVATE);
259
260 ::ShowWindow(m_hWnd, SW_NORMAL);
261 }
262 }

326ComPtr<IDXGISwapChain4> Window::CreateSwapChain(Application& app)
327{
328 ComPtr<IDXGISwapChain4> swapChain4;
329 ComPtr<IDXGIFactory4> dxgiFactory4;
330 UINT createFactoryFlags = 0;
331#if defined(_DEBUG)
332 createFactoryFlags = DXGI_CREATE_FACTORY_DEBUG;
333#endif
334
335 ThrowIfFaild(CreateDXGIFactory2(createFactoryFlags, IID_PPV_ARGS(&dxgiFactory4)));
336
337 DXGI_SWAP_CHAIN_DESC1 swapChainDesc = {};
338 swapChainDesc.Width = m_ClientWidth;
339 swapChainDesc.Height = m_ClientHeight;
340 swapChainDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
341 swapChainDesc.Stereo = FALSE;
342 swapChainDesc.SampleDesc = { 1, 0 };
343 swapChainDesc.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
344 swapChainDesc.BufferCount = BackBufferCount;
345 swapChainDesc.Scaling = DXGI_SCALING_STRETCH;
346 swapChainDesc.SwapEffect = DXGI_SWAP_EFFECT_FLIP_DISCARD;
347 swapChainDesc.AlphaMode = DXGI_ALPHA_MODE_UNSPECIFIED;
348 // It is recommended to always teraing if tearing support is available.
349 swapChainDesc.Flags = m_IsTearingSupported ? DXGI_SWAP_CHAIN_FLAG_ALLOW_TEARING : 0;
350
351 ID3D12CommandQueue* pCommandQueue = app.GetCommandQueue()->GetD3D12CommandQueue().Get();
352
353 ComPtr<IDXGISwapChain1> swapChain1;
354 ThrowIfFaild(dxgiFactory4->CreateSwapChainForHwnd(
355 pCommandQueue,
356 m_hWnd,
357 &swapChainDesc,
358 nullptr,
359 nullptr,
360 &swapChain1
361 ));
362
363 // Disable the Alt+Enter fullscreen toggle feature. Switching to fullscreen
364 // will be handled manually.
365 ThrowIfFaild(dxgiFactory4->MakeWindowAssociation(m_hWnd, DXGI_MWA_NO_ALT_ENTER));
366
367 ThrowIfFaild(swapChain1.As(&swapChain4));
368
369 m_CurrentBackBufferIndex = swapChain4->GetCurrentBackBufferIndex();
370
371 return swapChain4;
372}
