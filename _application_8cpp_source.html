193bool Application::CheckTearingSupport()
194{
195 BOOL allowTearing = FALSE;
196
197 // Rather than create the DXGI 1.5 factory interface directly, we create the
198 // DXGI 1.4 interface and query for the 1.5 interface. This is to enable the
199 // graphics debugging tools which will not support the 1.5 factory interface
200 // until a future update
201 if (SUCCEEDED(CreateDXGIFactory1(IID_PPV_ARGS(&m_dxgiFactory))))
202 {
203 ComPtr<IDXGIFactory5> factory5;
204 if (SUCCEEDED(m_dxgiFactory.As(&factory5)))
205 {
206 if (FAILED(factory5->CheckFeatureSupport(
207 DXGI_FEATURE_PRESENT_ALLOW_TEARING,
208 &allowTearing, sizeof(allowTearing)
209 )))
210 {
211 allowTearing = FALSE;
212 }
213 }
214 }
215
216 return allowTearing == TRUE;

268{
269 // First check if a window with the given name already exists

281 int windowWidth = windowRect.right - windowRect.left;
282 int windowHeight = windowRect.bottom - windowRect.top;
283
284 // Center the window within the screen. Clamp to 0, 0 for the top-left corner

290 WINDOW_CLASS_NAME,
291 windowName.c_str(),
292 WS_OVERLAPPEDWINDOW,
293 windowX,

297 m_ParentWnd,

301 );

342int Application::Run(std::shared_ptr<Engine> pEngine)
343{
344 if (!pEngine->Initialize()) return 1;
345 if (!pEngine->LoadContent()) return 2;
346 MSG msg = { 0 };
347 while (msg.message != WM_QUIT)
348 {
349 if (::PeekMessage(&msg, 0, 0, 0, PM_REMOVE))
350 {
351 ::TranslateMessage(&msg);
352 ::DispatchMessage(&msg);
353 }
354 }
355
356 // Flush any commands in the commands queues before quiting
357 Flush();
358
359 pEngine->UnloadContent();
360 pEngine->Destroy();
361
362 return 0;
