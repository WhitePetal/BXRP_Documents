44bool SampleScene::LoadContent()
45{
46 Application& app = Application::Get();
47 auto device = app.GetDevice();
48 auto commandQueue = app.GetCommandQueue(D3D12_COMMAND_LIST_TYPE_COPY);
49 auto commandList = commandQueue->GetCommandList();
50
51 m_Camera->CreateCameraBuffer();
52
53 // Upload vertex buffer data.
54 ComPtr<ID3D12Resource> intermediateVertexBuffer;
55 UpdateBufferResource(
56 commandList,
57 &m_VertexBuffer,
58 &intermediateVertexBuffer,
59 _countof(g_Vertices), sizeof(VertexPosColor), g_Vertices);
60
61 // Create the vertex buffer view.
62 m_VertexBufferView.BufferLocation = m_VertexBuffer->GetGPUVirtualAddress();
63 m_VertexBufferView.SizeInBytes = sizeof(g_Vertices);
64 m_VertexBufferView.StrideInBytes = sizeof(VertexPosColor);
65
66 // Upload index buffer data.
67 ComPtr<ID3D12Resource> intermediateIndexBuffer;
68 UpdateBufferResource(
69 commandList,
70 &m_IndexBuffer,
71 &intermediateIndexBuffer,
72 _countof(g_Indicies), sizeof(uint32_t), g_Indicies);
73
74 // Create index buffer view.
75 m_IndexBufferView.BufferLocation = m_IndexBuffer->GetGPUVirtualAddress();
76 m_IndexBufferView.Format = DXGI_FORMAT_R32_UINT;
77 m_IndexBufferView.SizeInBytes = _countof(g_Indicies) * sizeof(uint32_t);
78
79 // Create the descriptor heap for the depth-stencil view (DSV)
80 D3D12_DESCRIPTOR_HEAP_DESC dsvHeapDesc = {};
81 dsvHeapDesc.NumDescriptors = 1;
82 dsvHeapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_DSV;
83 dsvHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_NONE;
84 ThrowIfFaild(device->CreateDescriptorHeap(&dsvHeapDesc, IID_PPV_ARGS(&m_DSVHeap)));
85
86 auto fenceValue = commandQueue->ExecuteCommandList(commandList);
87 commandQueue->WaitForFenceValue(fenceValue);
88
89 m_ContentLoaded = true;
90
91 ResizeDepthBuffer(m_Width, m_Height);
92
93 m_RasterPipeline->LoadContent();
94 m_RaytracingPipeline->LoadContent();
95
96
97 return true;
98}

141void SampleScene::OnRender(RenderEventArgs& e)
142{
143 super::OnRender(e);
144
145 m_Camera->UpdateCameraBuffer();
146
147 auto commandQueue = Application::Get().GetCommandQueue(D3D12_COMMAND_LIST_TYPE_DIRECT);
148 auto commandList = commandQueue->GetCommandList();
149
150 UINT currentBackBufferIndex = m_pWindow->GetCurrentBackBufferIndex();
151 auto backBuffer = m_pWindow->GetCurrentBackBuffer();
152 auto rtv = m_pWindow->GetCurrentRenderTargetView();
153 auto dsv = m_DSVHeap->GetCPUDescriptorHandleForHeapStart();
154
155 commandList->RSSetViewports(1, &m_Viewport);
156 commandList->RSSetScissorRects(1, &m_ScissorRect);
157
158 commandList->OMSetRenderTargets(1, &rtv, FALSE, &dsv);
159
160 TransitionResource(commandList, backBuffer, D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET);
161
162 if (m_Raster)
163 {
164 m_RasterPipeline->OnRender(commandList, rtv, dsv);
165 }
166 else
167 {
168 m_RaytracingPipeline->OnRender(commandList, rtv, backBuffer);
169 }
170
171 // Present
172 {
173 TransitionResource(commandList, backBuffer, D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT);
174
175 m_FenceValues[currentBackBufferIndex] = commandQueue->ExecuteCommandList(commandList);
176
177 currentBackBufferIndex = m_pWindow->Present();
178
179 commandQueue->WaitForFenceValue(m_FenceValues[currentBackBufferIndex]);
180 }
