64 public CopyPasteGraph(IEnumerable<GroupData> groups,
65 IEnumerable<AbstractGeometryNode> nodes,
66 IEnumerable<Edge> edges,
67 IEnumerable<GeometryInput> inputs,
68 IEnumerable<CategoryData> categories,
69 IEnumerable<AbstractGeometryProperty> metaProperties,
70 IEnumerable<ShaderKeyword> metaKeywords,
71 IEnumerable<GeometryDropdown> metaDropdowns,
72 IEnumerable<StickyNoteData> notes,
73 bool keepOutputEdges = false,
74 bool removeOrphanEdges = true,
75 CopyPasteGraphSource copyPasteGraphSource = CopyPasteGraphSource.Default)
76 {
77 m_CopyPasteGraphSource = copyPasteGraphSource;
78 if (groups != null)
79 {
80 foreach (var groupData in groups)
81 AddGroup(groupData);
82 }
83
84 if (notes != null)
85 {
86 foreach (var stickyNote in notes)
87 AddNote(stickyNote);
88 }
89
90 var nodeSet = new HashSet<AbstractGeometryNode>();
91
92 if (nodes != null)
93 {
94 foreach (var node in nodes.Distinct())
95 {
96 if (!node.canCopyNode)
97 {
98 throw new InvalidOperationException($"Cannot copy node {node.name} ({node.objectId}).");
99 }
100
101 nodeSet.Add(node);
102 AddNode(node);
103 foreach (var edge in NodeUtils.GetAllEdges(node))
104 AddEdge((Edge)edge);
105 }
106 }
107
108 if (edges != null)
109 {
110 foreach (var edge in edges)
111 AddEdge(edge);
112 }
113
114 if (inputs != null)
115 {
116 foreach (var input in inputs)
117 AddInput(input);
118 }
119
120 if (categories != null)
121 {
122 foreach (var category in categories)
123 AddCategory(category);
124 }
125
126 if (metaProperties != null)
127 {
128 foreach (var metaProperty in metaProperties.Distinct())
129 AddMetaProperty(metaProperty);
130 }
131
132 if (metaKeywords != null)
133 {
134 foreach (var metaKeyword in metaKeywords.Distinct())
135 AddMetaKeyword(metaKeyword);
136 }
137
138 if (metaDropdowns != null)
139 {
140 foreach (var metaDropdown in metaDropdowns.Distinct())
141 AddMetaDropdown(metaDropdown);
142 }
143
144 var distinct = m_Edges.Distinct();
145 if (removeOrphanEdges)
146 {
147 distinct = distinct.Where(edge => nodeSet.Contains(edge.inputSlot.node) || (keepOutputEdges && nodeSet.Contains(edge.outputSlot.node)));
148 }
149 m_Edges = distinct.ToList();
150 }

163 // The only situation in which an input has an identical reference name to another input in a category, while not being the same instance, is if they are duplicates
164 public bool IsInputDuplicatedFromCategory(GeometryInput geometryInput, CategoryData inputCategory, GraphData targetGraphData)
165 {
166 foreach (var child in inputCategory.Children)
167 {
168 if (child.referenceName.Equals(geometryInput.referenceName, StringComparison.Ordinal) && child.objectId != geometryInput.objectId)
169 {
170 return true;
171 }
172 }
173
174 // Need to check if they share same graph owner as well, if not then we can early out
175 bool inputBelongsToTargetGraph = targetGraphData.ContainsInput(geometryInput);
176 if (inputBelongsToTargetGraph == false)
177 return false;
178
179 return false;
180 }
