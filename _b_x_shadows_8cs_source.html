101 public Vector4 SaveDirectionalShadows(Light light, int visibleLightIndex)
102 {
103 if (shadowedDirLightCount < maxShadowedDirLightCount && light.shadows != LightShadows.None &&
104 light.shadowStrength > 0f)
105 {
106 float maskChannel = -1;
107 LightBakingOutput lightBaking = light.bakingOutput;
108 if(lightBaking.lightmapBakeType == LightmapBakeType.Mixed && lightBaking.mixedLightingMode == MixedLightingMode.Shadowmask)
109 {
110 useShadowMask = true;
111 maskChannel = lightBaking.occlusionMaskChannel;
112 }
113 shadowedDirLights[shadowedDirLightCount] = new ShadowedDirectionalLight()
114 {
115 visibleLightIndex = visibleLightIndex,
116 slopeScaleBias = light.shadowBias,
117 nearPlaneOffst = light.shadowNearPlane
118 };
119
120 if(!cullingResults.GetShadowCasterBounds(visibleLightIndex, out Bounds bounds))
121 {
122 return new Vector4(-light.shadowStrength, 0f, 0f, maskChannel);
123 }
124
125 return new Vector4(light.shadowStrength, commonSettings.cascadeCount * shadowedDirLightCount++, light.shadowNormalBias, maskChannel);
126 }
127 else
128 {
129 return Vector4.zero;
130 }
131 }

133 public Vector4 SaveOtherShadows(Light light, int visibleLightIndex, int lightIndex)
134 {
135 if (light.shadows == LightShadows.None || light.shadowStrength <= 0f)
136 {
137 return new Vector4(0f, 0f, 0f, -1f);
138 }
139 float maskChannel = -1f;
140 LightBakingOutput lightBaking = light.bakingOutput;
141 if (lightBaking.lightmapBakeType == LightmapBakeType.Mixed && lightBaking.mixedLightingMode == MixedLightingMode.Shadowmask)
142 {
143 maskChannel = lightBaking.occlusionMaskChannel;
144 }
145 if (shadowedOtherLightCount >= maxShadowedOtherLightCount ||
146 !cullingResults.GetShadowCasterBounds(visibleLightIndex, out Bounds bounds))
147 {
148 return new Vector4(-light.shadowStrength, 0f, 0f, maskChannel);
149 }
150
151 bool isPoint = light.type == LightType.Point;
152
153 shadowedOtherLights[shadowedOtherLightCount] = new ShadowedOtherLight
154 {
155 lightIndex = lightIndex,
156 tileIndex = shadowedOtherLightTileCount,
157 visibleLightIndex = visibleLightIndex,
158 slopeScaleBias = light.shadowBias,
159 normalBias = light.shadowNormalBias,
160 isPoint = isPoint
161 };
162 ++shadowedOtherLightCount;
163 Vector4 data = new Vector4(light.shadowStrength, shadowedOtherLightTileCount, isPoint ? 1 : 0, maskChannel);
164 shadowedOtherLightTileCount += isPoint ? 6 : 1;
165 return data;
166 }

168 public void Render(List<BXRenderFeature> onDirShadowsRenderFeatures)
169 {
170 if (!commonSettings.drawShadows)
171 {
172 bool needExcuteCMD = false;
173 if (Shader.IsKeywordEnabled(in dirShadowKeyword))
174 {
175 needExcuteCMD = true;
176 commandBuffer.DisableKeyword(dirShadowKeyword);
177 }
178 if (Shader.IsKeywordEnabled(in otherShadowKeyword))
179 {
180 needExcuteCMD = true;
181 commandBuffer.DisableKeyword(otherShadowKeyword);
182 }
183 if (needExcuteCMD) ExecuteCommandBuffer();
184 return;
185 }
186
187 if (shadowedDirLightCount > 0)
188 {
189 if (!Shader.IsKeywordEnabled(in dirShadowKeyword))
190 commandBuffer.EnableKeyword(in dirShadowKeyword);
191 if (useShadowMask)
192 {
193 if(QualitySettings.shadowmaskMode == ShadowmaskMode.Shadowmask)
194 {
195 if (Shader.IsKeywordEnabled(shadowMaskDistanceKyeword)) commandBuffer.DisableKeyword(shadowMaskDistanceKyeword);
196 if (!Shader.IsKeywordEnabled(shadowMaskAlwaysKeyword)) commandBuffer.EnableKeyword(shadowMaskAlwaysKeyword);
197 }
198 else
199 {
200 if (!Shader.IsKeywordEnabled(shadowMaskDistanceKyeword)) commandBuffer.EnableKeyword(shadowMaskDistanceKyeword);
201 if (Shader.IsKeywordEnabled(shadowMaskAlwaysKeyword)) commandBuffer.DisableKeyword(shadowMaskAlwaysKeyword);
202 }
203 }
204 else
205 {
206 if(Shader.IsKeywordEnabled(shadowMaskDistanceKyeword)) commandBuffer.DisableKeyword(shadowMaskDistanceKyeword);
207 if(Shader.IsKeywordEnabled(shadowMaskAlwaysKeyword)) commandBuffer.DisableKeyword(shadowMaskAlwaysKeyword);
208 }
209
210 RenderDirectionalShadows(onDirShadowsRenderFeatures);
211 }
212 if (shadowedOtherLightCount > 0)
213 {
214 if (!Shader.IsKeywordEnabled(in otherShadowKeyword))
215 commandBuffer.EnableKeyword(in otherShadowKeyword);
216 RenderOtherShadows();
217 }
218
219 commandBuffer.BeginSample(BufferName);
220 commandBuffer.SetGlobalVector(ShaderProperties._ShadowMapSize_ID, shadowMapSizes);
221 commandBuffer.EndSample(BufferName);
222 ExecuteCommandBuffer();
223 }

246 private void RenderDirectionalShadows(List<BXRenderFeature> onDirShadowsRenderFeatures)
247 {
248 int shadowMapSize = commonSettings.shadowMapSize;
249 commandBuffer.GetTemporaryRT(ShaderProperties._DirectionalShadowMap_ID, shadowMapSize, shadowMapSize, commonSettings.shadowMapBits, FilterMode.Bilinear, RenderTextureFormat.Shadowmap, RenderTextureReadWrite.Linear, 1, false, RenderTextureMemoryless.Color);
250 commandBuffer.SetRenderTarget(ShaderProperties._DirectionalShadowMap_TargetID, RenderBufferLoadAction.DontCare, RenderBufferStoreAction.DontCare, RenderBufferLoadAction.DontCare, RenderBufferStoreAction.Store);
251 commandBuffer.ClearRenderTarget(true, false, Color.clear);
252 commandBuffer.SetGlobalInt(ShaderProperties._ShadowPancaking_ID, 1);
253 commandBuffer.BeginSample(BufferName);
254 ExecuteCommandBuffer();
255 shadowMapSizes.x = shadowMapSize;
256 shadowMapSizes.y = 1.0f / shadowMapSize;
257
258 int cascadeCount = commonSettings.cascadeCount;
259 Vector3 cascadeRatios = new Vector3(commonSettings.cascadeRatio1, commonSettings.cascadeRatio2, commonSettings.cascadeRatio3);
260 int tileCount = shadowedDirLightCount * cascadeCount;
261 int split = tileCount <= 1 ? 1 : tileCount <= 4 ? 2 : 4;
262 int tileSize = shadowMapSize / split;
263 float cullingFactor = Mathf.Max(0f, 0.8f - commonSettings.cascadeFade);
264 for (int i = 0; i < shadowedDirLightCount; ++i)
265 {
266 int tileIndexOffset = i * cascadeCount;
267 ShadowedDirectionalLight light = shadowedDirLights[i];
268 for (int cascadeIndex = 0; cascadeIndex < cascadeCount; ++cascadeIndex)
269 {
270 ShadowDrawingSettings dirShadowSettings = new ShadowDrawingSettings(cullingResults, light.visibleLightIndex, BatchCullingProjectionType.Orthographic);
271 cullingResults.ComputeDirectionalShadowMatricesAndCullingPrimitives(light.visibleLightIndex, cascadeIndex, cascadeCount, cascadeRatios, tileSize, 0f,
272 out Matrix4x4 viewMatrix, out Matrix4x4 projMatrix, out ShadowSplitData shadowSplitData);
273 shadowSplitData.shadowCascadeBlendCullingFactor = cullingFactor;
274 dirShadowSettings.splitData = shadowSplitData;
275 // 所有方向光都使用同一个级联球体，因此只需要存储第一个光源的即可
276 if (i == 0)
277 {
278 SetCascadeData(cascadeIndex, shadowSplitData.cullingSphere, tileSize);
279 }
280 int tileIndex = cascadeIndex + tileIndexOffset;
281 dirShadowMatrixs[tileIndex] = ConvertToShadowMapTileMatrix(projMatrix * viewMatrix, SetTileViewport(tileIndex, split, tileSize), 1f / split) * viewToWorldMatrix;
282 commandBuffer.SetViewProjectionMatrices(viewMatrix, projMatrix);
283 commandBuffer.SetGlobalDepthBias(1f, 2.5f + light.slopeScaleBias);
284 ExecuteCommandBuffer();
285 context.DrawShadows(ref dirShadowSettings);
286 ExecuteRenderFeatures(onDirShadowsRenderFeatures);
287 }
288 }
289 commandBuffer.SetGlobalDepthBias(0f, 0f);
290 float f = 1f - commonSettings.cascadeFade;
291 commandBuffer.SetGlobalVector(ShaderProperties._ShadowsDistanceFade_ID, new Vector4(
292 1f / mainCameraRender.maxShadowDistance,
293 1f / commonSettings.distanceFade,
294 1f / (1f - f * f))
295 );
296 commandBuffer.SetGlobalMatrixArray(ShaderProperties._DirectionalShadowMatrixs_ID, dirShadowMatrixs);
297 commandBuffer.SetGlobalInt(ShaderProperties._CascadeCount_ID, commonSettings.cascadeCount);
298 commandBuffer.SetGlobalVectorArray(ShaderProperties._CascadeCullingSpheres_ID, cascadeCullingSphere);
299 commandBuffer.SetGlobalVectorArray(ShaderProperties._CascadeDatas_ID, cascadeDatas);
300 commandBuffer.EndSample(BufferName);
301 ExecuteCommandBuffer();
302 }

304 private void SetCascadeData(int cascadeIndex, Vector4 cullingSphere, int tileSize)
305 {
306 float texelSize = 2f * cullingSphere.w / tileSize;
307 float filterSize = texelSize * (1 + 1f); // 1 means PCF3x3
308 cullingSphere.w -= filterSize;
309 cullingSphere.w *= cullingSphere.w;
310 Vector3 spherePos = mainCameraRender.worldToViewMatrix * new Vector4(cullingSphere.x, cullingSphere.y, cullingSphere.z, 1f);
311 cullingSphere = new Vector4(spherePos.x, spherePos.y, spherePos.z, cullingSphere.w);
312 cascadeCullingSphere[cascadeIndex] = cullingSphere;
313 cascadeDatas[cascadeIndex] = new Vector4(1f / cullingSphere.w, filterSize * 1.4142136f);
314 }

316 private Matrix4x4 ConvertToShadowMapTileMatrix(Matrix4x4 m, Vector2 offset, float scale)
317 {
318 if (SystemInfo.usesReversedZBuffer)
319 {
320 m.m20 = -m.m20;
321 m.m21 = -m.m21;
322 m.m22 = -m.m22;
323 m.m23 = -m.m23;
324 }
325
326 m.m00 = (0.5f * (m.m00 + m.m30) + offset.x * m.m30) * scale;
327 m.m01 = (0.5f * (m.m01 + m.m31) + offset.x * m.m31) * scale;
328 m.m02 = (0.5f * (m.m02 + m.m32) + offset.x * m.m32) * scale;
329 m.m03 = (0.5f * (m.m03 + m.m33) + offset.x * m.m33) * scale;
330 m.m10 = (0.5f * (m.m10 + m.m30) + offset.y * m.m30) * scale;
331 m.m11 = (0.5f * (m.m11 + m.m31) + offset.y * m.m31) * scale;
332 m.m12 = (0.5f * (m.m12 + m.m32) + offset.y * m.m32) * scale;
333 m.m13 = (0.5f * (m.m13 + m.m33) + offset.y * m.m33) * scale;
334 m.m20 = 0.5f * (m.m20 + m.m30);
335 m.m21 = 0.5f * (m.m21 + m.m31);
336 m.m22 = 0.5f * (m.m22 + m.m32);
337 m.m23 = 0.5f * (m.m23 + m.m33);
338 return m;
339 }

349 {
350 int shadowMapSize = commonSettings.otherLightShadowMapSize;
351 shadowMapSizes.z = shadowMapSize;
352 shadowMapSizes.w = 1f / shadowMapSize;
353 commandBuffer.GetTemporaryRT(ShaderProperties._OtherShadowMap_ID, shadowMapSize, shadowMapSize, commonSettings.otherLightShadowMapBits, FilterMode.Bilinear, RenderTextureFormat.Shadowmap, RenderTextureReadWrite.Linear, 1, false, RenderTextureMemoryless.Color);
354 commandBuffer.SetRenderTarget(ShaderProperties._OtherLightShadowMap_TargetID, RenderBufferLoadAction.DontCare, RenderBufferStoreAction.DontCare, RenderBufferLoadAction.DontCare, RenderBufferStoreAction.Store);
355 commandBuffer.ClearRenderTarget(true, false, Color.clear);
356 commandBuffer.SetGlobalInt(ShaderProperties._ShadowPancaking_ID, 0);

361 int split = tiles <= 1 ? 1 : tiles <= 4 ? 2 : tiles <= 16 ? 4 : 8;

380 private void RenderPointShadow(int index, int split, int tileSize)
381 {
382 ShadowedOtherLight light = shadowedOtherLights[index];
383 var shadowSettings = new ShadowDrawingSettings(cullingResults, light.visibleLightIndex, BatchCullingProjectionType.Perspective);
384 float texelSize = 2f / tileSize;
385 float filterSize = texelSize * (1 + 1f); // 1 means PCF3x3
386 float bias = light.normalBias * filterSize * 1.4142136f;
387 float tileScale = 1f / split;
388 float fovBias = Mathf.Atan(1f + bias + filterSize) * Mathf.Rad2Deg * 2f - 90f;
389 for(int i = 0; i < 6; ++i)
390 {
391 cullingResults.ComputePointShadowMatricesAndCullingPrimitives
392 (
393 light.visibleLightIndex, (CubemapFace)i, fovBias,
394 out Matrix4x4 viewMatrix, out Matrix4x4 projMatrix,
395 out ShadowSplitData splitData
396 );
397 viewMatrix.m11 = -viewMatrix.m11;
398 viewMatrix.m12 = -viewMatrix.m12;
399 viewMatrix.m13 = -viewMatrix.m13;
400 shadowSettings.splitData = splitData;
401 int tileIndex = light.tileIndex + i;
402 Vector2 offset = SetTileViewport(tileIndex, split, tileSize);
403 SetOtherTileData(tileIndex, offset, tileScale, bias);
404 // viewToWorldMatrix is for deferred shading => view space lighting
405 // in forward shading, is Matrix.identity
406 otherShadowMatrixs[tileIndex] = ConvertToShadowMapTileMatrix(projMatrix * viewMatrix, offset, tileScale) * viewToWorldMatrix;
407 commandBuffer.SetViewProjectionMatrices(viewMatrix, projMatrix);
408 commandBuffer.SetGlobalDepthBias(1f, 2.5f + light.slopeScaleBias);
409 ExecuteCommandBuffer();
410 context.DrawShadows(ref shadowSettings);
411 commandBuffer.SetGlobalDepthBias(0f, 0f);
412 }
413 }

415 private void RenderSpotShadow(int index, int split, int tileSize)
416 {

419 cullingResults.ComputeSpotShadowMatricesAndCullingPrimitives

421 light.visibleLightIndex,
422 out Matrix4x4 viewMatrix, out Matrix4x4 projMatrix,
423 out ShadowSplitData splitData
424 );
425 shadowSettings.splitData = splitData;
426 float texelSize = 2f / (tileSize * projMatrix.m00);
427 float filterSize = texelSize * (1 + 1f); // 1 means PCF3x3

431 float tileScale = 1f / split;
432 SetOtherTileData(tileIndex, offset, tileScale, bias);
433 otherShadowMatrixs[tileIndex] = ConvertToShadowMapTileMatrix(projMatrix * viewMatrix, offset, tileScale) * viewToWorldMatrix;

436 ExecuteCommandBuffer();
437 context.DrawShadows(ref shadowSettings);
438 commandBuffer.SetGlobalDepthBias(0f, 0f);
