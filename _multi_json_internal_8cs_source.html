469 static Dictionary<string, Type> CreateTypeMap()
470 {
471 var map = new Dictionary<string, Type>();
472 foreach (var type in TypeCache.GetTypesDerivedFrom<JsonObject>())
473 {
474 if (type.FullName != null)
475 {
476 map[type.FullName] = type;
477 }
478 }

480 foreach (var type in TypeCache.GetTypesWithAttribute(typeof(FormerNameAttribute)))
481 {
482 if (type.IsAbstract || !typeof(JsonObject).IsAssignableFrom(type))

484 continue;

488 {
489 var legacyAttribute = (FormerNameAttribute)attribute;

491 }

503 public static List<MultiJsonEntry> Parse(string str)
504 {
505 var result = new List<MultiJsonEntry>();
506 const string separatorStr = "\n\n";
507 var startIndex = 0;
508 var raw = new FakeJsonObject();
509
510 while (startIndex < str.Length)
511 {
512 var jsonBegin = str.IndexOf("{", startIndex, StringComparison.Ordinal);
513 if (jsonBegin == -1)
514 {
515 break;
516 }
517
518 var jsonEnd = str.IndexOf(separatorStr, jsonBegin, StringComparison.Ordinal);
519 if (jsonEnd == -1)
520 {
521 jsonEnd = str.IndexOf("\n\r\n", jsonBegin, StringComparison.Ordinal);
522 if (jsonEnd == -1)
523 {
524 jsonEnd = str.LastIndexOf("}", StringComparison.Ordinal) + 1;
525 }
526 }
527
528 var json = str.Substring(jsonBegin, jsonEnd - jsonBegin);
529
530 JsonUtility.FromJsonOverwrite(json, raw);
531 if (startIndex != 0 && string.IsNullOrWhiteSpace(raw.type))
532 {
533 throw new InvalidOperationException($"Type is null or whitespace in JSON:\n{json}");
534 }
535
536 result.Add(new MultiJsonEntry(raw.type, raw.id, json));
537 raw.Reset();
538
539 startIndex = jsonEnd + separatorStr.Length;
540 }
541
542 return result;
543 }

556 public static JsonObject CreateInstanceForDeserialization(string typeString)
557 {
558 if(!k_TypeMap.TryGetValue(typeString, out var type))
559 {
560 return new UnknownJsonObject(typeString);
561 }
562 var output = (JsonObject)Activator.CreateInstance(type, true);
563 //This CreateInstance function is supposed to essentially create a blank copy of whatever class we end up deserializing into.
564 //when we typically create new JsonObjects in all other cases, we want that object to be assumed to be the latest version.
565 //This doesn't work if any json object was serialized before we had the idea of version, as the blank copy would have the
566 //latest version on creation and since the serialized version wouldn't have a version member, it would not get overwritten
567 //and we would automatically upgrade all previously serialized json objects incorrectly and without user action. To avoid this,
568 //we default jsonObject version to 0, and if the serialized value has a different saved version it gets changed and if the serialized
569 //version does not have a different saved value it remains 0 (earliest version)
570 output.ChangeVersion(0);
571 output.OnBeforeDeserialize();
572 return output;
573 }

577 public static void Deserialize(JsonObject root, List<MultiJsonEntry> entries, bool rewriteIds)
578 {
579 if (isDeserializing)
580 {
581 throw new InvalidOperationException("Nested MultiJson deserialization is not supported.");
582 }
583
584 try
585 {
586 isDeserializing = true;
587 currentRoot = root;
588 root.ChangeVersion(0); //Same issue as described in CreateInstance
589 for(var index = 0; index < entries.Count; ++index)
590 {
591 var entry = entries[index];
592 try
593 {
594 JsonObject value = null;
595 if(index == 0)
596 {
597 value = root;
598 }
599 else
600 {
601 value = CreateInstanceForDeserialization(entry.type);
602 }
603
604 var id = entry.id;
605
606 if(id != null)
607 {
608 // Need to make sure that references looking for the old ID will find it in spite of
609 // ID rewriting.
610 valueMap[id] = value;
611 }
612
613 if(rewriteIds || entry.id == null)
614 {
615 id = value.objectId;
616 entries[index] = new MultiJsonEntry(entry.type, id, entry.json);
617 valueMap[id] = value;
618 }
619
620 s_ObjectIdField.SetValue(value, id);
621 }
622 catch(Exception e)
623 {
624 // External code could throw exceptions, but we don't want that to fail the whole thing.
625 // Potentially, the fallback type should also be used here.
626 Debug.LogException(e);
627 }
628 }
629
630 s_Entries = entries;
631
632 // Not a foreach because `entries` can be populated by calls to `Enqueue` as we go.
633 for(int i = 0; i < entries.Count; ++i)
634 {
635 var entry = entries[i];
636 try
637 {
638 var value = valueMap[entry.id];
639 value.Deserailize(entry.type, entry.json);
640 // Set ID again as it could be overwritten from JSON.
641 s_ObjectIdField.SetValue(value, entry.id);
642 value.OnAfterDeserialize(entry.json);
643 }
644 catch(Exception e)
645 {
646 if (!String.IsNullOrEmpty(entry.id))
647 {
648 var value = valueMap[entry.id];
649 if(value != null)
650 {
651 Debug.LogError($"Exception thrown while deserialize object of type {entry.type}: {e.Message}");
652 }
653 }
654 Debug.LogException(e);
655 }
656 }
657
658 s_Entries = null;
659
660 foreach(var entry in entries)
661 {
662 try
663 {
664 var value = valueMap[entry.id];
665 value.OnAfterMultiDeserialize(entry.json);
666 }
667 catch(Exception e)
668 {
669 Debug.LogException(e);
670 }
671 }
672 }
673 finally
674 {
675 valueMap.Clear();
676 currentRoot = null;
677 isDeserializing = false;
678 }

681 public static string Serialize(JsonObject mainObject)
682 {
683 if (isSerializing)
684 {
685 throw new InvalidOperationException("Nested MultiJson serialization is not supported.");
686 }
687
688 try
689 {
690 isSerializing = true;
691
692 serializedSet.Add(mainObject.objectId);
693 serializationQueue.Add(mainObject);
694
695 var idJsonList = new List<(string, string)>();
696
697 // Not a foreach because the queue is populated by `JsonData<T>`s as we go.
698 for(var i = 0; i < serializationQueue.Count; ++i)
699 {
700 var value = serializationQueue[i];
701 var json = value.Serialize();
702 idJsonList.Add((value.objectId, json));
703 }
704
705 if(jsonBlobs.TryGetValue(mainObject.objectId, out var blobs))
706 {
707 foreach(var blob in blobs)
708 {
709 if (!idJsonList.Contains((blob.Key, blob.Value)))
710 idJsonList.Add((blob.Key, blob.Value));
711 }
712 }
713
714 idJsonList.Sort((x, y) =>
715 // Main object needs to be placed first
716 x.Item1 == mainObject.objectId ? -1 :
717 y.Item1 == mainObject.objectId ? 1 :
718 // We sort everything else by ID to consistently maintain positions in the output
719 x.Item1.CompareTo(y.Item1));
720
721 const string k_NewLineString = "\n";
722 var sb = new StringBuilder();
723 foreach(var (id, json) in idJsonList)
724 {
725 sb.Append(json);
726 sb.Append(k_NewLineString);
727 sb.Append(k_NewLineString);
728 }
729
730 return sb.ToString();
731 }
732 finally
733 {
734 serializationQueue.Clear();
735 serializedSet.Clear();
736 isSerializing = false;
737 }
738 }
