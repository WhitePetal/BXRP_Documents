22 private void CollectLightDatas()
23 {
24 NativeArray<VisibleLight> visibleLights = cullingResults.visibleLights;
25 dirLightCount = 0;
26 clusterLightCount = 0;
27 for(int visbileLightIndex = 0; visbileLightIndex < visibleLights.Length; ++visbileLightIndex)
28 {
29 if (dirLightCount >= maxDirLightCount && clusterLightCount >= maxClusterLightCount) break;
30 ref var visibleLight = ref visibleLights.UnsafeElementAtMutable(visbileLightIndex);
31 LightBakingOutput lightBaking = visibleLight.light.bakingOutput;
32 if (lightBaking.lightmapBakeType == LightmapBakeType.Baked) continue;
33 switch (visibleLight.lightType)
34 {
35 case LightType.Directional:
36 if(dirLightCount < maxDirLightCount)
37 {
38 SetupDirectionalLight(dirLightCount++, visbileLightIndex, ref visibleLight);
39 }
40 break;
41 case LightType.Point:
42 if(clusterLightCount < maxClusterLightCount)
43 {
44 SetupPointLight(clusterLightCount++, visbileLightIndex, ref visibleLight);
45 }
46 break;
47 case LightType.Spot:
48 if(clusterLightCount < maxClusterLightCount)
49 {
50 SetupSpotLight(clusterLightCount++, visbileLightIndex, ref visibleLight);
51 }
52 break;
53 }
54 }

107 private void SetupSpotLight(int clusterLightIndex, int visibleLightIndex, ref VisibleLight visibleLight)
108 {
109 Matrix4x4 localToWorld = visibleLight.localToWorldMatrix;
110 float range = visibleLight.range;
111 Vector4 lightSphere = localToWorld.GetColumn(3);
112 float threshold = range * range;
113 lightSphere.w = 1f / threshold;
114 threshold = lightSphere.w;
115
116 Vector4 lightDir = -localToWorld.GetColumn(2);
117 Vector4 lightRight = localToWorld.GetColumn(0);
118 Vector4 lightUp = localToWorld.GetColumn(1);
119 float outerRad = Mathf.Deg2Rad * 0.5f * visibleLight.spotAngle;
120 float outerCos = Mathf.Cos(outerRad);
121 float outerSin = Mathf.Sin(outerRad);
122 float angleRangeInv = 1f / Mathf.Max(1f - outerCos, 0.001f);
123
124 float sinRange = outerSin * range;
125 Vector4 upward = lightUp * sinRange;
126 Vector4 rtward = lightRight * sinRange;
127 Vector4 p0 = lightSphere;
128 Vector4 pf = lightSphere - lightDir * range;
129 Vector4 p1 = pf + upward + rtward;
130 Vector4 p2 = pf + upward - rtward;
131 Vector4 p3 = pf - upward + rtward;
132 Vector4 p4 = pf - upward - rtward;
133
134 Vector4 maxBound = Vector4.Max(p4, Vector4.Max(p3, Vector4.Max(p2, Vector4.Max(p1, p0))));
135 Vector4 minBound = Vector4.Min(p4, Vector4.Min(p3, Vector4.Min(p2, Vector4.Min(p1, p0))));
136
137 otherLightSpheres[clusterLightIndex] = lightSphere;
138 clusterLightMaxBounds[clusterLightIndex] = maxBound;
139 clusterLightMinBounds[clusterLightIndex] = minBound;
140 otherLightDirections[clusterLightIndex] = lightDir;
141 otherLightThresholds[clusterLightIndex] = new Vector4(1f / (1f - threshold), threshold, angleRangeInv, -outerCos * angleRangeInv);
142 otherLightColors[clusterLightIndex] = visibleLight.finalColor.gamma;
143 otherShadowDatas[clusterLightIndex] = shadows.SaveOtherShadows(visibleLight.light, visibleLightIndex, clusterLightIndex);
144 otherLights[clusterLightIndex] = visibleLight;
145 }
