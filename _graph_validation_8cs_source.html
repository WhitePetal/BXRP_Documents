16 private static SubTarget GetActiveSubTarget(Target target)
17 {
18 // All targets have an active subtarget but it's not accessible from GeometryGraph
19 // so we use reflection to access it
20
21 var type = target.GetType();
22 if (!s_ActiveSubTarget.TryGetValue(type, out var activeSubTarget))
23 {
24 activeSubTarget = type.GetField("m_ActiveSubTarget", BindingFlags.Instance | BindingFlags.NonPublic);
25 s_ActiveSubTarget.Add(type, activeSubTarget);
26 }
27 if (activeSubTarget != null)
28 {
29 var jsonData = activeSubTarget.GetValue(target);
30 if (jsonData != null)
31 {
32 return s_JsonData.GetValue(jsonData) as SubTarget;
33 }
34 }
35 return null;
36 }

38 public static void ValidateNode(AbstractGeometryNode node)
39 {
40 Type t = node.GetType();
41 node.ValidateNode();
42 if (!(node is BlockNode))
43 {
44 bool disallowedByAnyTargets = false;
45 bool disallowedByAllTargets = true;
46 bool disallowedByAnySubTarget = false;
47 IEnumerable<Target> targets = node.owner.activeTargets;
48 if (node.owner.isSubGraph)
49 {
50 targets = node.owner.allPotentialTargets;
51 }
52 foreach (var target in targets)
53 {
54 //if at least one target doesn't allow a node, it is considered invalid
55 if (!target.IsNodeAllowedByTarget(t))
56 {
57 disallowedByAnyTargets = true;
58 node.isValid = false;
59 node.owner.AddValidationError(node.objectId, $"{node.name} Node is not allowed by {target.displayName} implementation", GeometryCompilerMessageSeverity.Warning);
60 node.owner.m_UnsupportedTargets.Add(target);
61 }
62 //at least one target does allow node, not going to be explicitly set inactive
63 else
64 {
65 disallowedByAllTargets = false;
66 }
67
68 var subtarget = GetActiveSubTarget(target);
69 if (subtarget != null && !subtarget.IsNodeAllowedBySubTarget(t))
70 {
71 disallowedByAnySubTarget = true;
72 node.isValid = false;
73 node.owner.AddValidationError(node.objectId, $"{node.name} Node is not allowed by {subtarget.displayName} implementation", GeometryCompilerMessageSeverity.Error);
74 }
75 }
76 if (!disallowedByAnyTargets && !disallowedByAnySubTarget)
77 {
78 node.isValid = true;
79 }
80
81 //Set ActiveState based on if all targets disallow this node
82 if (disallowedByAllTargets)
83 {
84 node.SetOverrideActiveState(AbstractGeometryNode.ActiveState.ExplicitInactive);
85 node.owner.AddValidationError(node.objectId, $"{node.name} Node is not allowed by any active targets, and will not be used in generation", GeometryCompilerMessageSeverity.Warning);
86 }
87 else
88 {
89 node.SetOverrideActiveState(AbstractGeometryNode.ActiveState.Implicit);
90 }
91 }
