170 [GenerateTestsForBurstCompatibility]
171 public unsafe JobHandle OnPerformCulling(BatchRendererGroup rendererGroup, BatchCullingContext cullingContext, BatchCullingOutput cullingOutput, IntPtr userContext)
172 {
173 // UnsafeUtility.Malloc() requires an alignment, so use the largest integer type's alignment
174 // which is a reasonable default.
175 int alignment = UnsafeUtility.AlignOf<long>();
176
177 // Acquire a pointer to the BatchCullingOutputDrawCommands struct so you can easily
178 // modify it directly.
179 var drawCommands = (BatchCullingOutputDrawCommands*)cullingOutput.drawCommands.GetUnsafePtr();
180
181 // Allocate memory for the output arrays. In a more complicated implementation, you would calculate
182 // the amount of memory to allocate dynamically based on what is visible.
183 // This example assumes that all of the instances are visible and thus allocates
184 // memory for each of them. The necessary allocations are as follows:
185 // - a single draw command (which draws kNumInstances instances)
186 // - a single draw range (which covers our single draw command)
187 // - kNumInstances visible instance indices.
188 // You must always allocate the arrays using Allocator.TempJob.
189 drawCommands->drawCommands = (BatchDrawCommand*)UnsafeUtility.Malloc(UnsafeUtility.SizeOf<BatchDrawCommand>(), alignment, Allocator.TempJob);
190 drawCommands->drawRanges = (BatchDrawRange*)UnsafeUtility.Malloc(UnsafeUtility.SizeOf<BatchDrawRange>(), alignment, Allocator.TempJob);
191 drawCommands->visibleInstances = (int*)UnsafeUtility.Malloc(kNumInstances * sizeof(int), alignment, Allocator.TempJob);
192 drawCommands->drawCommandPickingInstanceIDs = null;
193
194 drawCommands->drawCommandCount = 1;
195 drawCommands->drawRangeCount = 1;
196 drawCommands->visibleInstanceCount = kNumInstances;
197
198 // This example doens't use depth sorting, so it leaves instanceSortingPositions as null.
199 drawCommands->instanceSortingPositions = null;
200 drawCommands->instanceSortingPositionFloatCount = 0;
201
202 // Configure the single draw command to draw kNumInstances instances
203 // starting from offset 0 in the array, using the batch, material and mesh
204 // IDs registered in the Start() method. It doesn't set any special flags.
205 drawCommands->drawCommands[0].visibleOffset = 0;
206 drawCommands->drawCommands[0].visibleCount = kNumInstances;
207 drawCommands->drawCommands[0].batchID = m_BatchID;
208 drawCommands->drawCommands[0].materialID = m_MaterialID;
209 drawCommands->drawCommands[0].meshID = m_MeshID;
210 drawCommands->drawCommands[0].submeshIndex = 0;
211 drawCommands->drawCommands[0].splitVisibilityMask = 0xff;
212 drawCommands->drawCommands[0].flags = BatchDrawCommandFlags.None;
213 drawCommands->drawCommands[0].sortingPosition = 0;
214
215 // Configure the single draw range to cover the single draw command which
216 // is at offset 0.
217 drawCommands->drawRanges[0].drawCommandsBegin = 0;
218 drawCommands->drawRanges[0].drawCommandsCount = 1;
219
220 // This example doesn't care about shadows or motion vectors, so it leaves everything
221 // at the default zero values, except the renderingLayerMask which it sets to all ones
222 // so Unity renders the instances regardless of mask settings.
223 drawCommands->drawRanges[0].filterSettings = new BatchFilterSettings { renderingLayerMask = 0xffffffff, };
224
225 // Finally, write the actual visible instance indices to the array. In a more complicated
226 // implementation, this output would depend on what is visible, but this example
227 // assumes that everything is visible.
228 for(int i = 0; i < kNumInstances; ++i)
229 {
230 drawCommands->visibleInstances[i] = i;
231 }
232
233 // This simple example doesn't use jobs, so it returns an empty JobHandle.
234 // Performance-sensitive applications are encouraged to use Burst jobs to implement
235 // culling and draw command output. In this case, this function returns a
236 // handle here that completes when the Burst jobs finish.
237 return new JobHandle();
238 }

250 private void AllocateInstanceDataBuffer()

252 m_InstanceData = new GraphicsBuffer(GraphicsBuffer.Target.Raw, BufferCountForInstances(kBytesPerInstance, kNumInstances, kExtraBytes), sizeof(int));

254
255 // Raw buffers are allocated in ints. This is a utility method that calculates
256 // the required number of ints for the data.
257 private int BufferCountForInstances(int bytesPerInstance, int numInstances, int extraBytes = 0)
258 {
259 // Round byte counts to int multiples
260 bytesPerInstance = (bytesPerInstance + sizeof(int) - 1) / sizeof(int) * sizeof(int);
261 extraBytes = (extraBytes + sizeof(int) - 1) / sizeof(int) * sizeof(int);
262 int totalBytes = bytesPerInstance * numInstances + extraBytes;
263 return totalBytes / sizeof(int);
264 }

266 private void PopulateInstanceDataBuffer()
267 {
268 // Place a zero matrix at the start of the instance data buffer, so loads from address 0 return zero.
269 var zero = new Matrix4x4[1] { Matrix4x4.zero };
270
271 // Create transform matrices for three example instances.
272 var matrices = new Matrix4x4[kNumInstances]
273 {
274 Matrix4x4.Translate(new Vector3(-2, 0, 0)),
275 Matrix4x4.Translate(new Vector3(0, 0, 0)),
276 Matrix4x4.Translate(new Vector3(2, 0, 0))
277 };
278
279 // Convert the transform matrices into the packed format that the shader expects.
280 var objectToWorld = new PackedMatrix[kNumInstances]
281 {
282 new PackedMatrix(matrices[0]),
283 new PackedMatrix(matrices[1]),
284 new PackedMatrix(matrices[2])
285 };
286
287 // Also create packed inverse matrices.
288 var worldToObject = new PackedMatrix[kNumInstances]
289 {
290 new PackedMatrix(matrices[0].inverse),
291 new PackedMatrix(matrices[1].inverse),
292 new PackedMatrix(matrices[2].inverse)
293 };
294
295 // Make all instances have unique colors.
296 var colors = new Vector4[kNumInstances]
297 {
298 new Vector4(1, 0, 0, 1),
299 new Vector4(0, 1, 0, 1),
300 new Vector4(0, 0, 1, 1)
301 };
302
303 // In this simple example, the instance data is placed into the buffer like this:
304 // Offset | Description
305 // 0 | 64 bytes of zeroes, so loads from address 0 return zeroes
306 // 64 | 32 uninitialized bytes to make working with SetData easier, otherwise unnecessary
307 // 96 | unity_ObjectToWorld, three packed float3x4 matrices
308 // 240 | unity_WorldToObject, three packed float3x4 matrices
309 // 384 | _BaseColor, three float4s
310
311 // Calculates start addresses for the different instanced properties. unity_ObjectToWorld starts
312 // at address 96 instead of 64, because the computeBufferStartIndex parameter of SetData
313 // is expressed as source array elements, so it is easier to work in multiples of sizeof(PackedMatrix).
314 uint byteAddressObjectToWorld = kSizeOfPackedMatrix * 2;
315 uint byteAddressWorldToObject = byteAddressObjectToWorld + kSizeOfPackedMatrix * kNumInstances;
316 uint byteAddressColor = byteAddressWorldToObject + kSizeOfPackedMatrix * kNumInstances;
317
318 // Upload the instance data to the GraphicsBuffer so the shader can load them.
319 m_InstanceData.SetData(zero, 0, 0, 1);
320 m_InstanceData.SetData(objectToWorld, 0, (int)(byteAddressObjectToWorld / kSizeOfPackedMatrix), kNumInstances);
321 m_InstanceData.SetData(worldToObject, 0, (int)(byteAddressWorldToObject / kSizeOfPackedMatrix), kNumInstances);
322 m_InstanceData.SetData(colors, 0, (int)(byteAddressColor / kSizeOfFloat4), kNumInstances);
323
324 // Set up metadata values to point to the instance data. Set the most significant bit 0x80000000 in each
325 // which instructs the shader that the data is an array with one value per instance, indexed by the instance index.
326 // Any metadata values that the shader uses that are not set here will be 0. When a value of 0 is used with
327 // UNITY_ACCESS_DOTS_INSTANCED_PROP (i.e. without a default), the shader interprets the
328 // 0x00000000 metadata value and loads from the start of the buffer. The start of the buffer is
329 // a zero matrix so this sort of load is guaranteed to return zero, which is a reasonable default value.
330 var metadata = new NativeArray<MetadataValue>(3, Allocator.Temp);
331 metadata[0] = new MetadataValue { NameID = Shader.PropertyToID("unity_ObjectToWorld"), Value = 0x80000000 | byteAddressObjectToWorld };
332 metadata[1] = new MetadataValue { NameID = Shader.PropertyToID("unity_WorldToObject"), Value = 0x80000000 | byteAddressWorldToObject };
333 metadata[2] = new MetadataValue { NameID = Shader.PropertyToID("_DiffuseColor"), Value = 0x80000000 | byteAddressColor };
334
335 // Finally, create a batch for the instances and make the batch use the GraphicsBuffer with the
336 // instance data as well as the metadata values that specify where the properties are.
337 m_BatchID = m_BRG.AddBatch(metadata, m_InstanceData.bufferHandle);
338 }
339 }
