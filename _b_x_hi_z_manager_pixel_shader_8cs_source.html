108 private void Setup(BXMainCameraRenderBase mainRender)
109 {
110 isViewCamera = mainRender.camera.cameraType == CameraType.SceneView;
111 if (isViewCamera) return;
112 this.mat = mainRender.commonSettings.hizPixelMat;
113 this.projectionMatrix = GL.GetGPUProjectionMatrix(mainRender.camera.projectionMatrix * mainRender.camera.worldToCameraMatrix, false);
114 mipCount = 0;
115 this.screenSize = math.int2(mainRender.width, mainRender.height);
116 int texW = ((mainRender.width >> 3) + (mainRender.width >> 4)) + 1;
117 int texH = mainRender.height >> 3;
118 //texW = Mathf.IsPowerOfTwo(texW) ? texW : Mathf.NextPowerOfTwo(texW);
119 //texH = Mathf.IsPowerOfTwo(texH) ? texH : Mathf.NextPowerOfTwo(texH);
120 if (texW != texSize.x || texH != texSize.y)
121 {
122 GenerateHizBuffers(texW, texH);
123 texSize = math.int2(texW, texH);
124 }
125 if (!isReadbacking)
126 {
127 cullingObjectCount = 0;
128 }

131 private void Render(CommandBuffer commandBuffer)
132 {
133 if (isViewCamera) return;
134 commandBuffer.BeginSample(SampleName);
135 float2 mipSize = math.float2(screenSize.x >> 1, screenSize.y >> 1);
136 commandBuffer.GetTemporaryRT(_HiZTempMaps_ID[0], (int)mipSize.x, (int)mipSize.y, 0, FilterMode.Point, RenderTextureFormat.RFloat, RenderTextureReadWrite.Linear, 1);
137 DrawPostProcess(commandBuffer, BXShaderPropertyIDs._EncodeDepthBuffer_TargetID, _HiZTempMaps_TargetID[0], mat, 0);
138 float4 mipOffset = math.float4(0, 0, mipSize.x, mipSize.y);
139 mipSizes[mipCount++] = mipOffset;
140 int index = 1;
141 bool row = true;
142 while (mipSize.x > 1 && mipSize.y > 1)
143 {
144 if (mipSize.x <= 1 || mipSize.y <= 1) break;
145 if (row)
146 {
147 mipOffset += math.float4(mipSize.x, 0, 0, 0);
148 }
149 else
150 {
151 mipOffset += math.float4(0, mipSize.y, 0, 0);
152 }
153 row = !row;
154 mipSize *= 0.5f;
155 commandBuffer.GetTemporaryRT(_HiZTempMaps_ID[index], Mathf.CeilToInt(mipSize.x), Mathf.CeilToInt(mipSize.y), 0, FilterMode.Point, RenderTextureFormat.RFloat, RenderTextureReadWrite.Linear, 1);
156 DrawPostProcess(commandBuffer, _HiZTempMaps_TargetID[index-1], _HiZTempMaps_TargetID[index], mat, 1);
157 mipSizes[mipCount++] = math.float4(mipOffset.xy, mipSize.xy);
158 ++index;
159 }
160 commandBuffer.SetRenderTarget(hizMap, RenderBufferLoadAction.DontCare, RenderBufferStoreAction.Store);
161 commandBuffer.ClearRenderTarget(false, true, Color.clear);
162 Vector2 baseOffset = mipSizes[2];
163 for (int i = 2; i < mipCount; ++i)
164 {
165 float4 mipPort = mipSizes[i];
166 Vector2 offset = mipPort.xy;
167 offset = offset - baseOffset;
168 mipSizes[i - 2] = new Vector4(offset.x, offset.y, mipPort.z, mipPort.w);
169 commandBuffer.SetViewport(new Rect(offset, mipPort.zw));
170 commandBuffer.SetGlobalTexture(BXShaderPropertyIDs._PostProcessInput_ID, _HiZTempMaps_TargetID[i]);
171 commandBuffer.DrawProcedural(Matrix4x4.identity, mat, 2, MeshTopology.Triangles, 3);
172 }
173 for(int i = 0; i < mipCount; ++i)
174 {
175 commandBuffer.ReleaseTemporaryRT(_HiZTempMaps_ID[i]);
176 }
177
178 if (!isReadbacking && cullingObjectCount > 0)
179 {
180 commandBuffer.SetRenderTarget(hizCullResultRT, RenderBufferLoadAction.DontCare, RenderBufferStoreAction.Store);
181 commandBuffer.ClearRenderTarget(false, true, Color.clear);
182 commandBuffer.SetGlobalTexture(BaseShaderProperties._BoundCentersTex_ID, boundCentersTex);
183 commandBuffer.SetGlobalTexture(BaseShaderProperties._BoundSizesTex_ID, boundSizesTex);
184 commandBuffer.SetGlobalVector(BaseShaderProperties._HizTexSize_ID, new Vector4(texSize.x, texSize.y));
185 commandBuffer.SetGlobalTexture(BaseShaderProperties._HizMapInput_ID, hizMap);
186 commandBuffer.SetGlobalVector(BaseShaderProperties._HizParams_ID, new Vector4(screenSize.x, screenSize.y, mipCount - 3, cullingObjectCount));
187 commandBuffer.SetGlobalMatrix(BaseShaderProperties._HizProjectionMatrix_ID, projectionMatrix);
188 commandBuffer.SetGlobalVectorArray(BaseShaderProperties._HizMipSize_ID, mipSizes);
189 commandBuffer.DrawProcedural(Matrix4x4.identity, mat, 3, MeshTopology.Triangles, 3);
190
191
192 hizReadbackDatas = new NativeArray<float>(dataTexWidth * dataTexHeight, Allocator.Persistent);
193 commandBuffer.RequestAsyncReadbackIntoNativeArray(ref hizReadbackDatas, hizCullResultRT, ReadBack);
194 isReadbacking = true;
195 }
196 commandBuffer.EndSample(SampleName);
197 }

199 public override void Register(Renderer renderer, int instanceID)
200 {
201 if (isViewCamera || isReadbacking) return;
202 rendererDic[instanceID] = renderer;
203 var bounds = renderer.bounds;
204 Vector3 center = bounds.center;
205 Vector3 size = bounds.size;
206 CullingObjectData data = new CullingObjectData()
207 {
208 boundCenter = center,
209 boundSize = size,
210 instanceID = instanceID
211 };
212 int y = cullingObjectCount / 64;
213 int x = cullingObjectCount % 64;
214 boundCentersTex.SetPixel(x, y, new Color(center.x, center.y, center.z, 1f));
215 boundSizesTex.SetPixel(x, y, new Color(size.x, size.y, size.z, 1f));
216 cullingObjects[cullingObjectCount++] = data;
217 }
