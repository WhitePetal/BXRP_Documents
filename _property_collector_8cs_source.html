55 private static bool EquivalentHLSLProperties(AbstractGeometryProperty a, AbstractGeometryProperty b)
56 {
57 bool equivalent = true;
58 var bHLSLProps = new List<HLSLProperty>();
59 b.ForeachHLSLProperty(bh => bHLSLProps.Add(bh));
60 a.ForeachHLSLProperty(ah =>
61 {
62 var i = bHLSLProps.FindIndex(bh => bh.name == ah.name);
63 if (i < 0)
64 equivalent = false;
65 else
66 {
67 var bh = bHLSLProps[i];
68 if (!ah.ValueEquals(bh))
69 equivalent = false;
70 bHLSLProps.RemoveAt(i);
71 }
72 });
73 return equivalent && (bHLSLProps.Count == 0);
