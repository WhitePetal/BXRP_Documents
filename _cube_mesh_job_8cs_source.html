76 public override JobHandle Schedule(JobHandle dependsOn = default)
77 {
78 if (depenedJobs != null && depenedJobs.Length > 0)
79 {
80 for (int i = 0; i < depenedJobs.Length; ++i)
81 {
82 dependsOn = JobHandle.CombineDependencies(dependsOn, depenedJobs[i].Schedule());
83 }
84 }
85
86 size = m_SizeValueFrom == ValueFrom.Default ? m_SizeValueDefault : depenedJobs[0].GetFloat3(m_SizeValueID);
87 verticesX = m_VerticesXValueFrom == ValueFrom.Default ? m_VerticesXValueDefault : depenedJobs[1].GetInt(m_VerticesXValueID);
88 verticesY = m_VerticesYValueFrom == ValueFrom.Default ? m_VerticesYValueDefault : depenedJobs[2].GetInt(m_VerticesYValueID);
89 verticesZ = m_VerticesZValueFrom == ValueFrom.Default ? m_VerticesZValueDefault : depenedJobs[3].GetInt(m_VerticesZValueID);
90
91 JobHandle jobHandle;
92 int dimensions = ((verticesX - 1) > 0 ? 1 : 0) + ((verticesY - 1) > 0 ? 1 : 0) + ((verticesZ - 1) > 0 ? 1 : 0);
93
94 if (dimensions == 0)
95 {
96 (jobHandle, this.mesh) = mesh_primitive_line.create_line_mesh(float3.zero, float3.zero, 1, dependsOn);
97 }
98 else if(dimensions == 1)
99 {
100 float3 start;
101 float3 delta;
102 if (verticesX > 1)
103 {
104 start = new float3(-size.x * 0.5f, 0f, 0f);
105 delta = new float3(size.x / (verticesX - 1), 0f, 0f);
106 }
107 else if (verticesY > 1)
108 {
109 start = new float3(0f, -size.y * 0.5f, 0f);
110 delta = new float3(0f, size.y / (verticesY - 1), 0f);
111 }
112 else
113 {
114 start = new float3(0f, 0f, -size.z * 0.5f);
115 delta = new float3(0f, 0f, size.z / (verticesZ - 1));
116 }
117
118 (jobHandle, this.mesh) = mesh_primitive_line.create_line_mesh(start, delta, verticesX * verticesY * verticesZ, dependsOn);
119 }
120 else if(dimensions == 2)
121 {
122 // XY Plane
123 if (verticesZ <= 1)
124 {
125 (jobHandle, this.mesh) = mesh_primitive_grid.create_grid_mesh(verticesX, verticesY, size.x, size.y, dependsOn);
126 }
127 else if (verticesY <= 1)
128 {
129 (jobHandle, this.mesh) = mesh_primitive_grid.create_grid_mesh(verticesX, verticesZ, size.x, size.z, dependsOn);
130 // TODO
131 // transform_mesh
132 }
133 else
134 {
135 (jobHandle, this.mesh) = mesh_primitive_grid.create_grid_mesh(verticesZ, verticesY, size.z, size.y, dependsOn);
136 // TODO
137 // transform_mesh
138 }
139 }
140 else
141 {
142 (jobHandle, this.mesh) = mesh_primitive_cuboid.create_cuboid_mesh(size, verticesX, verticesY, verticesZ, dependsOn);
143 }
144
145 return jobHandle;
146 }
