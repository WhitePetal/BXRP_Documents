235 EditorGUI.BeginChangeCheck();
236 {

273 m_MaterialEditor.ShaderProperty(enableDoubleSidedOn, Styles.doubleSidedLabel);

275
276 m_MaterialEditor.ShaderProperty(enableVertexLM, Styles.vertexLMLabel);
277 if (enableVertexLM.floatValue > 0)
278 {
279 m_MaterialEditor.ShaderProperty(enableVertexLMdir, Styles.vertexLMdirLabel);
280 //if (enableVertexLMdir.floatValue > 0) enableVertexLMSH.floatValue = 0;
281 }
282 if (enableVertexLM.floatValue > 0)
283 {
284 m_MaterialEditor.ShaderProperty(enableVertexLMSH, Styles.vertexLMSHLabel);
285 //if (enableVertexLMSH.floatValue > 0) enableVertexLMdir.floatValue = 0;

287 if (enableVertexLM.floatValue > 0)

299 m_MaterialEditor.ShaderProperty(enableSpecOcclusion, Styles.specOcclusionLabel);
300 }
301 m_MaterialEditor.ShaderProperty(enableBicubic, Styles.bicubicLabel);
302 m_MaterialEditor.ShaderProperty(enablePSHN, Styles.pshnLabel);
303
304 try
305 {
306 m_MaterialEditor.ShaderProperty(enableVolumes, Styles.volLabel);
307 if (enableVolumes.floatValue > 0)

309 var prevAssignedVolume = assignedVolume;
310 assignedVolume = EditorGUILayout.ObjectField(volume0.textureValue == null ? "Assign volume" : "Assign different volume", assignedVolume, typeof(BakeryVolume), true) as BakeryVolume;

374 public override void AssignNewShaderToMaterial(Material material, Shader oldShader, Shader newShader)
375 {
376 // _Emission property is lost after assigning Standard shader to the material
377 // thus transfer it before assigning the new shader
378 if (material.HasProperty("_Emission"))
379 {
380 material.SetColor("_EmissionColor", material.GetColor("_Emission"));
381 }
382
383 base.AssignNewShaderToMaterial(material, oldShader, newShader);
384
385 if (oldShader == null || !oldShader.name.Contains("Legacy Shaders/"))
386 {
387 SetupMaterialWithBlendMode(material, (BlendMode)material.GetFloat("_Mode"));
388 return;
389 }
390
391 BlendMode blendMode = BlendMode.Opaque;
392 if (oldShader.name.Contains("/Transparent/Cutout/"))
393 {
394 blendMode = BlendMode.Cutout;
395 }
396 else if (oldShader.name.Contains("/Transparent/"))
397 {
398 // NOTE: legacy shaders did not provide physically based transparency
399 // therefore Fade mode
400 blendMode = BlendMode.Fade;
401 }
402 material.SetFloat("_Mode", (float)blendMode);
403
404 DetermineWorkflow(MaterialEditor.GetMaterialProperties(new Material[] { material }));
405 MaterialChanged(material, m_WorkflowMode);
406 }

456 void DoSpecularMetallicArea()
457 {
458 bool hasGlossMap = false;
459 if (m_WorkflowMode == WorkflowMode.Specular)
460 {
461 hasGlossMap = specularMap.textureValue != null;
462 m_MaterialEditor.TexturePropertySingleLine(Styles.specularMapText, specularMap, hasGlossMap ? null : specularColor);
463 }
464 else if (m_WorkflowMode == WorkflowMode.Metallic)
465 {
466 hasGlossMap = metallicMap.textureValue != null;
467 m_MaterialEditor.TexturePropertySingleLine(Styles.metallicMapText, metallicMap, hasGlossMap ? null : metallic);
468 }
469
470 bool showSmoothnessScale = hasGlossMap;
471 if (smoothnessMapChannel != null)
472 {
473 int smoothnessChannel = (int)smoothnessMapChannel.floatValue;
474 if (smoothnessChannel == (int)SmoothnessMapChannel.AlbedoAlpha)
475 showSmoothnessScale = true;
476 }
477
478 int indentation = 2; // align with labels of texture properties
479 m_MaterialEditor.ShaderProperty(showSmoothnessScale ? smoothnessScale : smoothness, showSmoothnessScale ? Styles.smoothnessScaleText : Styles.smoothnessText, indentation);
480
481 ++indentation;
482 if (smoothnessMapChannel != null)
483 m_MaterialEditor.ShaderProperty(smoothnessMapChannel, Styles.smoothnessMapChannelText, indentation);
484 }

486 public static void SetupMaterialWithBlendMode(Material material, BlendMode blendMode)
487 {
488 switch (blendMode)
489 {
490 case BlendMode.Opaque:
491 material.SetOverrideTag("RenderType", "Opaque");
492 material.SetInt("_SrcBlend", (int)UnityEngine.Rendering.BlendMode.One);
493 material.SetInt("_DstBlend", (int)UnityEngine.Rendering.BlendMode.Zero);
494 material.SetInt("_ZWrite", 1);
495 material.DisableKeyword("_ALPHATEST_ON");
496 material.DisableKeyword("_ALPHABLEND_ON");
497 material.DisableKeyword("_ALPHAPREMULTIPLY_ON");
498 material.renderQueue = -1;
499 break;
500 case BlendMode.Cutout:
501 material.SetOverrideTag("RenderType", "TransparentCutout");
502 material.SetInt("_SrcBlend", (int)UnityEngine.Rendering.BlendMode.One);
503 material.SetInt("_DstBlend", (int)UnityEngine.Rendering.BlendMode.Zero);
504 material.SetInt("_ZWrite", 1);
505 material.EnableKeyword("_ALPHATEST_ON");
506 material.DisableKeyword("_ALPHABLEND_ON");
507 material.DisableKeyword("_ALPHAPREMULTIPLY_ON");
508 material.renderQueue = (int)UnityEngine.Rendering.RenderQueue.AlphaTest;
509 break;
510 case BlendMode.Fade:
511 material.SetOverrideTag("RenderType", "Transparent");
512 material.SetInt("_SrcBlend", (int)UnityEngine.Rendering.BlendMode.SrcAlpha);
513 material.SetInt("_DstBlend", (int)UnityEngine.Rendering.BlendMode.OneMinusSrcAlpha);
514 material.SetInt("_ZWrite", 0);
515 material.DisableKeyword("_ALPHATEST_ON");
516 material.EnableKeyword("_ALPHABLEND_ON");
517 material.DisableKeyword("_ALPHAPREMULTIPLY_ON");
518 material.renderQueue = (int)UnityEngine.Rendering.RenderQueue.Transparent;
519 break;
520 case BlendMode.Transparent:
521 material.SetOverrideTag("RenderType", "Transparent");
522 material.SetInt("_SrcBlend", (int)UnityEngine.Rendering.BlendMode.One);
523 material.SetInt("_DstBlend", (int)UnityEngine.Rendering.BlendMode.OneMinusSrcAlpha);
524 material.SetInt("_ZWrite", 0);
525 material.DisableKeyword("_ALPHATEST_ON");
526 material.DisableKeyword("_ALPHABLEND_ON");
527 material.EnableKeyword("_ALPHAPREMULTIPLY_ON");
528 material.renderQueue = (int)UnityEngine.Rendering.RenderQueue.Transparent;
529 break;
530 }

548 static void SetMaterialKeywords(Material material, WorkflowMode workflowMode)
549 {
550 // Note: keywords must be based on Material value not on MaterialProperty due to multi-edit & material animation
551 // (MaterialProperty value might come from renderer material property block)
552 SetKeyword(material, "_NORMALMAP", material.GetTexture("_BumpMap") || material.GetTexture("_DetailNormalMap"));

557 SetKeyword(material, "_PARALLAXMAP", material.GetTexture("_ParallaxMap"));
558 SetKeyword(material, "_DETAIL_MULX2", material.GetTexture("_DetailAlbedoMap") || material.GetTexture("_DetailNormalMap"));
559

561 SetKeyword(material, "_EMISSION", shouldEmissionBeEnabled);
562
563 if (material.HasProperty("_SmoothnessTextureChannel"))
564 {
565 SetKeyword(material, "_SMOOTHNESS_TEXTURE_ALBEDO_CHANNEL_A", GetSmoothnessMapChannel(material) == SmoothnessMapChannel.AlbedoAlpha);
566 }
567
568 // Setup lightmap emissive flags
569 MaterialGlobalIlluminationFlags flags = material.globalIlluminationFlags;
570 if ((flags & (MaterialGlobalIlluminationFlags.BakedEmissive | MaterialGlobalIlluminationFlags.RealtimeEmissive)) != 0)
571 {
572 flags &= ~MaterialGlobalIlluminationFlags.EmissiveIsBlack;
573 if (!shouldEmissionBeEnabled)
574 flags |= MaterialGlobalIlluminationFlags.EmissiveIsBlack;
575
576 material.globalIlluminationFlags = flags;
577 }
