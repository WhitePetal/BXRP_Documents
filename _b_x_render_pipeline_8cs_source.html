176 protected override void Render(ScriptableRenderContext context, Camera[] cameras)
177 {

179 {
180 var camera = cameras[i];

183 mainCameraRender.Render(context, camera, useDynamicBatching, useGPUInstancing,
184 beforeRenderFeatures, onDirShadowRenderFeatures,
185 beforeOpaqueRenderFeatures, afterOpaqueRenderFeatures,
186 beforeTransparentFeatures, afterTransparentFeatures,

189 else

193 // in runtime we always only use main camera
