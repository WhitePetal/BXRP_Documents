252 public sealed override void UpdateNodeAfterDeserialization()
253 {
254 var method = GetFunctionToConvert();
255
256 if (method == null)
257 throw new ArgumentException("Mapped method is null on node" + this);
258
259 if (method.ReturnType != typeof(string))
260 throw new ArgumentException("Mapped function should return string");
261
262 // validate no duplicates
263 var slotAtributes = method.GetParameters().Select(GetSlotAttribute).ToList();
264 if (slotAtributes.Any(x => x == null))
265 throw new ArgumentException("Missing SlotAttribute on " + method.Name);
266
267 if (slotAtributes.GroupBy(x => x.slotId).Any(x => x.Count() > 1))
268 throw new ArgumentException("Duplicate SlotAttribute on " + method.Name);
269
270 List<GeometrySlot> slots = new List<GeometrySlot>();
271 foreach (var par in method.GetParameters())
272 {
273 var attribute = GetSlotAttribute(par);
274 var name = GraphUtil.ConvertCamelCase(par.Name, true);
275
276 GeometrySlot s;
277 if (attribute.binding == Binding.None && !par.IsOut && par.ParameterType == typeof(Color))
278 s = new ColorGeometrySlot(attribute.slotId, name, par.Name, SlotType.Input, attribute.defaultValue ?? Vector4.zero, ColorMode.Default, stageCapability: attribute.stageCapability, hidden: attribute.hidden);
279 else if (attribute.binding == Binding.None && !par.IsOut && par.ParameterType == typeof(ColorRGBA))
280 s = new ColorGeometrySlot(attribute.slotId, name, par.Name, SlotType.Input, attribute.defaultValue ?? Vector4.zero, ColorMode.Default, stageCapability: attribute.stageCapability, hidden: attribute.hidden);
281 else if (attribute.binding == Binding.None && !par.IsOut && par.ParameterType == typeof(ColorRGB))
282 s = new ColorGeometrySlot(attribute.slotId, name, par.Name, SlotType.Input, attribute.defaultValue ?? Vector4.zero, ColorMode.Default, stageCapability: attribute.stageCapability, hidden: attribute.hidden);
283 else if (attribute.binding == Binding.None || par.IsOut)
284 {
285 Debug.Log("convertToSlotValueType: " + ConvertTypeToSlotValueType(par));
286 s = GeometrySlot.CreateGeometrySlot(
287 ConvertTypeToSlotValueType(par),
288 attribute.slotId,
289 name,
290 par.Name,
291 par.IsOut ? SlotType.Output : SlotType.Input,
292 attribute.defaultValue ?? Vector4.zero,
293 geometryStageCapability: attribute.stageCapability,
294 hidden: attribute.hidden);
295 }
296 else
297 s = CreateBoundSlot(attribute.binding, attribute.slotId, name, par.Name, attribute.stageCapability, attribute.hidden);
298 slots.Add(s);
299
300 m_Slots.Add(attribute);
301 }
302 foreach (var slot in slots)
303 {
304 AddSlot(slot);
305 }
306 RemoveSlotsNameNotMatching(slots.Select(x => x.id), true);
307 }

370 public void GenerateNodeShaderCode(ShaderStringBuilder sb, GenerationMode generationMode)
371 {
372 using (var tempSlots = PooledList<GeometrySlot>.Get())
373 {
374 GetOutputSlots(tempSlots);
375 foreach (var outSlot in tempSlots)
376 {
377 sb.AppendLine(outSlot.concreteValueType.ToGeometryString(PrecisionUtil.Token) + " " + GetVariableNameForSlot(outSlot.id) + ";");
378 }
379
380 string call = GetFunctionName() + "(";
381 bool first = true;
382 tempSlots.Clear();
383 GetSlots(tempSlots);
384 tempSlots.Sort((slot1, slot2) => slot1.id.CompareTo(slot2.id));
385 foreach (var slot in tempSlots)
386 {
387 if (!first)
388 {
389 call += ", ";
390 }
391 first = false;
392
393 if (slot.isInputSlot)
394 call += GetSlotValue(slot.id, generationMode);
395 else
396 call += GetVariableNameForSlot(slot.id);
397 }
398 call += ");";
399
400 sb.AppendLine(call);
401 }

446 private string GetFunctionBody(MethodInfo info)
447 {
448 var args = new List<object>();
449 foreach (var param in info.GetParameters())
450 args.Add(GetDefault(param.ParameterType));
451
452 var result = info.Invoke(this, args.ToArray()) as string;
453
454 if (string.IsNullOrEmpty(result))
455 return string.Empty;
456
457 // stomp any newline differences that might try to sneak in via this path
458 result = result.Replace("\r\n", "\n");
459
460 using (var tempSlots = PooledList<GeometrySlot>.Get())
461 {
462 GetSlots(tempSlots);
463 foreach (var slot in tempSlots)
464 {
465 var toReplace = string.Format("{{slot{0}dimension}}", slot.id);
466 var replacement = NodeUtils.GetSlotDimension(slot.concreteValueType);
467 result = result.Replace(toReplace, replacement);
468 }
469 }
470
471 return result;
