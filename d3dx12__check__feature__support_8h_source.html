975// Helper function to decide the highest shader model supported by the system
976// Stores the result in m_dShaderModel
977// Must be updated whenever a new shader model is added to the d3d12.h header
978inline HRESULT CD3DX12FeatureSupport::QueryHighestShaderModel()
979{
980 // Check support in descending order
981 HRESULT result;
982
983 const D3D_SHADER_MODEL allModelVersions[] =
984 {
985#if defined(D3D12_SDK_VERSION) && (D3D12_SDK_VERSION >= 612)
986 D3D_SHADER_MODEL_6_9,
987#endif
988#if defined(D3D12_SDK_VERSION) && (D3D12_SDK_VERSION >= 606)
989 D3D_SHADER_MODEL_6_8,
990#endif
991#if defined(D3D12_SDK_VERSION) && (D3D12_SDK_VERSION >= 3)
992 D3D_SHADER_MODEL_6_7,
993#endif
994 D3D_SHADER_MODEL_6_6,
995 D3D_SHADER_MODEL_6_5,
996 D3D_SHADER_MODEL_6_4,
997 D3D_SHADER_MODEL_6_3,
998 D3D_SHADER_MODEL_6_2,
999 D3D_SHADER_MODEL_6_1,
1000 D3D_SHADER_MODEL_6_0,
1001 D3D_SHADER_MODEL_5_1
1002 };
1003 constexpr size_t numModelVersions = sizeof(allModelVersions) / sizeof(D3D_SHADER_MODEL);
1004
1005 for (size_t i = 0; i < numModelVersions; i++)
1006 {
1007 m_dShaderModel.HighestShaderModel = allModelVersions[i];
1008 result = m_pDevice->CheckFeatureSupport(D3D12_FEATURE_SHADER_MODEL, &m_dShaderModel, sizeof(D3D12_FEATURE_DATA_SHADER_MODEL));
1009 if (result != E_INVALIDARG)
1010 {
1011 // Indicates that the version is recognizable by the runtime and stored in the struct
1012 // Also terminate on unexpected error code
1013 if (FAILED(result))
1014 {
1015 m_dShaderModel.HighestShaderModel = static_cast<D3D_SHADER_MODEL>(0);
1016 }
1017 return result;
1018 }
1019 }
1020
1021 // Shader model may not be supported. Continue the rest initializations
1022 m_dShaderModel.HighestShaderModel = static_cast<D3D_SHADER_MODEL>(0);
1023 return S_OK;
1024}

1026// Helper function to decide the highest root signature supported
1027// Must be updated whenever a new root signature version is added to the d3d12.h header
1028inline HRESULT CD3DX12FeatureSupport::QueryHighestRootSignatureVersion()
1029{
1030 HRESULT result;
1031
1032 const D3D_ROOT_SIGNATURE_VERSION allRootSignatureVersions[] =
1033 {
1034#if defined(D3D12_SDK_VERSION) && (D3D12_SDK_VERSION >= 609)
1035 D3D_ROOT_SIGNATURE_VERSION_1_2,
1036#endif
1037 D3D_ROOT_SIGNATURE_VERSION_1_1,
1038 D3D_ROOT_SIGNATURE_VERSION_1_0,
1039 D3D_ROOT_SIGNATURE_VERSION_1,
1040 };
1041 constexpr size_t numRootSignatureVersions = sizeof(allRootSignatureVersions) / sizeof(D3D_ROOT_SIGNATURE_VERSION);
1042
1043 for (size_t i = 0; i < numRootSignatureVersions; i++)
1044 {
1045 m_dRootSignature.HighestVersion = allRootSignatureVersions[i];
1046 result = m_pDevice->CheckFeatureSupport(D3D12_FEATURE_ROOT_SIGNATURE, &m_dRootSignature, sizeof(D3D12_FEATURE_DATA_ROOT_SIGNATURE));
1047 if (result != E_INVALIDARG)
1048 {
1049 if (FAILED(result))
1050 {
