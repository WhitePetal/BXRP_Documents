256 int InsertionIndex(Vector2 pos)
257 {
258 VisualElement owner = contentContainer != null ? contentContainer : this;
259 Vector2 localPos = this.ChangeCoordinatesTo(owner, pos);
260
261 int index = BlackboardUtils.GetInsertionIndex(owner, localPos, Children());
262
263 // Clamps the index between the min and max of the child indices based on the mouse position relative to the categories on the y-axis (up/down)
264 // Checking for at least 2 children to make sure Children.First() and Children.Last() don't throw an exception
265 if (index == -1 && childCount >= 2)
266 {
267 index = localPos.y < Children().First().layout.yMin ? 0 :
268 localPos.y > Children().Last().layout.yMax ? childCount : -1;
269 }
270
271 // Don't allow the default category to be displaced
272 return Mathf.Clamp(index, 1, index);
273 }

276 {
277 var selection = DragAndDrop.GetGenericData("DragSelection") as List<ISelectable>;
278 if (selection == null)
279 {
280 SetCategoryDragIndicatorVisible(false);
281 return;
282 }
283
284 foreach (ISelectable selectedElement in selection)
285 {
286 var sourceItem = selectedElement as VisualElement;
287 // Don't allow user to move the default category

330 void OnDragPerformEvent(DragPerformEvent evt)
331 {
332 // Don't bubble up drop operations onto blackboard upto the graph view, as it leads to nodes being created without users knowledge behind the blackboard
333 evt.StopPropagation();
334
335 var selection = DragAndDrop.GetGenericData("DragSelection") as List<ISelectable>;
336 if (selection == null || selection.Count == 0)
337 {
338 SetCategoryDragIndicatorVisible(false);
339 return;
340 }
341
342 // Hide the category drag indicator if no categories in selection
343 if (!selection.OfType<GGBlackboardCategory>().Any())
344 {
345 SetCategoryDragIndicatorVisible(false);
346 }
347
348 Vector2 localPosition = evt.localMousePosition;
349 m_InsertIndex = InsertionIndex(localPosition);
350
351 // Any categories in the selection that are from other graphs, would have to be copied as opposed to moving the categories within the same graph
352 foreach (var item in selection.ToList())
353 {
354 if (item is GGBlackboardCategory category)
355 {
356 var selectedCategoryData = category.controller.Model;
357 bool doesCategoryExistInGraph = controller.Model.ContainsCategory(selectedCategoryData);
358 if (doesCategoryExistInGraph == false)
359 {
360 var copyCategoryAction = new CopyCategoryAction();
361 copyCategoryAction.categoryToCopyReference = selectedCategoryData;
362 ViewModel.requestModelChangeAction(copyCategoryAction);
363 selection.Remove(item);
364
365 // Remove any child inputs that belong to this category from the selection, to prevent duplicates from being copied onto the graph
366 foreach (var otherItem in selection.ToList())
367 {
368 if (otherItem is GGBlackboardField blackboardField && category.Contains(blackboardField))
369 selection.Remove(otherItem);
370 }
371 }
372 }
373 }
374
375 // Same as above, but for blackboard items (properties, keywords, dropdowns)
376 foreach (var item in selection.ToList())
377 {
378 if (item is GGBlackboardField blackboardField)
379 {
380 var selectedBlackboardItem = blackboardField.controller.Model;
381 bool doesInputExistInGraph = controller.Model.ContainsInput(selectedBlackboardItem);
382 if (doesInputExistInGraph == false)
383 {
384 var copyGeometryInputAction = new CopyGeometryInputAction();
385 copyGeometryInputAction.geometryInputToCopy = selectedBlackboardItem;
386 ViewModel.requestModelChangeAction(copyGeometryInputAction);
387 selection.Remove(item);
388 }
389 }
390 }
391
392 var moveCategoryAction = new MoveCategoryAction();
393 moveCategoryAction.newIndexValue = m_InsertIndex;
394 moveCategoryAction.categoryGuids = selection.OfType<GGBlackboardCategory>().OrderBy(sgcat => sgcat.GetPosition().y).Select(cat => cat.viewModel.associatedCategoryGuid).ToList();
395 ViewModel.requestModelChangeAction(moveCategoryAction);
396
397 SetCategoryDragIndicatorVisible(false);
398 }

449 void OnFieldDragUpdate(DragUpdatedEvent dragUpdatedEvent)
450 {
451 // how far is the mouse into the drag boundary.
452 float dragCoeff
453 = m_ScrollToTop ? 1 - dragUpdatedEvent.localMousePosition.y / m_ScrollBoundaryBottom.contentRect.height
454 : m_ScrollToBottom ? dragUpdatedEvent.localMousePosition.y / m_ScrollBoundaryBottom.contentRect.height
455 : 0;
456
457 dragCoeff = Mathf.Clamp(dragCoeff, .15f, .85f);
458
459 // factor in fixed base speed and relative to % of total scrollable height.
460 float dragSpeed = dragCoeff * k_DraggedPropertyScrollSpeed * (scrollableHeight / 100f);
461
462 // Lastly, make sure the drag speed can't ever get too slow.
463 dragSpeed = Mathf.Max(dragSpeed, k_DraggedPropertyScrollSpeed);
464
465 if (m_ScrollToTop)
466 m_ScrollView.scrollOffset = new Vector2(m_ScrollView.scrollOffset.x, Mathf.Clamp(m_ScrollView.scrollOffset.y - dragSpeed, 0, scrollableHeight));
467 else if (m_ScrollToBottom)
468 m_ScrollView.scrollOffset = new Vector2(m_ScrollView.scrollOffset.x, Mathf.Clamp(m_ScrollView.scrollOffset.y + dragSpeed, 0, scrollableHeight));
469 }
