116void RasterPipeline::OnRender(ComPtr<ID3D12GraphicsCommandList4> commandList, D3D12_CPU_DESCRIPTOR_HANDLE rtv, D3D12_CPU_DESCRIPTOR_HANDLE dsv)
117{
118 commandList->SetPipelineState(m_PipelineState.Get());
119 commandList->SetGraphicsRootSignature(m_RootSignature.Get());
120
121 // Clear the render targets
122 FLOAT clearColor[] = { 0.4f, 0.6f, 0.9f, 1.0f };
123 m_Engine->ClearRTV(commandList, rtv, clearColor);
124 ClearDepth(commandList, dsv);
125
126 commandList->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
127 commandList->IASetVertexBuffers(0, 1, m_Engine->GetVertexBufferView());
128 commandList->IASetIndexBuffer(m_Engine->GetIndexBufferView());
129
130 // Update the Camera Matrixs
131 ComPtr<ID3D12DescriptorHeap> camerConstHeap = m_Engine->GetCamera()->GetConstHeap();
132 std::vector<ID3D12DescriptorHeap*> heaps = { camerConstHeap.Get() };
133 commandList->SetDescriptorHeaps(static_cast<UINT>(heaps.size()), heaps.data());
134 commandList->SetGraphicsRootDescriptorTable(
135 0, camerConstHeap->GetGPUDescriptorHandleForHeapStart()
136 );
137
138 commandList->DrawIndexedInstanced(_countof(g_Indicies), 1, 0, 0, 0);
