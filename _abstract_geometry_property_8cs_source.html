158 // converts the old m_GPUInstanced data into the new override HLSLDeclaration system.
159 public static void UpgradeToHLSLDeclarationOverride(string json, AbstractGeometryProperty property)
160 {
161 // this maintains the old behavior for versioned properties:
162 // old exposed GPUInstanced properties are declared hybrid (becomes override in new system)
163 // old unexposed GPUInstanced properties are declared global (becomes override in new system)
164 // old exposed properties are declared UnityPerMaterial (default behavior, no override necessary)
165 // old unexposed properties are declared Global (default behavior, no override necessary)
166 // moving forward, users can use the overrides directly to control what it does
167
168 var legacyShaderPropertyData = new LegacyShaderPropertyData();
169 JsonUtility.FromJsonOverwrite(json, legacyShaderPropertyData);
170 if (legacyShaderPropertyData.m_GPUInstanced)
171 {
172 property.overrideHLSLDeclaration = true;
173 if (property.generatePropertyBlock)
174 property.hlslDeclarationOverride = HLSLDeclaration.HybridPerInstance;
175 else
176 property.hlslDeclarationOverride = HLSLDeclaration.Global;
177 }
178 }
