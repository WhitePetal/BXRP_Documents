279 void OnMouseHover(EventBase evt, GeometryInput input)
280 {
281 var graphView = ViewModel.parentView.GetFirstAncestorOfType<GeometryGraphView>();
282 if (evt.eventTypeId == MouseEnterEvent.TypeId())
283 {
284 foreach (var node in graphView.nodes.ToList())
285 {
286 if (input is AbstractGeometryProperty property)
287 {
288 if (node.userData is PropertyNode propertyNode)
289 {
290 if (propertyNode.property == input)
291 {
292 m_SelectedNodes.Add(node);
293 node.AddToClassList("hovered");
294 }
295 }
296 }
297 else if (input is ShaderKeyword keyword)
298 {
299 if (node.userData is KeywordNode keywordNode)
300 {
301 if (keywordNode.keyword == input)
302 {
303 m_SelectedNodes.Add(node);
304 node.AddToClassList("hovered");
305 }
306 }
307 }
308 else if (input is GeometryDropdown dropdown)
309 {
310 if (node.userData is DropdownNode dropdownNode)
311 {
312 if (dropdownNode.dropdown == input)
313 {
314 m_SelectedNodes.Add(node);
315 node.AddToClassList("hovered");
316 }
317 }
318 }
319 }
320 }
321 else if (evt.eventTypeId == MouseLeaveEvent.TypeId() && m_SelectedNodes.Any())
322 {
323 foreach (var node in m_SelectedNodes)
324 {
325 node.RemoveFromClassList("hovered");
326 }
327 m_SelectedNodes.Clear();
328 }
329 }
