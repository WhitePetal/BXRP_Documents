115 public unsafe void UpdateGPUData(CommandBuffer cmd, ref CullingResults cullingResults)
116 {
117 var probes = cullingResults.visibleReflectionProbes;
118 // Should probe come after otherProbe?
119 static bool IsProbeGreater(VisibleReflectionProbe probe, VisibleReflectionProbe otherProbe)
120 {
121 return probe.importance < otherProbe.importance ||
122 (probe.importance == otherProbe.importance && probe.bounds.extents.sqrMagnitude > otherProbe.bounds.extents.sqrMagnitude);
123 }
124
125 for (var i = 1; i < probes.Length; i++)
126 {
127 var probe = probes[i];
128 var j = i - 1;
129 while (j >= 0 && IsProbeGreater(probes[j], probe))
130 {
131 probes[j + 1] = probes[j];
132 j--;
133 }
134
135 probes[j + 1] = probe;
136 }
137 var probeCount = math.min(probes.Length, k_MaxVisibleReflectionProbeCount);
138 var frameIndex = Time.renderedFrameCount;
139
140 // Populate list of probes we need to remove to avoid modifying dictionary while iterating.
141 foreach(var (id, cachedProbe) in m_Cache)
142 {
143 // Evict probe if not used for more than 1 frame, if the texture no longer exists, or if the size changed.
144 if(Math.Abs(cachedProbe.lastUsed - frameIndex) > 1 || !cachedProbe.texture || cachedProbe.size != cachedProbe.texture.width)
145 {
146 m_NeedsRemove.Add(id);
147 for(var i = 0; i < k_MaxMipCount; ++i)
148 {
149 if (cachedProbe.dataIndices[i] != -1) m_AtlasAllocator.Free(new BuddyAllocation(cachedProbe.levels[i], cachedProbe.dataIndices[i]));
150 }
151 }
152 }
153
154 foreach(var probeIndex in m_NeedsRemove)
155 {
156 m_Cache.Remove(probeIndex);
157 }
158 m_NeedsRemove.Clear();
159
160 foreach(var (id, lastUsed) in m_WarningCache)
161 {
162 if(Math.Abs(lastUsed - frameIndex) > 1)
163 {
164 m_NeedsRemove.Add(id);
165 }
166 }
167
168 foreach(var probeIndex in m_NeedsRemove)
169 {
170 m_WarningCache.Remove(probeIndex);
171 }
172 m_NeedsRemove.Clear();
173
174 var showFullWarning = false;
175 var requiredAtlasSize = math.int2(0, 0);
176
177 for(var probeIndex = 0; probeIndex < probeCount; ++probeIndex)
178 {
179 var probe = probes[probeIndex];
180
181 var texture = probe.texture;
182 var id = probe.reflectionProbe.GetInstanceID();
183 var wasCached = m_Cache.TryGetValue(id, out var cacheProbe);
184
185 if (!texture) continue;
186
187 if (!wasCached)
188 {
189 cacheProbe.size = texture.width;
190 var mipCount = math.ceillog2(cacheProbe.size * 4) + 1; // * 4 means CubeMap width to 2D width
191 var level = m_AtlasAllocator.levelCount + 2 - mipCount;
192 cacheProbe.mipCount = math.min(mipCount, k_MaxMipCount);
193 cacheProbe.texture = texture;
194
195 var mip = 0;
196 for(; mip < cacheProbe.mipCount; ++mip)
197 {
198 // Clamp to maximum level. This is relevant for 64x64 and lower, which will have valid content
199 // in 1x1 mip. The octahedron size is double the face size, so that ends up at 2x2. Due to
200 // borders the final mip must be 4x4 as that leaves 2x2 texels for the octahedron.
201 var mipLevel = math.min(level + mip, m_AtlasAllocator.levelCount - 1);
202 if (!m_AtlasAllocator.TryAllocate(mipLevel, out var allocation)) break;
203 // We split up the allocation struct because C# cannot do struct fixed arrays :(
204 cacheProbe.levels[mip] = allocation.level;
205 cacheProbe.dataIndices[mip] = allocation.index;
206 var scaleOffset = (int4)(GetScaleOffset(mipLevel, allocation.index, true, false) * m_Resolution.xyxy);
207 requiredAtlasSize = math.max(requiredAtlasSize, scaleOffset.zw + scaleOffset.xy);
208 }
209
210 // Check if we ran out of space in the atlas.
211 if (mip < cacheProbe.mipCount)
212 {
213 if (!m_WarningCache.ContainsKey(id)) showFullWarning = true;
214 m_WarningCache[id] = frameIndex;
215 for (var i = 0; i < mip; ++i) m_AtlasAllocator.Free(new BuddyAllocation(cacheProbe.levels[i], cacheProbe.dataIndices[i]));
216 for (var i = 0; i < k_MaxMipCount; ++i) cacheProbe.dataIndices[i] = -1;
217 continue;
218 }
219
220 for (; mip < k_MaxMipCount; ++mip)
221 {
222 cacheProbe.dataIndices[mip] = -1;
223 }
224 }
225
226 var needsUpdate = !wasCached || cacheProbe.updateCount != texture.updateCount;
227#if UNITY_EDITOR
228 needsUpdate |= cacheProbe.imageContentsHash != texture.imageContentsHash;
229#endif
230 needsUpdate |= cacheProbe.hdrData != probe.hdrData; // The probe needs update if the runtime intensity multiplier changes
231 //needsUpdate = true;
232
233 if (needsUpdate)
234 {
235 cacheProbe.updateCount = texture.updateCount;
236#if UNITY_EDITOR
237 cacheProbe.imageContentsHash = texture.imageContentsHash;
238#endif
239 m_NeedsUpdate.Add(id);
240 }
241
242 // If the probe is set to be updated every frame, we assign the last used frame to -1 so it's evicted in next frame.
243 if (probe.reflectionProbe.mode == ReflectionProbeMode.Realtime && probe.reflectionProbe.refreshMode == ReflectionProbeRefreshMode.EveryFrame)
244 cacheProbe.lastUsed = -1;
245 else
246 cacheProbe.lastUsed = frameIndex;
247
248 cacheProbe.hdrData = probe.hdrData;
249 m_Cache[id] = cacheProbe;
250 }
251
252 // Grow the atlas if it's not big enough to contain the current allocations.
253 if(math.any(m_Resolution < requiredAtlasSize))
254 {
255 requiredAtlasSize = math.max(m_Resolution, math.ceilpow2(requiredAtlasSize));
256 var desc = m_AtlasTexture0.descriptor;
257 desc.width = requiredAtlasSize.x;
258 desc.height = requiredAtlasSize.y;
259 m_AtlasTexture1.width = requiredAtlasSize.x;
260 m_AtlasTexture1.height = requiredAtlasSize.y;
261 m_AtlasTexture1.Create();
262
263 if(m_AtlasTexture0.width != 1)
264 {
265 if(SystemInfo.copyTextureSupport != CopyTextureSupport.None)
266 {
267 Graphics.CopyTexture(m_AtlasTexture0, 0, 0, 0, 0, m_Resolution.x, m_Resolution.y, m_AtlasTexture1, 0, 0, 0, 0);
268 }
269 else
270 {
271 Graphics.Blit(m_AtlasTexture0, m_AtlasTexture1, (float2)m_Resolution / requiredAtlasSize, Vector2.zero);
272 }
273 }
274
275 m_AtlasTexture0.Release();
276 (m_AtlasTexture0, m_AtlasTexture1) = (m_AtlasTexture1, m_AtlasTexture0);
277 m_Resolution = requiredAtlasSize;
278 }
279
280 var skipCount = 0;
281 for(var probeIndex = 0; probeIndex < probeCount; ++probeIndex)
282 {
283 var probe = probes[probeIndex];
284 var id = probe.reflectionProbe.GetInstanceID();
285 var dataIndex = probeIndex - skipCount;
286 if(!m_Cache.TryGetValue(id, out var cacheProbe) || !probe.texture)
287 {
288 skipCount++;
289 continue;
290 }
291 m_Probes[dataIndex] = probe;
292 m_BoxMax[dataIndex] = new Vector4(probe.bounds.max.x, probe.bounds.max.y, probe.bounds.max.z, probe.blendDistance);
293 m_BoxMin[dataIndex] = new Vector4(probe.bounds.min.x, probe.bounds.min.y, probe.bounds.min.z, probe.importance);
294 m_ProbePostion[dataIndex] = new Vector4(probe.localToWorldMatrix.m03, probe.localToWorldMatrix.m13, probe.localToWorldMatrix.m23, (probe.isBoxProjection ? 1 : -1) * (cacheProbe.mipCount));
295 for (var i = 0; i < cacheProbe.mipCount; ++i) m_MipScaleOffset[dataIndex * k_MaxMipCount + i] = GetScaleOffset(cacheProbe.levels[i], cacheProbe.dataIndices[i], false, false);
296 }
297
298 if (showFullWarning)
299 {
300 Debug.LogWarning("A numbter of reflection probes have been skipped due to the reflection probe atlas being full.\nTo ix this, you can decrease the number or resolution of probes.");
301 }
302
303 using(new ProfilingScope(cmd, ProfilingSampler.Get(BXProfileId.UpdateReflectionProbeAtlas)))
304 {
305 if (m_NeedsUpdate.Count > 0)
306 {
307 cmd.SetRenderTarget(m_AtlasTexture0);
308 foreach (var probeId in m_NeedsUpdate)
309 {
310 var cacheProbe = m_Cache[probeId];
311 for (var mip = 0; mip < cacheProbe.mipCount; ++mip)
312 {
313 var level = cacheProbe.levels[mip];
314 var dataIndex = cacheProbe.dataIndices[mip];
315 // If we need to y-flip we will instead flip the atlas since that is updated less frequent and then the lookup should be correct.
316 // By doing this we won't have to y-flip the lookup in the shader code.
317 var scaleBias = GetScaleOffset(level, dataIndex, true, !SystemInfo.graphicsUVStartsAtTop);
318 var sizeWaithoutPadding = (1 << (m_AtlasAllocator.levelCount + 1 - level)) - 2;
319 Blitter.BlitCubeToOctahedral2DQuadWithPadding(cmd, cacheProbe.texture, new Vector2(sizeWaithoutPadding, sizeWaithoutPadding), scaleBias, mip, true, 2, cacheProbe.hdrData);
320 }
321 }
322 }
323 cmd.SetGlobalVectorArray(ShaderProperties.bx_ReflProbes_BoxMin_ID, m_BoxMin);
324 cmd.SetGlobalVectorArray(ShaderProperties.bx_ReflProbes_BoxMax_ID, m_BoxMax);
325 cmd.SetGlobalVectorArray(ShaderProperties.bx_ReflProbes_ProbePosition_ID, m_ProbePostion);
326 cmd.SetGlobalVectorArray(ShaderProperties.bx_ReflProbes_MipScaleOffset_ID, m_MipScaleOffset);
327 this.probeCount = probeCount - skipCount;
328 cmd.SetGlobalInt(ShaderProperties.bx_ReflProbes_Count_ID, this.probeCount);
329 cmd.SetGlobalTexture(ShaderProperties.bx_ReflProbes_Atlas_ID, m_AtlasTexture0);
330 cmd.SetGlobalTexture("_GlossyEnvironmentCubeMap", ReflectionProbe.defaultTexture);
331 cmd.SetGlobalVector("_GlossyEnvironmentCubeMap_HDR", ReflectionProbe.defaultTextureHDRDecodeValues);
332 }
333
334 m_NeedsUpdate.Clear();
335 }

353 private float4 GetScaleOffset(int level, int dataIndex, bool includePadding, bool yflip)
354 {
355 // level = m_AtlasAllocator.levelCount + 2 - (log2(size) + 1) <=>
356 // log2(size) + 1 = m_AtlasAllocator.levelCount + 2 - level <=>
357 // log2(size) = m_AtlasAllocator.levelCount + 1 - level <=>
358 // size = 2^(m_AtlasAllocator.levelCount + 1 - level)
359 var size = (1 << (m_AtlasAllocator.levelCount + 1 - level));
360 var coordinate = BXSpaceFillingCurves.DecodeMorton2D((uint)dataIndex);
361 var scale = (size - (includePadding ? 0 : 2)) / ((float2)m_Resolution);
362 var bias = ((float2) coordinate * size + (includePadding ? 0 : 1)) / (m_Resolution);
363 if (yflip) bias.y = 1f - bias.y - scale.y;
364 return math.float4(scale, bias);
365 }
