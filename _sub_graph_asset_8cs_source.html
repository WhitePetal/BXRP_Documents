214 internal void LoadDependencyData()
215 {
216 m_InputDependencies.Clear();
217 m_OutputDependencies.Clear();
218
219 foreach(var capabilityInfo in inputCapabilities)
220 {
221 var dependencyInfo = new SlotDependencyInfo();
222 dependencyInfo.slotName = capabilityInfo.slotName;
223 dependencyInfo.capabilities = capabilityInfo.capabilities;
224 if (m_InputDependencies.ContainsKey(dependencyInfo.slotName))
225 {
226 Debug.LogWarning($"SubGraph '{hlslName}' has multiple input slots named '{dependencyInfo.slotName}', which is unsupported. Please assign the input slots unique names.");
227 continue;
228 }
229 m_InputDependencies.Add(dependencyInfo.slotName, dependencyInfo);
230 }
231 foreach(var capabilityInfo in outputCapabilities)
232 {
233 var dependencyInfo = new SlotDependencyInfo();
234 dependencyInfo.slotName = capabilityInfo.slotName;
235 dependencyInfo.capabilities = capabilityInfo.capabilities;
236 if (m_OutputDependencies.ContainsKey(dependencyInfo.slotName))
237 {
238 Debug.LogWarning($"SubGraph '{hlslName}' has multiple output slots named '{dependencyInfo.slotName}', which is unsupported. Please assign the output slots unique names.");
239 continue;
240 }
241 m_OutputDependencies.Add(dependencyInfo.slotName, dependencyInfo);
242 }
243 foreach(var slotDependency in slotDependencies)
244 {
245 // This shouldn't fail since every input/output must be in the above lists...
246 if (m_InputDependencies.ContainsKey(slotDependency.inputSlotName))
247 m_InputDependencies[slotDependency.inputSlotName].AddDepencencySlotName(slotDependency.outputSlotName);
248 if (m_OutputDependencies.ContainsKey(slotDependency.outputSlotName))
249 m_OutputDependencies[slotDependency.outputSlotName].AddDepencencySlotName(slotDependency.inputSlotName);
250 }
251 }
