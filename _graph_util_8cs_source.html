135 internal static bool CheckForRecursiveDependencyOnPendingSave(string saveFilePath, IEnumerable<SubGraphNode> subGraphNodes, string context = null)
136 {
137 var overwriteGUID = AssetDatabase.AssetPathToGUID(saveFilePath);
138 if (!string.IsNullOrEmpty(overwriteGUID))
139 {
140 foreach (var sgNode in subGraphNodes)
141 {
142 var asset = sgNode?.asset;
143 if (asset == null)
144 {
145 // cannot read the asset; might be recursive but we can't tell... should we return "maybe"?
146 // I think to be minimally intrusive to the user we can assume "No" in this case,
147 // even though this may miss recursions in extraordinary cases.
148 // it's more important to allow the user to save their files than to catch 100% of recursions
149 continue;
150 }
151 else if ((asset.assetGuid == overwriteGUID) || asset.descendents.Contains(overwriteGUID))
152 {
153 if (context != null)
154 {
155 Debug.LogWarning(context + " CANCELLED to avoid a generating a reference loop: the SubGraph '" + sgNode.asset.name + "' references the target file '" + saveFilePath + "'");
156 EditorUtility.DisplayDialog(
157 context + " CANCELLED",
158 "Saving the file would generate a reference loop, because the SubGraph '" + sgNode.asset.name + "' references the target file '" + saveFilePath + "'", "Cancel");
159 }
160 return true;
161 }
162 }
163 }
164 return false;
165 }

167 internal static string ConvertCamelCase(string text, bool preserveAcronyms)
168 {
169 if (string.IsNullOrEmpty(text))
170 return string.Empty;
171 StringBuilder newText = new StringBuilder(text.Length * 2);
172 newText.Append(text[0]);
173 for (int i = 1; i < text.Length; i++)
174 {
175 if (char.IsUpper(text[i]))
176 if ((text[i - 1] != ' ' && !char.IsUpper(text[i - 1])) ||
177 (preserveAcronyms && char.IsUpper(text[i - 1]) &&
178 i < text.Length - 1 && !char.IsUpper(text[i + 1])))
179 newText.Append(' ');
180 newText.Append(text[i]);
181 }
182 return newText.ToString();
183 }

239 static void Visit(List<AbstractGeometryNode> outputList, Dictionary<string, AbstractGeometryNode> unmarkedNodes, AbstractGeometryNode node)
240 {
241 if (!unmarkedNodes.ContainsKey(node.objectId))
242 return;
243 foreach (var slot in node.GetInputSlots<GeometrySlot>())
244 {
245 foreach (var edge in node.owner.GetEdges(slot.slotReference))
246 {
247 var inputNode = edge.outputSlot.node;
248 Visit(outputList, unmarkedNodes, inputNode);
249 }
250 }
251 unmarkedNodes.Remove(node.objectId);
252 outputList.Add(node);
253 }

311 internal static string DeduplicateName(IEnumerable<string> existingNames, string duplicateFormat, string name)
312 {
313 if (!existingNames.Contains(name))
314 return name;
315
316 string escapedDuplicateFormat = Regex.Escape(duplicateFormat);
317
318 // Escaped format will escape string interpolation, so the escape characters must be removed for these.
319 escapedDuplicateFormat = escapedDuplicateFormat.Replace(@"\{0}", @"{0}");
320 escapedDuplicateFormat = escapedDuplicateFormat.Replace(@"\{1}", @"{1}");
321
322 var baseRegex = new Regex(string.Format(escapedDuplicateFormat, @"^(.*)", @"(\d+)"));
323
324 var baseMatch = baseRegex.Match(name);
325 if (baseMatch.Success)
326 name = baseMatch.Groups[1].Value;
327
328 string baseNameExpression = string.Format(@"^{0}", Regex.Escape(name));
329 var regex = new Regex(string.Format(escapedDuplicateFormat, baseNameExpression, @"(\d+)") + "$");
330
331 var existingDuplicateNumbers = existingNames.Select(existingName => regex.Match(existingName)).Where(m => m.Success).Select(m => int.Parse(m.Groups[1].Value)).Where(n => n > 0).Distinct().ToList();
332
333 var duplicateNumber = 1;
334 existingDuplicateNumbers.Sort();
335 if (existingDuplicateNumbers.Any() && existingDuplicateNumbers.First() == 1)
336 {
337 duplicateNumber = existingDuplicateNumbers.Last() + 1;
338 for (var i = 1; i < existingDuplicateNumbers.Count; i++)
339 {
340 if (existingDuplicateNumbers[i - 1] != existingDuplicateNumbers[i] - 1)
341 {
342 duplicateNumber = existingDuplicateNumbers[i - 1] + 1;
343 break;
344 }
345 }
346 }
347
348 return string.Format(duplicateFormat, name, duplicateNumber);
349 }

393 //
394 // Find all nodes of the given type downstream from the given node
395 // Returns a unique list. So even if a node can be reached through different paths it will be present only once.
396 //
397 public static List<NodeType> FindDownStreamNodesOfType<NodeType>(AbstractGeometryNode node) where NodeType : AbstractGeometryNode
398 {
399 // Should never be called without a node
400 Debug.Assert(node != null);
401
402 HashSet<AbstractGeometryNode> visitedNodes = new HashSet<AbstractGeometryNode>();
403 List<NodeType> vtNodes = new List<NodeType>();
404 Queue<AbstractGeometryNode> nodeStack = new Queue<AbstractGeometryNode>();
405 nodeStack.Enqueue(node);
406 visitedNodes.Add(node);
407
408 while (nodeStack.Count > 0)
409 {
410 AbstractGeometryNode visit = nodeStack.Dequeue();
411
412 // Flood fill through all the nodes
413 foreach (var slot in visit.GetInputSlots<GeometrySlot>())
414 {
415 foreach (var edge in visit.owner.GetEdges(slot.slotReference))
416 {
417 var inputNode = edge.outputSlot.node;
418 if (!visitedNodes.Contains(inputNode))
419 {
420 nodeStack.Enqueue(inputNode);
421 visitedNodes.Add(inputNode);
422 }
423 }
424 }
425
426 // Extract vt node
427 if (visit is NodeType)
428 {
429 NodeType vtNode = visit as NodeType;
430 vtNodes.Add(vtNode);
431 }
432 }
433
434 return vtNodes;
435 }
