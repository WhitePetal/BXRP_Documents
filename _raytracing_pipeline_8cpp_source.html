172AccelerationStructureBuffers RaytracingPipeline::CreateBottomLevelAS(ComPtr<ID3D12Device5> device, ComPtr<ID3D12GraphicsCommandList4> commandList, std::vector <std::pair<ComPtr<ID3D12Resource>, uint32_t>> vVertexBuffers,
173 std::vector<std::pair<ComPtr<ID3D12Resource>, uint32_t>> vIndexBuffers)
174{
175 nv_helpers_dx12::BottomLevelASGenerator bottomLevelAS;
176
177 // Adding all vertex buffers and not transforming their positon
178 for (size_t i = 0; i < vIndexBuffers.size(); ++i)
179 {
180 //bottomLevelAS.AddVertexBuffer(buffer.first.Get(), 0, buffer.second, sizeof(VertexPosColor), m_IndexBuffer.Get(), 0, _countof(g_Indicies), nullptr, 0, true);
181 bottomLevelAS.AddVertexBuffer(vVertexBuffers[i].first.Get(), 0, vVertexBuffers[i].second, sizeof(VertexPosColor),
182 vIndexBuffers[i].first.Get(), 0, vIndexBuffers[i].second, nullptr, true);
183 }
184
185 // The AS build requires some scratch space to store temporary information.
186 // The amount of scratch memory is dependent on the scene complexity.
187 UINT64 scratchSizeInBytes = 0;
188 // The final AS also needs to be stored in addition to the existing vertex buffers.
189 // It size is alos dependent on the scene complexity.
190 UINT64 resultSizeInBytes = 0;
191
192 bottomLevelAS.ComputeASBufferSizes(device.Get(), false, &scratchSizeInBytes, &resultSizeInBytes);
193
194 // Once the size are obtained, the application is responsible for allocating the necessary buffers.
195 // Since the entire generation will be done on the GPU,
196 // we can directly allocate those on the default heap
197 AccelerationStructureBuffers buffers;
198 buffers.pScratch = nv_helpers_dx12::CreateBuffer(
199 device.Get(), scratchSizeInBytes,
200 D3D12_RESOURCE_FLAG_ALLOW_UNORDERED_ACCESS,
201 D3D12_RESOURCE_STATE_COMMON,
202 nv_helpers_dx12::kDefaultHeapProps
203 );
204 buffers.pResult = nv_helpers_dx12::CreateBuffer(
205 device.Get(), resultSizeInBytes,
206 D3D12_RESOURCE_FLAG_ALLOW_UNORDERED_ACCESS,
207 D3D12_RESOURCE_STATE_RAYTRACING_ACCELERATION_STRUCTURE,
208 nv_helpers_dx12::kDefaultHeapProps
209 );
210
211 // Build the aceeleration structure. Note that this call integrates a barrier on the generated AS,
212 // so that it can be used to compute a top-level AS right after this method.
213 bottomLevelAS.Generate(commandList.Get(), buffers.pScratch.Get(), buffers.pResult.Get(), false, nullptr);
214
215 return buffers;
216}

218void RaytracingPipeline::CreateTopLevelAS(ComPtr<ID3D12Device5> device, ComPtr<ID3D12GraphicsCommandList4> commandList, const std::vector<std::pair<ComPtr<ID3D12Resource>, XMMATRIX>>& instances)
219{
220 // Gather all the instances into the builder helper
221 for (size_t i = 0; i < instances.size(); ++i)
222 {
223 m_TopLevelASGenerator.AddInstance(
224 instances[i].first.Get(),
225 instances[i].second,
226 static_cast<UINT>(i), static_cast<UINT>(0));
227 }
228
229 // As for the bottom-levels AS, the building the AS requires some scratch space
230 // to store stemporary data in addition to the actual AS.
231 // In the case of the top-level AS, the instance descriptors also need to be stored in GPU memory.
232 // This call ouputs the memory requirements for each (scratch, results, instance descriptors)
233 // so that the application can allocate the corresponding memory
234 UINT64 scratchSize, resultSize, instanceDescsSize;
235
236 m_TopLevelASGenerator.ComputeASBufferSizes(device.Get(), true, &scratchSize, &resultSize, &instanceDescsSize);
237
238 // Create the scratch and result buffers. Since the build is all done on GPU,
239 // those can be allocated on the default heap
240 m_TopLevelASBuffers.pScratch = nv_helpers_dx12::CreateBuffer(
241 device.Get(), scratchSize,
242 D3D12_RESOURCE_FLAG_ALLOW_UNORDERED_ACCESS,
243 D3D12_RESOURCE_STATE_UNORDERED_ACCESS,
244 nv_helpers_dx12::kDefaultHeapProps
245 );
246 m_TopLevelASBuffers.pResult = nv_helpers_dx12::CreateBuffer(
247 device.Get(), resultSize,
248 D3D12_RESOURCE_FLAG_ALLOW_UNORDERED_ACCESS,
249 D3D12_RESOURCE_STATE_RAYTRACING_ACCELERATION_STRUCTURE,
250 nv_helpers_dx12::kDefaultHeapProps
251 );
252
253 // The buffer describing the instances: ID, shader binding information, matrices ...
254 // Those will be copied into the buffer by the helper through mapping,
255 // so the buffer has to be allocated on the upload heap.
256 m_TopLevelASBuffers.pInstanceDesc = nv_helpers_dx12::CreateBuffer(
257 device.Get(), instanceDescsSize,
258 D3D12_RESOURCE_FLAG_NONE,
259 D3D12_RESOURCE_STATE_GENERIC_READ,
260 nv_helpers_dx12::kUploadHeapProps
261 );
262
263 // After all the buffers are allocated, or if only an update is required,
264 // we can build the acceleration structure. Note that in the case of the update
265 // we also pass the existing AS as the 'previous' AS, so that it can be refitted in place.
266 m_TopLevelASGenerator.Generate(
267 commandList.Get(),
268 m_TopLevelASBuffers.pScratch.Get(),
269 m_TopLevelASBuffers.pResult.Get(),
270 m_TopLevelASBuffers.pInstanceDesc.Get()
271 );
272}

274void RaytracingPipeline::CreateAccelerationStructures()
275{
276 auto device = Application::Get().GetDevice();
277 auto commandQueue = Application::Get().GetCommandQueue(D3D12_COMMAND_LIST_TYPE_DIRECT);
278 auto commandList = commandQueue->GetCommandList();
279
280 // Build the bottom AS
281 m_BottomLevelASBuffers = CreateBottomLevelAS(device, commandList,
282 { {m_Engine->GetVertexBuffer().Get(), _countof(g_Vertices)} },
283 { {m_Engine->GetIndexBuffer().Get(), _countof(g_Indicies)} });
284
285 // Just one instance for now
286 m_Instances = { {m_BottomLevelASBuffers.pResult, XMMatrixIdentity()} };
287 CreateTopLevelAS(device, commandList, m_Instances);
288
289 // Flush the command list and wait for it to finish
290 auto fenceValue = commandQueue->ExecuteCommandList(commandList);
291 commandQueue->WaitForFenceValue(fenceValue);
292}

325void RaytracingPipeline::CreateRaytracingPipline()
326{
327 auto device = Application::Get().GetDevice();
328 nv_helpers_dx12::RayTracingPipelineGenerator pipeline(device.Get());
329
330 m_RayGenLibrary = nv_helpers_dx12::CompileShaderLibrary(L"RayTracingShaders/RayGen.hlsl");
331 m_MissLibrary = nv_helpers_dx12::CompileShaderLibrary(L"RayTracingShaders/Miss.hlsl");
332 m_HitLibrary = nv_helpers_dx12::CompileShaderLibrary(L"RayTracingShaders/Hit.hlsl");
333
334 pipeline.AddLibrary(m_RayGenLibrary.Get(), { L"RayGen" });
335 pipeline.AddLibrary(m_MissLibrary.Get(), { L"Miss" });
336 pipeline.AddLibrary(m_HitLibrary.Get(), { L"ClosestHit" });
337
338 m_RayGenSignature = CreateRayGenSignature();
339 m_MissSignature = CreateMissSignature();
340 m_HitSignature = CreateHitSignature();
341
342 pipeline.AddHitGroup(L"HitGroup", L"ClosestHit");
343
344 pipeline.AddRootSignatureAssociation(m_RayGenSignature.Get(), { L"RayGen" });
345 pipeline.AddRootSignatureAssociation(m_MissSignature.Get(), { L"Miss" });
346 pipeline.AddRootSignatureAssociation(m_HitSignature.Get(), { L"HitGroup" });
347
348 pipeline.SetMaxPayloadSize(4 * sizeof(float));
349
350 pipeline.SetMaxAttributeSize(2 * sizeof(float));
351
352 pipeline.SetMaxRecursionDepth(1);
353
354 m_RTStateObject = pipeline.Generate();
355
356 ThrowIfFaild(m_RTStateObject->QueryInterface(IID_PPV_ARGS(&m_RTStateObjectProps)));
357}
