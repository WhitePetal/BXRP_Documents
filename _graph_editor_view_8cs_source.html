484 GraphViewChange GraphViewChanged(GraphViewChange graphViewChange)
485 {
486 if (graphViewChange.edgesToCreate != null)
487 {
488 foreach (var edge in graphViewChange.edgesToCreate)
489 {
490 var leftSlot = edge.output.GetSlot();
491 var rightSlot = edge.input.GetSlot();
492 if (leftSlot != null && rightSlot != null)
493 {
494 m_Graph.owner.RegisterCompleteObjectUndo("Connect Edge");
495 m_Graph.Connect(leftSlot.slotReference, rightSlot.slotReference);
496 }
497 }
498 graphViewChange.edgesToCreate.Clear();
499 }
500
501 if (graphViewChange.movedElements != null)
502 {
503 m_Graph.owner.RegisterCompleteObjectUndo("Move Elements");
504
505 List<GraphElement> nodesInsideGroup = new List<GraphElement>();
506 foreach (var element in graphViewChange.movedElements)
507 {
508 var groupNode = element as GeometryGroup;
509 if (groupNode == null)
510 continue;
511
512 foreach (GraphElement graphElement in groupNode.containedElements)
513 {
514 nodesInsideGroup.Add(graphElement);
515 }
516
517 SetGroupPosition(groupNode);
518 }
519
520 if (nodesInsideGroup.Any())
521 graphViewChange.movedElements.AddRange(nodesInsideGroup);
522
523 foreach (var element in graphViewChange.movedElements)
524 {
525 if (element.userData is AbstractGeometryNode node)
526 {
527 var drawState = node.drawState;
528 drawState.position = element.parent.ChangeCoordinatesTo(m_GraphView.contentViewContainer, element.GetPosition());
529 node.drawState = drawState;
530
531 // BlockNode moved outside a Context
532 // This isnt allowed but there is no way to disallow it on the GraphView
533 if (node is BlockNode blockNode &&
534 element.GetFirstAncestorOfType<ContextView>() == null)
535 {
536 var context = graphView.GetContext(blockNode.contextData);
537
538 // isDragging ensures we arent calling this when moving
539 // the BlockNode into the GraphView during dragging
540 if (context.isDragging)
541 continue;
542
543 // Remove from GraphView and add back to Context
544 m_GraphView.RemoveElement(element);
545 context.InsertBlock(element as GeometryNodeView);
546 }
547 }
548
549
550 if (element is StickyNote stickyNote)
551 {
552 SetStickyNotePosition(stickyNote);
553 }
554
555 if (element is ContextView contextView)
556 {
557 var rect = element.parent.ChangeCoordinatesTo(m_GraphView.contentViewContainer, element.GetPosition());
558 contextView.contextData.position = rect.position;
559 }
560 }
561 }
562
563 var nodesToUpdate = m_NodeViewHashSet;
564 nodesToUpdate.Clear();
565
566 if(graphViewChange.elementsToRemove != null)
567 {
568 m_Graph.owner.RegisterCompleteObjectUndo("Remove Elements");
569 m_Graph.RemoveElements(
570 graphViewChange.elementsToRemove.OfType<IGeometryNodeView>().Select(v => v.node).ToArray(),
571 graphViewChange.elementsToRemove.OfType<UnityEditor.Experimental.GraphView.Edge>().Select(e => (IEdge)e.userData).ToArray(),
572 graphViewChange.elementsToRemove.OfType<GeometryGroup>().Select(g => g.userData).ToArray(),
573 graphViewChange.elementsToRemove.OfType<StickyNote>().Select(n => n.userData).ToArray(),
574 graphViewChange.elementsToRemove.OfType<GGBlackboardField>().Select(f => (GeometryInput)f.userData).ToArray()
575 );
576 foreach (var edge in graphViewChange.elementsToRemove.OfType<UnityEditor.Experimental.GraphView.Edge>())
577 {
578 if (edge.input != null)
579 {
580 if (edge.input.node is IGeometryNodeView geometryNodeView)
581 nodesToUpdate.Add(geometryNodeView);
582 }
583 if (edge.output != null)
584 {
585 if (edge.output.node is IGeometryNodeView geometryNodeView)
586 nodesToUpdate.Add(geometryNodeView);
587 }
588 }
589 }
590
591 foreach (var node in nodesToUpdate)
592 {
593 node.OnModified(ModificationScope.Topological);
594 }
595
596 UpdateEdgeColors(nodesToUpdate);
597 return graphViewChange;
598 }
