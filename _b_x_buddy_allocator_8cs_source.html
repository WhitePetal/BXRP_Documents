65 Assert.IsTrue(branchingOrder is >= 1 and <= 3);

94 public bool TryAllocate(int requestedLevel, out BuddyAllocation allocation)
95 {
96 allocation = default;
97
98 // Find the highest level that has a block available.
99 var level = requestedLevel;
100 var maskCounts = freeMaskCounts;
101 while(level >= 0)
102 {
103 if (maskCounts[level] > 0) break;
104 level--;
105 }
106
107 // No blocks available.
108 if (level < 0) return false;
109
110 // Split a block at the level we found.
111 int dataIndex;
112 {
113 var freeMaskIndices = FreeMaskIndices(level);
114 var maskIndex = freeMaskIndices[--maskCounts[level]];
115 var freeMasks = FreeMasks(level);
116 var freeMask = freeMasks[maskIndex];
117 Assert.AreNotEqual(freeMask, 0ul);
118 var bitIndex = math.tzcnt(freeMask);
119 freeMask ^= 1ul << bitIndex;
120 freeMasks[maskIndex] = freeMask;
121 if (freeMask != 0) freeMaskIndices[maskCounts[level]++] = maskIndex;
122 dataIndex = maskIndex * 64 + bitIndex;
123 }
124
125 // Walk up the levels until we hit the requested level. For each level we want to mark the remaining parts
126 // of the newly split blocks as free.
127 while(level < requestedLevel)
128 {
129 level++;
130 dataIndex <<= header.branchingOrder;
131 var maskIndex = dataIndex >> 6;
132 var bitIndex = dataIndex & 63;
133 var freeMasks = FreeMasks(level);
134 var freeMask = freeMasks[maskIndex];
135 // We might have hit a mask that already contained free blocks. If not, add the mask index to the free list.
136 if(freeMask == 0)
137 {
138 var freeMaskIndices = FreeMaskIndices(level);
139 freeMaskIndices[maskCounts[level]++] = maskIndex;
140 }
141
142 // Mark other bits in the block we just broke apart as free.
143 // In binary form, 2^b will give us a 1 followed by b 0s. So to get b ones, we subtract 1. Since we want
144 // the least significant bit to be 0, we subtract another 1.
145 // E.g. for branching order 1 we get 10b, for 2 we get 1110b, for 3 we get 11111110b.
146 // Finally we shift according to the data index.
147 Assert.IsTrue(bitIndex + Pow2(header.branchingOrder) - 1 < 64);
148 freeMask |= ((1ul << Pow2(header.branchingOrder)) - 2ul) << bitIndex;
149 freeMasks[maskIndex] = freeMask;
150 }
151
152 allocation.level = level;
153 allocation.index = dataIndex;
154 return true;
155 }

157 public void Free(BuddyAllocation allocation)
158 {
159 var level = allocation.level;
160 var dataIndex = allocation.index;
161 while(level >= 0)
162 {
163 var maskIndex = dataIndex >> 6;
164 var bitIndex = dataIndex & 63;
165 var freeMasks = FreeMasks(level);
166 var freeMask = freeMasks[maskIndex];
167 var wasZero = freeMask == 0;
168 freeMask |= 1ul << bitIndex;
169
170 var indices = FreeMaskIndices(level);
171 var counts = freeMaskCounts;
172
173 var supoerBlockMask = ((1ul << Pow2(header.branchingOrder)) - 1) << ((bitIndex >> header.branchingOrder) * Pow2(header.branchingOrder));
174 // Check if the whole super-block (i.e. making up one block of the next level) is free.
175 // If it is, we let the loop continue upwards.
176 if(level == 0 || (~freeMask & supoerBlockMask) != 0)
177 {
178 freeMasks[maskIndex] = freeMask;
179 if (wasZero)
180 {
181 indices[counts[level]++] = maskIndex;
182 }
183 break;
184 }
185
186 freeMask &= ~supoerBlockMask;
187 freeMasks[maskIndex] = freeMask;
188
189 if(!wasZero && freeMask == 0)
190 {
191 for(var i = 0; i < indices.Length; ++i)
192 {
193 if(indices[i] == maskIndex)
194 {
195 indices[i] = indices[--counts[level]];
196 break;
197 }
198 }
199 }
200
201 level--;
202 dataIndex >>= header.branchingOrder;
203 }
204 }

227 static int Pow2(int n) => 1 << n;
228 static int Pow2N(int x, int n) => 1 << (x * n);

230 // sum x^i for i=0..(n-1) = (x ^ n - 1) / (x - 1) where n is order and n is 2^branchingOrder
231 static int LevelOffset(int level, int branchingOrder) => (Pow2(branchingOrder) * (Pow2(branchingOrder * (level - 1) + branchingOrder) - 1)) / (Pow2(branchingOrder) - 1);
232 static int LevelLength(int level, int branchingOrder) => Pow2N(branchingOrder, level + 1);
233
234 // These are for when orders of length <= 64 only take up 1 item, e.g. ulong bitmasks.
235 static int LevelOffset64(int level, int branchingOrder) => math.min(level, 6 / branchingOrder) + LevelOffset(math.max(0, level - 6 / branchingOrder), branchingOrder);
236 static int LevelLength64(int level, int branchingOrder) => Pow2N(branchingOrder, math.max(0, level - 6 / branchingOrder + 1));

238 static int AlignForward(int offset, int alignment)
239 {
240 var modulo = offset % alignment;
