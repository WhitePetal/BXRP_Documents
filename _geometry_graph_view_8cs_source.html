181 public override List<Port> GetCompatiblePorts(Port startAnchor, NodeAdapter nodeAdapter)
182 {
183 var compatibleAnchors = new List<Port>();
184 var startSlot = startAnchor.GetSlot();
185 if (startSlot == null)
186 return compatibleAnchors;
187
188 var startStage = startSlot.stageCapability;
189 // If this is a sub-graph node we always have to check the effective stage as we might have to trace back through the sub-graph
190 if (startStage == GeometryStageCapability.All || startSlot.owner is SubGraphNode)
191 startStage = NodeUtils.GetEffectiveGeometryStageCapability(startSlot, true) & NodeUtils.GetEffectiveGeometryStageCapability(startSlot, false);
192
193 foreach (var candidateAnchor in ports.ToList())
194 {
195 var candidateSlot = candidateAnchor.GetSlot();
196
197 if (!startSlot.IsCompatibleWith(candidateSlot))
198 continue;
199
200 if (startStage != GeometryStageCapability.All)
201 {
202 var candidateStage = candidateSlot.stageCapability;
203 if (candidateStage == GeometryStageCapability.All || candidateSlot.owner is SubGraphNode)
204 candidateStage = NodeUtils.GetEffectiveGeometryStageCapability(candidateSlot, true)
205 & NodeUtils.GetEffectiveGeometryStageCapability(candidateSlot, false);
206 if (candidateStage != GeometryStageCapability.All && candidateStage != startStage)
207 continue;
208
209 // None stage can only connect to All stage, otherwise you can connect invalid connections
210 if (startStage == GeometryStageCapability.None && candidateStage != GeometryStageCapability.All)
211 continue;
212 }
213
214 compatibleAnchors.Add(candidateAnchor);
215 }
216 return compatibleAnchors;
217 }

219 internal bool ResetSelectedBlockNodes()
220 {
221 bool anyNodesWereReset = false;
222 var selectedBlocknodes = selection.FindAll(e => e is GeometryNodeView && ((GeometryNodeView)e).node is BlockNode).Cast<GeometryNodeView>().ToArray();
223 foreach (var mNode in selectedBlocknodes)
224 {
225 var bNode = mNode.node as BlockNode;
226 var context = GetContext(bNode.contextData);
227
228 // Check if the node is currently floating (it's parent isn't the context view that owns it).
229 // If the node's not floating then the block doesn't need to be reset.
230 bool isFloating = mNode.parent != context;
231 if (!isFloating)
232 continue;
233
234 anyNodesWereReset = true;
235 RemoveElement(mNode);
236 context.InsertBlock(mNode);
237
238 // TODO: StackNode in GraphView (Trunk) has no interface to reset drop previews. The least intrusive
239 // solution is to call its DragLeave until its interface can be improved.
240 context.DragLeave(null, null, null, null);
241 }
242 if (selectedBlocknodes.Length > 0)
243 graph.ValidateCustomBlockLimit();
244 return anyNodesWereReset;

976 void DeleteSelectionImplementation(string operationName, GraphView.AskUser askUser)
977 {
978 // Selection state of Graph elements and the Focus state of UIElements are not mutually exclusive.
979 // For Hotkeys, askUser should be AskUser mode, which should early out so that the focused Element can win.
980 if (this.focusController.focusedElement != null
981 && focusController.focusedElement is UnityEditor.UIElements.ObjectField
982 && askUser == GraphView.AskUser.AskUser)
983 {
984 return;
985 }
986
987 bool containsProperty = false;
988
989 // Keywords need to be tested against variant limit based on multiple factors
990 bool keywordsDirty = false;
991 bool dropdownsDirty = false;
992
993 // Track dependent keyword nodes to remove them
994 List<KeywordNode> keywordNodes = new List<KeywordNode>();
995 List<DropdownNode> dropdownNodes = new List<DropdownNode>();
996
997 foreach (var selectable in selection)
998 {
999 if (selectable is GGBlackboardField propertyView && propertyView.userData != null)
1000 {
1001 switch (propertyView.userData)
1002 {
1003 case AbstractGeometryProperty property:
1004 containsProperty = true;
1005 break;
1006 case ShaderKeyword keyword:
1007 keywordNodes.AddRange(graph.GetNodes<KeywordNode>().Where(x => x.keyword == keyword));
1008 break;
1009 case GeometryDropdown dropdown:
1010 dropdownNodes.AddRange(graph.GetNodes<DropdownNode>().Where(x => x.dropdown == dropdown));
1011 break;
1012 default:
1013 throw new ArgumentOutOfRangeException();
1014 }
1015 }
1016 }
1017
1018 if (containsProperty)
1019 {
1020 if (graph.isSubGraph)
1021 {
1022 if (!EditorUtility.DisplayDialog("Sub Graph Will Change", "If you remove a property and save the sub graph, you might change other graphs that are using this sub graph.\n\nDo you want to continue?", "Yes", "No"))
1023 return;
1024 }
1025 }
1026
1027 // Filter nodes that cannot be deleted
1028 var nodesToDelete = selection.OfType<IGeometryNodeView>().Where(v => !(v.node is SubGraphOutputNode) && v.node.canDeleteNode).Select(x => x.node);
1029
1030 // Add keyword nodes dependent on deleted keywords
1031 nodesToDelete = nodesToDelete.Union(keywordNodes);
1032 nodesToDelete = nodesToDelete.Union(dropdownNodes);
1033
1034 // If deleting a Sub Graph node whose asset contains Keywords test variant limit
1035 foreach (SubGraphNode subGraphNode in nodesToDelete.OfType<SubGraphNode>())
1036 {
1037 if (subGraphNode.asset == null)
1038 {
1039 continue;
1040 }
1041 if (subGraphNode.asset.keywords.Any())
1042 {
1043 keywordsDirty = true;
1044 }
1045 if (subGraphNode.asset.dropdowns.Any())
1046 {
1047 dropdownsDirty = true;
1048 }
1049 }
1050
1051 graph.owner.RegisterCompleteObjectUndo(operationName);
1052 graph.RemoveElements(nodesToDelete.ToArray(),
1053 selection.OfType<UnityEditor.Experimental.GraphView.Edge>().Select(x => x.userData).OfType<IEdge>().ToArray(),
1054 selection.OfType<GeometryGroup>().Select(x => x.userData).ToArray(),
1055 selection.OfType<StickyNote>().Select(x => x.userData).ToArray());
1056
1057
1058 var copiedSelectionList = new List<ISelectable>(selection);
1059 var deleteGeometryInputAction = new DeleteGeometryInputAction();
1060 var deleteCategoriesAction = new DeleteCategoryAction();
1061
1062 for (int index = 0; index < copiedSelectionList.Count; ++index)
1063 {
1064 var selectable = copiedSelectionList[index];
1065 if (selectable is GGBlackboardField field && field.userData != null)
1066 {
1067 var input = (GeometryInput)field.userData;
1068 deleteGeometryInputAction.geometryInputsToDelete.Add(input);
1069
1070 // If deleting a Keyword test variant limit
1071 if (input is ShaderKeyword keyword)
1072 {
1073 keywordsDirty = true;
1074 }
1075 if (input is GeometryDropdown dropdown)
1076 {
1077 dropdownsDirty = true;
1078 }
1079 }
1080 // Don't allow the default category to be deleted
1081 else if (selectable is GGBlackboardCategory category && category.controller.Model.IsNamedCategory())
1082 {
1083 deleteCategoriesAction.categoriesToRemoveGuids.Add(category.viewModel.associatedCategoryGuid);
1084 }
1085 }
1086
1087 if (deleteGeometryInputAction.geometryInputsToDelete.Count != 0)
1088 graph.owner.graphDataStore.Dispatch(deleteGeometryInputAction);
1089 if (deleteCategoriesAction.categoriesToRemoveGuids.Count != 0)
1090 graph.owner.graphDataStore.Dispatch(deleteCategoriesAction);
1091
1092 // Test Keywords against variant limit
1093 if (keywordsDirty)
1094 {
1095 graph.OnKeywordChangedNoValidate();
1096 }
1097 if (dropdownsDirty)
1098 {
1099 graph.OnDropdownChangedNoValidate();
1100 }
1101
1102 selection.Clear();
1103 m_InspectorUpdateDelegate?.Invoke();
1104 }

1330 internal static void InsertCopyPasteGraph(this GeometryGraphView graphView, CopyPasteGraph copyGraph)
1331 {
1332 if (copyGraph == null)
1333 return;
1334
1335 // Keywords need to be tested against variant limit based on multiple factors
1336 bool keywordsDirty = false;
1337
1338 bool dropdownsDirty = false;
1339
1340 var blackboardController = graphView.GetFirstAncestorOfType<GraphEditorView>().blackboardController;
1341
1342 // Get the position to insert the new shader inputs per category
1343 int insertionIndex = blackboardController.GetInsertionIndexForPaste();
1344
1345 // Any child of the categories need to be removed from selection as well (there's a Graphview issue where these don't get properly added to selection before the duplication sometimes, have to do it manually)
1346 foreach (var selectable in graphView.selection)
1347 {
1348 if (selectable is GGBlackboardCategory blackboardCategory)
1349 {
1350 foreach (var blackboardChild in blackboardCategory.Children())
1351 {
1352 if (blackboardChild is GGBlackboardRow blackboardRow)
1353 {
1354 var blackboardField = blackboardRow.Q<GGBlackboardField>();
1355 if (blackboardField != null)
1356 {
1357 blackboardField.selected = false;
1358 blackboardField.OnUnselected();
1359 }
1360 }
1361 }
1362 }
1363 }
1364
1365 var cachedSelection = graphView.selection.ToList();
1366
1367 // Before copy-pasting, clear all current selections so the duplicated items may be selected instead
1368 graphView.ClearSelectionNoUndoRecord();
1369
1370 // Make new inputs from the copied graph
1371 foreach (GeometryInput input in copyGraph.inputs)
1372 {
1373 // Disregard any inputs that belong to a category in the CopyPasteGraph already,
1374 // GraphData handles copying those child inputs over when the category is copied
1375 if (copyGraph.IsInputCategorized(input))
1376 continue;
1377
1378 string associatedCategoryGuid = String.Empty;
1379
1380 foreach (var category in graphView.graph.categories)
1381 {
1382 if (copyGraph.IsInputDuplicatedFromCategory(input, category, graphView.graph))

1508 private static void ClampNodesWithinView(GeometryGraphView graphView, IEnumerable<IRectInterface> rectList)
1509 {
1510 // Compute the centroid of the copied elements at their original positions
1511 var positions = rectList.Select(n => n.rect.position);
1512 var centroid = UIUtilities.CalculateCentroid(positions);
1513
1514 /* Ensure nodes get pasted at cursor */
1515 var graphMousePosition = graphView.contentViewContainer.WorldToLocal(graphView.cachedMousePosition);
1516 var copiedNodesOrigin = graphMousePosition;
1517 float xMin = float.MaxValue, xMax = float.MinValue, yMin = float.MaxValue, yMax = float.MinValue;
1518
1519 // Calculate bounding rectangle min and max coordinates for these elements, to use in clamping later
1520 foreach (var element in rectList)
1521 {
1522 var position = element.rect.position;
1523 xMin = Mathf.Min(xMin, position.x);
1524 yMin = Mathf.Min(yMin, position.y);
1525 xMax = Mathf.Max(xMax, position.x);
1526 yMax = Mathf.Max(yMax, position.y);
1527 }
1528
1529 // Get center of the current view
1530 var center = graphView.contentViewContainer.WorldToLocal(graphView.layout.center);
1531 // Get offset from center of view to mouse position
1532 var mouseOffset = center - graphMousePosition;
1533
1534 var zoomAdjustedViewScale = 1.0f / graphView.scale;
1535 var graphViewScaledHalfWidth = (graphView.layout.width * zoomAdjustedViewScale) / 2.0f;
1536 var graphViewScaledHalfHeight = (graphView.layout.height * zoomAdjustedViewScale) / 2.0f;
1537 const float widthThreshold = 40.0f;
1538 const float heightThreshold = 20.0f;
1539
1540 if ((Mathf.Abs(mouseOffset.x) + widthThreshold > graphViewScaledHalfWidth ||
1541 (Mathf.Abs(mouseOffset.y) + heightThreshold > graphViewScaledHalfHeight)))
1542 {
1543 // Out of bounds - Adjust taking into account the size of the bounding box around elements and the current graph zoom level
1544 var adjustedPositionX = (xMax - xMin) + widthThreshold * zoomAdjustedViewScale;
1545 var adjustedPositionY = (yMax - yMin) + heightThreshold * zoomAdjustedViewScale;
1546 adjustedPositionY *= -1.0f * Mathf.Sign(copiedNodesOrigin.y);
1547 adjustedPositionX *= -1.0f * Mathf.Sign(copiedNodesOrigin.x);
1548 copiedNodesOrigin.x += adjustedPositionX;
1549 copiedNodesOrigin.y += adjustedPositionY;
1550 }
1551
1552 foreach (var element in rectList)
1553 {
1554 var rect = element.rect;
1555
1556 // Get the relative offset from the calculated centroid
1557 var relativeOffsetFromCentroid = rect.position - centroid;
1558 // Reapply that offset to ensure element positions are consistent when multiple elements are copied
1559 rect.x = copiedNodesOrigin.x + relativeOffsetFromCentroid.x;
1560 rect.y = copiedNodesOrigin.y + relativeOffsetFromCentroid.y;
1561 element.rect = rect;
1562 }
1563 }
1564 }
