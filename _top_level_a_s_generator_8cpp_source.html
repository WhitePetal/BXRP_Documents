26/*
27Contacts for feedback:
28- pgautron@nvidia.com (Pascal Gautron)
29- mlefrancois@nvidia.com (Martin-Karl Lefrancois)
30
31The top-level hierarchy is used to store a set of instances represented by
32bottom-level hierarchies in a way suitable for fast intersection at runtime. To
33be built, this data structure requires some scratch space which has to be
34allocated by the application. Similarly, the resulting data structure is stored
35in an application-controlled buffer.
36
37To be used, the application must first add all the instances to be contained in
38the final structure, using AddInstance. After all instances have been added,
39ComputeASBufferSizes will prepare the build, and provide the required sizes for
40the scratch data and the final result. The Build call will finally compute the
41acceleration structure and store it in the result buffer.
42
43Note that the build is enqueued in the command list, meaning that the scratch
44buffer needs to be kept until the command list execution is finished.
45
46*/

61// Add an instance to the top-level acceleration structure. The instance is
62// represented by a bottom-level AS, a transform, an instance ID and the index
63// of the hit group indicating which shaders are executed upon hitting any

66 ID3D12Resource* bottomLevelAS, // Bottom-level acceleration structure containing the
67 // actual geometric data of the instance
68 const DirectX::XMMATRIX& transform, // Transform matrix to apply to the instance, allowing the
69 // same bottom-level AS to be used at several world-space
70 // positions
71 UINT instanceID, // Instance ID, which can be used in the shaders to
72 // identify this specific instance
73 UINT hitGroupIndex // Hit group index, corresponding the the index of the
74 // hit group in the Shader Binding Table that will be
75 // invocated upon hitting the geometry
76)
77{
78 m_instances.emplace_back(Instance(bottomLevelAS, transform, instanceID, hitGroupIndex));
79}

81//--------------------------------------------------------------------------------------------------
82//
83// Compute the size of the scratch space required to build the acceleration
84// structure, as well as the size of the resulting structure. The allocation of
85// the buffers is then left to the application
86void TopLevelASGenerator::ComputeASBufferSizes(
87 ID3D12Device5* device, // Device on which the build will be performed
88 bool allowUpdate, // If true, the resulting acceleration structure will
89 // allow iterative updates
90 UINT64* scratchSizeInBytes, // Required scratch memory on the GPU to build
91 // the acceleration structure
92 UINT64* resultSizeInBytes, // Required GPU memory to store the acceleration
93 // structure
94 UINT64* descriptorsSizeInBytes // Required GPU memory to store instance
95 // descriptors, containing the matrices,
96 // indices etc.
97)
98{
99 // The generated AS can support iterative updates. This may change the final
100 // size of the AS as well as the temporary memory requirements, and hence has
101 // to be set before the actual build
102 m_flags = allowUpdate ? D3D12_RAYTRACING_ACCELERATION_STRUCTURE_BUILD_FLAG_ALLOW_UPDATE
103 : D3D12_RAYTRACING_ACCELERATION_STRUCTURE_BUILD_FLAG_NONE;
104
105 // Describe the work being requested, in this case the construction of a
106 // (possibly dynamic) top-level hierarchy, with the given instance descriptors
107 D3D12_BUILD_RAYTRACING_ACCELERATION_STRUCTURE_INPUTS
108 prebuildDesc = {};
109 prebuildDesc.Type = D3D12_RAYTRACING_ACCELERATION_STRUCTURE_TYPE_TOP_LEVEL;
110 prebuildDesc.DescsLayout = D3D12_ELEMENTS_LAYOUT_ARRAY;
111 prebuildDesc.NumDescs = static_cast<UINT>(m_instances.size());
112 prebuildDesc.Flags = m_flags;
113
114 // This structure is used to hold the sizes of the required scratch memory and
115 // resulting AS
116 D3D12_RAYTRACING_ACCELERATION_STRUCTURE_PREBUILD_INFO info = {};
117
118 // Building the acceleration structure (AS) requires some scratch space, as
119 // well as space to store the resulting structure This function computes a
120 // conservative estimate of the memory requirements for both, based on the
121 // number of bottom-level instances.
122 device->GetRaytracingAccelerationStructurePrebuildInfo(&prebuildDesc, &info);
123
124 // Buffer sizes need to be 256-byte-aligned
125 info.ResultDataMaxSizeInBytes =
126 ROUND_UP(info.ResultDataMaxSizeInBytes, D3D12_CONSTANT_BUFFER_DATA_PLACEMENT_ALIGNMENT);
127 info.ScratchDataSizeInBytes =
128 ROUND_UP(info.ScratchDataSizeInBytes, D3D12_CONSTANT_BUFFER_DATA_PLACEMENT_ALIGNMENT);
129
130 m_resultSizeInBytes = info.ResultDataMaxSizeInBytes;
131 m_scratchSizeInBytes = info.ScratchDataSizeInBytes;
132 // The instance descriptors are stored as-is in GPU memory, so we can deduce
133 // the required size from the instance count
134 m_instanceDescsSizeInBytes =
135 ROUND_UP(sizeof(D3D12_RAYTRACING_INSTANCE_DESC) * static_cast<UINT64>(m_instances.size()),
136 D3D12_CONSTANT_BUFFER_DATA_PLACEMENT_ALIGNMENT);
137
138 *scratchSizeInBytes = m_scratchSizeInBytes;
139 *resultSizeInBytes = m_resultSizeInBytes;
140 *descriptorsSizeInBytes = m_instanceDescsSizeInBytes;
141}
142
143//--------------------------------------------------------------------------------------------------
144//
145// Enqueue the construction of the acceleration structure on a command list,
146// using application-provided buffers and possibly a pointer to the previous
147// acceleration structure in case of iterative updates. Note that the update can
148// be done in place: the result and previousResult pointers can be the same.
149void TopLevelASGenerator::Generate(
150 ID3D12GraphicsCommandList4* commandList, // Command list on which the build will be enqueued
151 ID3D12Resource* scratchBuffer, // Scratch buffer used by the builder to
152 // store temporary data
153 ID3D12Resource* resultBuffer, // Result buffer storing the acceleration structure
154 ID3D12Resource* descriptorsBuffer, // Auxiliary result buffer containing the instance
155 // descriptors, has to be in upload heap
156 bool updateOnly /*= false*/, // If true, simply refit the existing
157 // acceleration structure
158 ID3D12Resource* previousResult /*= nullptr*/ // Optional previous acceleration
159 // structure, used if an iterative update
160 // is requested
161)
162{
163 // Copy the descriptors in the target descriptor buffer
164 D3D12_RAYTRACING_INSTANCE_DESC* instanceDescs;
165 descriptorsBuffer->Map(0, nullptr, reinterpret_cast<void**>(&instanceDescs));
166 if (!instanceDescs)
167 {
168 throw std::logic_error("Cannot map the instance descriptor buffer - is it "
169 "in the upload heap?");
170 }
171
172 auto instanceCount = static_cast<UINT>(m_instances.size());
173
174 // Initialize the memory to zero on the first time only
175 if (!updateOnly)
176 {
177 ZeroMemory(instanceDescs, m_instanceDescsSizeInBytes);
178 }
179
180 // Create the description for each instance
181 for (uint32_t i = 0; i < instanceCount; i++)
182 {
183 // Instance ID visible in the shader in InstanceID()
184 instanceDescs[i].InstanceID = m_instances[i].instanceID;
185 // Index of the hit group invoked upon intersection
186 instanceDescs[i].InstanceContributionToHitGroupIndex = m_instances[i].hitGroupIndex;
187 // Instance flags, including backface culling, winding, etc - TODO: should
188 // be accessible from outside
189 instanceDescs[i].Flags = D3D12_RAYTRACING_INSTANCE_FLAG_NONE;
190 // Instance transform matrix
191 DirectX::XMMATRIX m = XMMatrixTranspose(
192 m_instances[i].transform); // GLM is column major, the INSTANCE_DESC is row major
193 memcpy(instanceDescs[i].Transform, &m, sizeof(instanceDescs[i].Transform));
194 // Get access to the bottom level
195 instanceDescs[i].AccelerationStructure = m_instances[i].bottomLevelAS->GetGPUVirtualAddress();
196 // Visibility mask, always visible here - TODO: should be accessible from
197 // outside
198 instanceDescs[i].InstanceMask = 0xFF;
199 }
200
201 descriptorsBuffer->Unmap(0, nullptr);
202
203 // If this in an update operation we need to provide the source buffer
204 D3D12_GPU_VIRTUAL_ADDRESS pSourceAS = updateOnly ? previousResult->GetGPUVirtualAddress() : 0;
205
206 D3D12_RAYTRACING_ACCELERATION_STRUCTURE_BUILD_FLAGS flags = m_flags;
207 // The stored flags represent whether the AS has been built for updates or
208 // not. If yes and an update is requested, the builder is told to only update
209 // the AS instead of fully rebuilding it
210 if (flags == D3D12_RAYTRACING_ACCELERATION_STRUCTURE_BUILD_FLAG_ALLOW_UPDATE && updateOnly)
211 {
212 flags = D3D12_RAYTRACING_ACCELERATION_STRUCTURE_BUILD_FLAG_PERFORM_UPDATE;
213 }
214
215 // Sanity checks
216 if (m_flags != D3D12_RAYTRACING_ACCELERATION_STRUCTURE_BUILD_FLAG_ALLOW_UPDATE && updateOnly)
217 {
218 throw std::logic_error("Cannot update a top-level AS not originally built for updates");
219 }
220 if (updateOnly && previousResult == nullptr)
221 {
222 throw std::logic_error("Top-level hierarchy update requires the previous hierarchy");
223 }
224
225 // Create a descriptor of the requested builder work, to generate a top-level
226 // AS from the input parameters
227 D3D12_BUILD_RAYTRACING_ACCELERATION_STRUCTURE_DESC buildDesc = {};
228 buildDesc.Inputs.Type = D3D12_RAYTRACING_ACCELERATION_STRUCTURE_TYPE_TOP_LEVEL;
229 buildDesc.Inputs.DescsLayout = D3D12_ELEMENTS_LAYOUT_ARRAY;
230 buildDesc.Inputs.InstanceDescs = descriptorsBuffer->GetGPUVirtualAddress();
231 buildDesc.Inputs.NumDescs = instanceCount;
232 buildDesc.DestAccelerationStructureData = {resultBuffer->GetGPUVirtualAddress()
233 };
234 buildDesc.ScratchAccelerationStructureData = {scratchBuffer->GetGPUVirtualAddress()
235 };
236 buildDesc.SourceAccelerationStructureData = pSourceAS;
237 buildDesc.Inputs.Flags = flags;
238
239 // Build the top-level AS
240 commandList->BuildRaytracingAccelerationStructure(&buildDesc, 0, nullptr);
241
242 // Wait for the builder to complete by setting a barrier on the resulting
243 // buffer. This can be important in case the rendering is triggered
244 // immediately afterwards, without executing the command list
245 D3D12_RESOURCE_BARRIER uavBarrier;
246 uavBarrier.Type = D3D12_RESOURCE_BARRIER_TYPE_UAV;
247 uavBarrier.UAV.pResource = resultBuffer;
248 uavBarrier.Flags = D3D12_RESOURCE_BARRIER_FLAG_NONE;
249 commandList->ResourceBarrier(1, &uavBarrier);
250}
