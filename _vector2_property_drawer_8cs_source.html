30 m_KeyDownCallback = new EventCallback<KeyDownEvent>(evt =>
31 {
32 // Record Undo for input field edit
33 if (mUndoGroup == -1)
34 {
35 mUndoGroup = Undo.GetCurrentGroup();
36 preValueChangeCallback?.Invoke();
37 }
38 // Handle escaping input field edit
39 if (evt.keyCode == KeyCode.Escape && mUndoGroup > -1)
40 {
41 Undo.RevertAllDownToGroup(mUndoGroup);
42 mUndoGroup = -1;
43 evt.StopPropagation();
44 }
45 // Dont record Undo again until input field is unfocused
46 mUndoGroup++;
47 postValueChangeCallback?.Invoke();
48 });
49
50 m_FocusOutCallback = new EventCallback<FocusOutEvent>(evt =>
51 {
52 // Reset UndoGroup when done editing input field
53 mUndoGroup = -1;
54 });
