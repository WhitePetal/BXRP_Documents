193 public override void OnAfterDeserialize()
194 {
195 // TODO: Go find someone to tell @esme not to do this.

198 string descName = k_CustomBlockDefaultName;

202 name = $"{descTag}.{descName}";

204 var wsplit = m_SerializedDescriptor.Split(new char[] { '#', '.' });
205
206 try
207 {
208 descWidth = (CustomBlockType)int.Parse(wsplit[2]);
209 }
210 catch
211 {
212 Debug.LogWarning(String.Format("Bad width found while deserializing custom interpolator {0}, defaulting to 4.", m_SerializedDescriptor));
213 descWidth = CustomBlockType.Vector4;
214 }
215
216 IControl control;
217 try { control = (IControl)FindSlot<GeometrySlot>(0).InstantiateControl(); }
218 catch { control = WidthToControl((int)descWidth); }
219
220 descName = NodeUtils.ConvertToValidHLSLIdentifier(wsplit[1]);
221 m_Descriptor = new BlockFieldDescriptor(descTag, descName, "", control, GeometryStage.Geometry, isCustom: true);
222 }

225 private static BlockFieldDescriptor MakeCustomBlockField(string name, CustomBlockType width)
226 {
227 name = NodeUtils.ConvertToValidHLSLIdentifier(name);
228 var referenceName = name;
229 var define = "";
230 IControl control = WidthToControl((int)width);
231 var tag = BlockFields.GeometryDescription.name;
232
233 return new BlockFieldDescriptor(tag, referenceName, define, control, GeometryStage.Geometry, isCustom: true);
234 }

236 private static IControl WidthToControl(int width)
237 {
238 switch (width)
239 {
240 case 1: return new FloatControl(default(float));
241 case 2: return new Vector2Control(default(Vector2));
242 case 3: return new Vector3Control(default(Vector3));
243 case 4: return new Vector4Control(default(Vector4));
244 default: return null;
245 }
246 }
