31The ShaderBindingTable is a helper to construct the SBT. It helps to maintain the
32proper offsets of each element, required when constructing the SBT, but also when filling the
33dispatch rays description.
34
35Example:
36
37
38D3D12_GPU_DESCRIPTOR_HANDLE srvUavHeapHandle = m_srvUavHeap->GetGPUDescriptorHandleForHeapStart();
39UINT64* heapPointer = reinterpret_cast< UINT64* >(srvUavHeapHandle.ptr);
40
41m_sbtHelper.AddRayGenerationProgram(L"RayGen", {heapPointer});
42m_sbtHelper.AddMissProgram(L"Miss", {});
43
44m_sbtHelper.AddHitGroup(L"HitGroup",
45{(void*)(m_constantBuffers[i]->GetGPUVirtualAddress())});
46m_sbtHelper.AddHitGroup(L"ShadowHitGroup", {});
47
48
49// Create the SBT on the upload heap
50uint32_t sbtSize = 0;
51m_sbtHelper.ComputeSBTSize(GetRTDevice(), &sbtSize);
52m_sbtStorage = nv_helpers_dx12::CreateBuffer(m_device.Get(), sbtSize,
53D3D12_RESOURCE_FLAG_NONE, D3D12_RESOURCE_STATE_GENERIC_READ,
54nv_helpers_dx12::kUploadHeapProps);
55
56m_sbtHelper.Generate(m_sbtStorage.Get(), m_rtStateObjectProps.Get());
57
58
59//--------------------------------------------------------------------
60Then setting the descriptor for the dispatch rays become way easier
61//--------------------------------------------------------------------
62
63D3D12_DISPATCH_RAYS_DESC desc = {};
64// The layout of the SBT is as follows: ray generation shaders, miss shaders,
65hit groups. As
66// described in the CreateShaderBindingTable method, all SBT entries have the
67same size to allow
68// a fixed stride.
69
70// The ray generation shaders are always at the beginning of the SBT. In this
71// example we have only one RG, so the size of this SBT sections is
72m_sbtEntrySize. uint32_t rayGenerationSectionSizeInBytes =
73m_sbtHelper.GetRayGenSectionSize(); desc.RayGenerationShaderRecord.StartAddress
74= m_sbtStorage->GetGPUVirtualAddress();
75desc.RayGenerationShaderRecord.SizeInBytes = rayGenerationSectionSizeInBytes;
76
77// The miss shaders are in the second SBT section, right after the ray
78generation shader. We
79// have one miss shader for the camera rays and one for the shadow rays, so this
80section has a
81// size of 2*m_sbtEntrySize. We also indicate the stride between the two miss
82shaders, which is
83// the size of a SBT entry
84uint32_t missSectionSizeInBytes = m_sbtHelper.GetMissSectionSize();
85desc.MissShaderTable.StartAddress =
86m_sbtStorage->GetGPUVirtualAddress() + rayGenerationSectionSizeInBytes;
87desc.MissShaderTable.SizeInBytes = missSectionSizeInBytes;
88desc.MissShaderTable.StrideInBytes = m_sbtHelper.GetMissEntrySize();
89
90// The hit groups section start after the miss shaders. In this sample we have 4
91hit groups: 2
92// for the triangles (1 used when hitting the geometry from a camera ray, 1 when
93hitting the
94// same geometry from a shadow ray) and 2 for the plane. We also indicate the
95stride between the
96// two miss shaders, which is the size of a SBT entry
97// #Pascal: experiment with different sizes for the SBT entries
98uint32_t hitGroupsSectionSize = m_sbtHelper.GetHitGroupSectionSize();
99desc.HitGroupTable.StartAddress =
100m_sbtStorage->GetGPUVirtualAddress() + rayGenerationSectionSizeInBytes +
101missSectionSizeInBytes; desc.HitGroupTable.SizeInBytes = hitGroupsSectionSize;
102desc.HitGroupTable.StrideInBytes = m_sbtHelper.GetHitGroupEntrySize();
103
104
105
