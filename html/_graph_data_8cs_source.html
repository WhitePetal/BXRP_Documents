446 public void SetTargetActive(Target target, bool skipSortAndUpdate = false)
447 {
448 int activeIndex = m_ActiveTargets.IndexOf(target);
449 if(activeIndex < 0)
450 {
451 activeIndex = m_ActiveTargets.Count;
452 m_ActiveTargets.Add(target);
453 }
454
455 // active known targets should replace the stored Target in AllPotentialTargets
456 if(target is MultiJsonInternal.UnknownTargetType unknownTarget)
457 {
458 // find any existing potential target with the same unknown jsonData
459 int targetIndex = m_AllPotentialTargets.FindIndex(
460 pt => pt.IsUnknown() && (pt.GetUnknown().jsonData == unknownTarget.jsonData));
461
462 // replace existing target, or add it if there is none
463 if (targetIndex > 0)
464 m_AllPotentialTargets[targetIndex] = new PotentialTarget(target);
465 else
466 m_AllPotentialTargets.Add(new PotentialTarget(target));
467 }
468 else
469 {
470 // known types should already have been registered
471 Type targetType = target.GetType();
472 int targetIndex = GetTargetIndexByKnownType(targetType);
473 Assert.IsTrue(targetIndex >= 0);
474 m_AllPotentialTargets[targetIndex].ReplaceStoredTarget(target);
475 }
476
477 if (!skipSortAndUpdate)
478 SortAndUpdateActiveTargets();
479 }

488 {

564 var asmTypes = TypeCache.GetTypesWithAttribute<GenerateBlocksAttribute>();

603 }

807 public List<BlockFieldDescriptor> GetActiveBlocksForAllActiveTargets()
808 {
809 // Get list of active Block types
810 var currentBlocks = GetNodes<BlockNode>();
811 var context = new TargetActiveBlockContext(currentBlocks.Select(x => x.descriptor).ToList(), null);
812 foreach(var target in activeTargets)
813 {
814 target.GetActiveBlocks(ref context);
815 }
816
817 // custom blocks aren't going to exist in GetActiveBlocks, we need to ensure we grab those too.
818 foreach(var cibnode in currentBlocks.Where(bn => bn.isCustomBlock))

973 {

975 var toNode = toSlotRef.node;
976
977 if (fromNode == null || toNode == null)
978 return null;
979
980 // both nodes must belong to this graph
981 if ((fromNode.owner != this) || (toNode.owner != this))
982 return null;

984 // if fromNode is already connected to toNode
985 // do now allow a connection as toNode will then
986 // have an edge to fromNode creating a cycle.
987 // if this is parsed it will lead to an infinite loop.

997 var toSlot = toNode.FindSlot<GeometrySlot>(toSlotRef.slotId);

1002 if (fromSlot.isOutputSlot == toSlot.isOutputSlot)
1003 return null;

1006 var inputSlot = fromSlot.isInputSlot ? fromSlotRef : toSlotRef;
