46// Add a set of heap range descriptors as a parameter of the root signature.
47void RootSignatureGenerator::AddHeapRangesParameter(
48 const std::vector<D3D12_DESCRIPTOR_RANGE>& ranges)
49{
50 m_ranges.push_back(ranges);
51
52 // A set of ranges on the heap is a descriptor table parameter
53 D3D12_ROOT_PARAMETER param = {};
54 param.ParameterType = D3D12_ROOT_PARAMETER_TYPE_DESCRIPTOR_TABLE;
55 param.DescriptorTable.NumDescriptorRanges = static_cast<UINT>(ranges.size());
56 // The range pointer is kept null here, and will be resolved when generating the root signature
57 // (see explanation of m_rangeLocations below)
58 param.DescriptorTable.pDescriptorRanges = nullptr;
59
60 // All parameters (heap ranges and root parameters) are added to the same parameter list to
61 // preserve order
62 m_parameters.push_back(param);
63
64 // The descriptor table descriptor ranges require a pointer to the descriptor ranges. Since new
65 // ranges can be dynamically added in the vector, we separately store the index of the range set.
66 // The actual address will be solved when generating the actual root signature
67 m_rangeLocations.push_back(static_cast<UINT>(m_ranges.size() - 1));
68}
69
70//--------------------------------------------------------------------------------------------------
71//
72// Add a set of heap ranges as a parameter of the root signature. Each range
73// is defined as follows:
74// - UINT BaseShaderRegister: the first register index in the range, e.g. the
75// register of a UAV with BaseShaderRegister==0 is defined in the HLSL code
76// as register(u0)
77// - UINT NumDescriptors: number of descriptors in the range. Those will be
78// mapped to BaseShaderRegister, BaseShaderRegister+1 etc. UINT
79// RegisterSpace: Allows using the same register numbers multiple times by
80// specifying a space where they are defined, similarly to a namespace. For
81// example, a UAV with BaseShaderRegister==0 and RegisterSpace==1 is accessed
82// in HLSL using the syntax register(u0, space1)
83// - D3D12_DESCRIPTOR_RANGE_TYPE RangeType: The range type, such as
84// D3D12_DESCRIPTOR_RANGE_TYPE_CBV for a constant buffer
85// - UINT OffsetInDescriptorsFromTableStart: The first slot in the heap
86// corresponding to the buffers mapped by the root signature. This can either
87// be explicit, or implicit using D3D12_DESCRIPTOR_RANGE_OFFSET_APPEND. In
88// this case the index in the heap is the one directly following the last
89// parameter range (or 0 if it's the first)
90void RootSignatureGenerator::AddHeapRangesParameter(
91 std::vector<std::tuple<UINT, /* BaseShaderRegister, */ UINT, /* NumDescriptors */ UINT,
92 /* RegisterSpace */ D3D12_DESCRIPTOR_RANGE_TYPE,
93 /* RangeType */ UINT /* OffsetInDescriptorsFromTableStart */>>
94 ranges)
95{
96 // Build and store the set of descriptors for the ranges
97 std::vector<D3D12_DESCRIPTOR_RANGE> rangeStorage;
98 for (const auto& input : ranges)
99 {
100 D3D12_DESCRIPTOR_RANGE r = {};
101 r.BaseShaderRegister = std::get<RSC_BASE_SHADER_REGISTER>(input);
102 r.NumDescriptors = std::get<RSC_NUM_DESCRIPTORS>(input);
103 r.RegisterSpace = std::get<RSC_REGISTER_SPACE>(input);
104 r.RangeType = std::get<RSC_RANGE_TYPE>(input);
105 r.OffsetInDescriptorsFromTableStart =
106 std::get<RSC_OFFSET_IN_DESCRIPTORS_FROM_TABLE_START>(input);
107 rangeStorage.push_back(r);
108 }
109
110 // Add those ranges to the heap parameters
111 AddHeapRangesParameter(rangeStorage);
112}

114//--------------------------------------------------------------------------------------------------
115//
116// Add a root parameter to the shader, defined by its type: constant buffer (CBV), shader
117// resource (SRV), unordered access (UAV), or root constant (CBV, directly defined by its value
118// instead of a buffer). The shaderRegister and registerSpace indicate how to access the
119// parameter in the HLSL code, e.g a SRV with shaderRegister==1 and registerSpace==0 is
120// accessible via register(t1, space0).
121// In case of a root constant, the last parameter indicates how many successive 32-bit constants
122// will be bound.
123void RootSignatureGenerator::AddRootParameter(D3D12_ROOT_PARAMETER_TYPE type,
124 UINT shaderRegister /*= 0*/,
125 UINT registerSpace /*= 0*/,
126 UINT numRootConstants /*= 1*/)
127{
128 D3D12_ROOT_PARAMETER param = {};
129 param.ParameterType = type;
130 // The descriptor is an union, so specific values need to be set in case the parameter is a
131 // constant instead of a buffer.
132 if (type == D3D12_ROOT_PARAMETER_TYPE_32BIT_CONSTANTS)
133 {
134 param.Constants.Num32BitValues = numRootConstants;
135 param.Constants.RegisterSpace = registerSpace;
136 param.Constants.ShaderRegister = shaderRegister;
137 }
138 else
139 {
140 param.Descriptor.RegisterSpace = registerSpace;
141 param.Descriptor.ShaderRegister = shaderRegister;
142 }
143
144 // We default the visibility to all shaders
145 param.ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
146
147 // Add the root parameter to the set of parameters,
148 m_parameters.push_back(param);
149 // and indicate that there will be no range
150 // location to indicate since this parameter is not part of the heap
151 m_rangeLocations.push_back(~0u);
152}

154//--------------------------------------------------------------------------------------------------
155//
156// Create the root signature from the set of parameters, in the order of the addition calls
157ID3D12RootSignature* RootSignatureGenerator::Generate(ID3D12Device* device, bool isLocal)
158{
159 // Go through all the parameters, and set the actual addresses of the heap range descriptors based
160 // on their indices in the range set array
161 for (size_t i = 0; i < m_parameters.size(); i++)
162 {
163 if (m_parameters[i].ParameterType == D3D12_ROOT_PARAMETER_TYPE_DESCRIPTOR_TABLE)
164 {
165 m_parameters[i].DescriptorTable.pDescriptorRanges = m_ranges[m_rangeLocations[i]].data();
166 }
167 }
168 // Specify the root signature with its set of parameters
169 D3D12_ROOT_SIGNATURE_DESC rootDesc = {};
170 rootDesc.NumParameters = static_cast<UINT>(m_parameters.size());
171 rootDesc.pParameters = m_parameters.data();
172 // Set the flags of the signature. By default root signatures are global, for example for vertex
173 // and pixel shaders. For raytracing shaders the root signatures are local.
174 rootDesc.Flags =
175 isLocal ? D3D12_ROOT_SIGNATURE_FLAG_LOCAL_ROOT_SIGNATURE : D3D12_ROOT_SIGNATURE_FLAG_NONE;
176
177 // Create the root signature from its descriptor
178 ID3DBlob* pSigBlob;
179 ID3DBlob* pErrorBlob;
180 HRESULT hr = D3D12SerializeRootSignature(&rootDesc, D3D_ROOT_SIGNATURE_VERSION_1_0, &pSigBlob,
181 &pErrorBlob);
182 if (FAILED(hr))
183 {
184 throw std::logic_error("Cannot serialize root signature");
185 }
186 ID3D12RootSignature* pRootSig;
187 hr = device->CreateRootSignature(0, pSigBlob->GetBufferPointer(), pSigBlob->GetBufferSize(),
188 IID_PPV_ARGS(&pRootSig));
189 if (FAILED(hr))
190 {
191 throw std::logic_error("Cannot create root signature");
192 }
193 return pRootSig;
194}
