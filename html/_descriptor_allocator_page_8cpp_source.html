47DescriptorAllocation DescriptorAllocatorPage::Allocate(uint32_t numDescriptors)
48{
49 std::lock_guard<std::mutex> lock(m_AllocationMutex);
50
51 // There are less than the requested number of descriptors left in the heap
52 // Return a NULL descriptor and try another heap
53 if (numDescriptors > m_NumFreeHandles)
54 {
55 return DescriptorAllocation();
56 }
57
58 // Get the first block that is large enough to satisfy the request
59 auto smallestBlockIt = m_FreeListBySize.lower_bound(numDescriptors);
60 if (smallestBlockIt == m_FreeListBySize.end())
61 {
62 // There was no free block that could satisfy the request
63 return DescriptorAllocation();
64 }
65
66 // The size of the smallest block that satisfies the request
67 auto blockSize = smallestBlockIt->first;
68
69 // The pointer to the same entry in the FreeListByOffset map
70 auto offsetIt = smallestBlockIt->second;
71
72 // The offset in the descriptor heap
73 auto offset = offsetIt->first;
74
75 // Remove the existing free block from the free list
76 m_FreeListBySize.erase(smallestBlockIt);
77 m_FreeListByOffset.erase(offsetIt);
78
79 // Compute the new free block that reults from splitting this block
80 auto newSize = blockSize - numDescriptors;
81 if (newSize > 0)
82 {
83 // If the allocation didn't exactly match the requested size
84 // return the left-over to the free list
85 auto newOffset = offset + numDescriptors;
86 AddNewBlock(newOffset, newSize);
87 }
88
89 // Decrement free handles
90 m_NumFreeHandles -= numDescriptors;
91
92 return DescriptorAllocation(
93 CD3DX12_CPU_DESCRIPTOR_HANDLE(m_BaseDescriptor, offset, m_DescriptorHandleIncrementSize),
94 numDescriptors, m_DescriptorHandleIncrementSize, shared_from_this());
95}

112void DescriptorAllocatorPage::FreeBlock(uint32_t offset, uint32_t numDescriptors)
113{
114 // Find the first element whose offset is greather than the specified offset.
115 // This is the block that should appear after the block that is being freed
116 auto nextBlockIt = m_FreeListByOffset.upper_bound(offset);
117
118 // Find the block that appears before the block being freed.
119 auto prevBlockIt = nextBlockIt;
120 if (prevBlockIt != m_FreeListByOffset.begin())
121 {
122 // Go to the previous block in the list
123 --prevBlockIt;
124 }
125 else
126 {
127 // Otherwise, just set it to the end of the list to indicate that no
128 // block comes before the one being freed.
129 prevBlockIt = m_FreeListByOffset.end();
130 }
131
132 // Add the number of free handles back to the heap.
133 // This needs to be done before merging any blocks since merging
134 // blocks modifies the numDescriptors variable.
135 m_NumFreeHandles += numDescriptors;
136
137 if (prevBlockIt != m_FreeListByOffset.end() && offset == prevBlockIt->first + prevBlockIt->second.Size)
138 {
139 // The previous block is exactly behind the block that is to be freed.
140 //
141 // PrevBlock.Offset Offset
142 // | |
143 // |<----PrevBlock.size---->|<-----Size----->|
144 //
145
146 // Increase the block size by the size of merging with the previous block
147 offset = prevBlockIt->first;
148 numDescriptors += prevBlockIt->second.Size;
149
150 m_FreeListBySize.erase(prevBlockIt->second.FreeListBySizeIt);
151 m_FreeListByOffset.erase(prevBlockIt);
152 }
153
154 if (nextBlockIt != m_FreeListByOffset.end() && offset + numDescriptors == nextBlockIt->first)
155 {
156 // The next block is exactly in front of the block that is to be freed.
157 //
158 // Offset NextBlock.Offset
159 // | |
160 // |<-------Size------->|<-----NextBlock.Size----->|
161
162 // Increase the block size by the size of merging with the next block.
163 numDescriptors += nextBlockIt->second.Size;
164
165 // Remove the next block from the free list
166 m_FreeListBySize.erase(nextBlockIt->second.FreeListBySizeIt);
167 m_FreeListByOffset.erase(nextBlockIt);
168 }
169
170 // Add the freed block to the free list.
171 AddNewBlock(offset, numDescriptors);
172}
