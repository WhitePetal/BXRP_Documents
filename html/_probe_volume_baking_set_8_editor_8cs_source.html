440 internal static int MaxSubdivLevelInProbeVolume(Vector3 volumeSize, int maxSubdiv)
441 {
442 float maxSizeDim = Mathf.Max(volumeSize.x, Mathf.Max(volumeSize.y, volumeSize.z));
443 float maxSideInBricks = maxSizeDim / ProbeReferenceVolume.instance.MinDistanceBetweenProbes();
444 int subdiv = Mathf.FloorToInt(Mathf.Log(maxSideInBricks, 3)) - 1;
445
446 return Mathf.Max(subdiv, maxSubdiv);
447 }

449 private static void InflateBound(ref Bounds bounds, ProbeVolume pv)
450 {
451 Bounds originalBounds = bounds;
452 // Round the probe volume bounds to cell size
453 float cellSize = ProbeReferenceVolume.instance.MaxBrickSize();
454 Vector3 cellOffset = ProbeReferenceVolume.instance.ProbeOffset();
455
456 // Expand the probe volume bounds to snap on the cell size grid
457 bounds.Encapsulate((Vector3)math.floor((bounds.min - cellOffset) / cellSize) * cellSize + cellOffset);
458 bounds.Encapsulate((Vector3)math.ceil((bounds.max - cellOffset) / cellSize) * cellSize + cellOffset);
459
460 // calculate how much padding we need to remove according to the brick generation in ProbePlacement.cs:
461 var cellSizeVector = new Vector3(cellSize, cellSize, cellSize);
462 var minPadding = (bounds.min - originalBounds.min);
463 var maxPadding = (bounds.max - originalBounds.max);
464 minPadding = cellSizeVector - new Vector3(Mathf.Abs(minPadding.x), Mathf.Abs(minPadding.y), Mathf.Abs(minPadding.z));
465 maxPadding = cellSizeVector - new Vector3(Mathf.Abs(maxPadding.x), Mathf.Abs(maxPadding.y), Mathf.Abs(maxPadding.z));
466
467 // Find the size of the brick we can put for every axis given the padding size
468 int maxSubdiv = ProbeReferenceVolume.instance.GetMaxSubdivision() - 1;
469 if (pv.overridesSubdivLevels) maxSubdiv = Mathf.Min(pv.highestSubdivLevelOverride, maxSubdiv);
470
471 float rightPaddingSubdivLevel = ProbeReferenceVolume.instance.BrickSize(MaxSubdivLevelInProbeVolume(new Vector3(maxPadding.x, originalBounds.size.y, originalBounds.size.z), maxSubdiv));
472 float leftPaddingSubdivLevel = ProbeReferenceVolume.instance.BrickSize(MaxSubdivLevelInProbeVolume(new Vector3(minPadding.x, originalBounds.size.y, originalBounds.size.z), maxSubdiv));
473 float topPaddingSubdivLevel = ProbeReferenceVolume.instance.BrickSize(MaxSubdivLevelInProbeVolume(new Vector3(originalBounds.size.x, maxPadding.y, originalBounds.size.z), maxSubdiv));
474 float bottomPaddingSubdivLevel = ProbeReferenceVolume.instance.BrickSize(MaxSubdivLevelInProbeVolume(new Vector3(originalBounds.size.x, minPadding.y, originalBounds.size.z), maxSubdiv));
475 float forwardPaddingSubdivLevel = ProbeReferenceVolume.instance.BrickSize(MaxSubdivLevelInProbeVolume(new Vector3(originalBounds.size.x, originalBounds.size.y, maxPadding.z), maxSubdiv));
476 float backPaddingSubdivLevel = ProbeReferenceVolume.instance.BrickSize(MaxSubdivLevelInProbeVolume(new Vector3(originalBounds.size.x, originalBounds.size.y, minPadding.z), maxSubdiv));
477
478 // Remove the extra padding caused by cell rounding
479 bounds.min = bounds.min + new Vector3(
480 leftPaddingSubdivLevel * Mathf.Floor(Mathf.Abs(bounds.min.x - originalBounds.min.x) / (float)leftPaddingSubdivLevel),
481 bottomPaddingSubdivLevel * Mathf.Floor(Mathf.Abs(bounds.min.y - originalBounds.min.y) / (float)bottomPaddingSubdivLevel),
482 backPaddingSubdivLevel * Mathf.Floor(Mathf.Abs(bounds.min.z - originalBounds.min.z) / (float)backPaddingSubdivLevel));
483 bounds.max = bounds.max - new Vector3(
484 rightPaddingSubdivLevel * Mathf.Floor(Mathf.Abs(bounds.max.x - originalBounds.max.x) / (float)rightPaddingSubdivLevel),
485 topPaddingSubdivLevel * Mathf.Floor(Mathf.Abs(bounds.max.y - originalBounds.max.y) / (float)topPaddingSubdivLevel),
486 forwardPaddingSubdivLevel * Mathf.Floor(Mathf.Abs(bounds.max.z - originalBounds.max.z) / (float)forwardPaddingSubdivLevel)
487 );
488 }

490 internal void UpdateSceneBounds(Scene scene, string sceneGUID, bool onSceneSave)
491 {
492 var volumes = FindObjectsByType<ProbeVolume>(FindObjectsSortMode.None);
493 float prevBrickSize = ProbeReferenceVolume.instance.MinBrickSize();
494 int prevMaxSubdiv = ProbeReferenceVolume.instance.GetMaxSubdivision();
495 Vector3 prevOffset = ProbeReferenceVolume.instance.ProbeOffset();
496
497 if (onSceneSave) // Use baked valuse
498 ProbeReferenceVolume.instance.SetSubdivisionDimensions(minBrickSize, maxSubdivision, bakedProbeOffset);
499 else // Use displayed values
500 ProbeReferenceVolume.instance.SetSubdivisionDimensions(GetMinBrickSize(minDistanceBetweenProbes), GetMaxSubdivision(simplificationLevels), probeOffset);
501
502 bool boundFound = false;
503 Bounds newBound = new Bounds();
504 foreach(var volume in volumes)
505 {
506 bool forceUpdate = onSceneSave && volume.mode == ProbeVolume.Mode.Global;
507 if (!forceUpdate && volume.gameObject.scene != scene)
508 continue;
509
510 if (volume.mode != ProbeVolume.Mode.Local)
511 volume.UpdateGlobalVolume(volume.mode == ProbeVolume.Mode.Global ? GIContributor.ContributorFilter.All : GIContributor.ContributorFilter.Scene);
512
513 var obb = new ProbeReferenceVolume.Volume(volume.GetVolume(), 0, 0);
514 Bounds localBounds = obb.CalculateAABB();
515
516 InflateBound(ref localBounds, volume);
517
518 if(!boundFound)
519 {
520 newBound = localBounds;
521 boundFound = true;
522 }
523 else
524 {
525 newBound.Encapsulate(localBounds);
526 }
527 }
528
529 bool bakeDataExist = m_SceneBakeData.TryGetValue(sceneGUID, out var bakeData);
530 Debug.Assert(bakeDataExist, "Scene should have been added to the baking set with default bake data instance.");
531 if (boundFound)
532 bakeData.bounds = newBound;
533
534 ProbeReferenceVolume.instance.SetSubdivisionDimensions(prevBrickSize, prevMaxSubdiv, prevOffset);
535 EditorUtility.SetDirty(this);
536 }
