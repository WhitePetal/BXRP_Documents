30void RasterPipeline::LoadContent()
31{
32 Application& app = Application::Get();
33 auto device = app.GetDevice();
34
35 // Load the vertex shader
36 ComPtr<ID3DBlob> vertexShaderBlob;
37 ThrowIfFaild(D3DReadFileToBlob(L"VertexShader.cso", &vertexShaderBlob));
38
39 // Load the pixel shader
40 ComPtr<ID3DBlob> pixelShaderBlob;
41 ThrowIfFaild(D3DReadFileToBlob(L"PixelShader.cso", &pixelShaderBlob));
42
43 // Create the vertex input layout
44 D3D12_INPUT_ELEMENT_DESC inputLayout[] = {
45 {"POSITION", 0, DXGI_FORMAT_R32G32B32_FLOAT, 0, D3D12_APPEND_ALIGNED_ELEMENT, D3D12_INPUT_CLASSIFICATION_PER_VERTEX_DATA, 0},
46 {"COLOR", 0, DXGI_FORMAT_R32G32B32_FLOAT, 0, D3D12_APPEND_ALIGNED_ELEMENT, D3D12_INPUT_CLASSIFICATION_PER_VERTEX_DATA, 0}
47 };
48
49 // Create a root signature.
50 D3D12_FEATURE_DATA_ROOT_SIGNATURE featureData = {};
51 featureData.HighestVersion = D3D_ROOT_SIGNATURE_VERSION_1_1;
52 if (FAILED(device->CheckFeatureSupport(D3D12_FEATURE_ROOT_SIGNATURE, &featureData, sizeof(featureData))))
53 {
54 featureData.HighestVersion = D3D_ROOT_SIGNATURE_VERSION_1_0;
55 }
56
57 // Allow input layout and deny unnecessary access to certain pipeline stages
58 D3D12_ROOT_SIGNATURE_FLAGS rootSignatureFlags =
59 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT |
60 D3D12_ROOT_SIGNATURE_FLAG_DENY_HULL_SHADER_ROOT_ACCESS |
61 D3D12_ROOT_SIGNATURE_FLAG_DENY_DOMAIN_SHADER_ROOT_ACCESS |
62 D3D12_ROOT_SIGNATURE_FLAG_DENY_GEOMETRY_SHADER_ROOT_ACCESS |
63 D3D12_ROOT_SIGNATURE_FLAG_DENY_PIXEL_SHADER_ROOT_ACCESS;
64
65 // Camera Matrixs
66 CD3DX12_ROOT_PARAMETER1 rootParameter;
67 CD3DX12_DESCRIPTOR_RANGE1 range;
68 range.Init(D3D12_DESCRIPTOR_RANGE_TYPE_CBV, 1, 0);
69 rootParameter.InitAsDescriptorTable(1, &range, D3D12_SHADER_VISIBILITY_VERTEX);
70
71 CD3DX12_VERSIONED_ROOT_SIGNATURE_DESC rootSignatureDescription;
72 rootSignatureDescription.Init_1_1(1, &rootParameter, 0, nullptr, D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
73
74 // Serialize the root signature
75 ComPtr<ID3DBlob> rootSignatureBlob;
76 ComPtr<ID3DBlob> errorBlob;
77 ThrowIfFaild(D3DX12SerializeVersionedRootSignature(&rootSignatureDescription, featureData.HighestVersion, &rootSignatureBlob, &errorBlob));
78 // Create the root signature.
79 ThrowIfFaild(device->CreateRootSignature(0, rootSignatureBlob->GetBufferPointer(), rootSignatureBlob->GetBufferSize(), IID_PPV_ARGS(&m_RootSignature)));

107 ThrowIfFaild(device->CreatePipelineState(&pipelineStateStreamDesc, IID_PPV_ARGS(&m_PipelineState)));

116void RasterPipeline::OnRender(ComPtr<ID3D12GraphicsCommandList4> commandList, D3D12_CPU_DESCRIPTOR_HANDLE rtv, D3D12_CPU_DESCRIPTOR_HANDLE dsv)
117{
118 commandList->SetPipelineState(m_PipelineState.Get());
119 commandList->SetGraphicsRootSignature(m_RootSignature.Get());
120
121 // Clear the render targets
122 FLOAT clearColor[] = { 0.4f, 0.6f, 0.9f, 1.0f };
123 m_Engine->ClearRTV(commandList, rtv, clearColor);
124 ClearDepth(commandList, dsv);
125
126 commandList->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
127 commandList->IASetVertexBuffers(0, 1, m_Engine->GetVertexBufferView());
128 commandList->IASetIndexBuffer(m_Engine->GetIndexBufferView());
129
130 // Update the Camera Matrixs
131 ComPtr<ID3D12DescriptorHeap> camerConstHeap = m_Engine->GetCamera()->GetConstHeap();
132 std::vector<ID3D12DescriptorHeap*> heaps = { camerConstHeap.Get() };
133 commandList->SetDescriptorHeaps(static_cast<UINT>(heaps.size()), heaps.data());
134 commandList->SetGraphicsRootDescriptorTable(
135 0, camerConstHeap->GetGPUDescriptorHandleForHeapStart()
136 );
137
138 commandList->DrawIndexedInstanced(_countof(g_Indicies), 1, 0, 0, 0);
139}
