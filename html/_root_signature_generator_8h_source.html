26/*
27Contacts for feedback:
28- pgautron@nvidia.com (Pascal Gautron)
29- mlefrancois@nvidia.com (Martin-Karl Lefrancois)
30
31Utility class to create root signatures. The order in which the addition methods are called is
32important as it will define the slots of the heap or of the Shader Binding Table to which buffer
33pointers will be bound.
34
35Example to create an empty root signature:
36nv_helpers_dx12::RootSignatureGenerator rsc;
37return rsc.Generate(m_device.Get(), true);
38
39Example to create a signature with one constant buffer:
40nv_helpers_dx12::RootSignatureGenerator rsc;
41rsc.AddRootParameter(D3D12_ROOT_PARAMETER_TYPE_CBV);
42return rsc.Generate(m_device.Get(), true);
43
44More advance root signature:
45nv_helpers_dx12::RootSignatureGenerator rsc;
46rsc.AddRangeParameter({{0,1,0, D3D12_DESCRIPTOR_RANGE_TYPE_UAV, 0},
47{0,1,0, D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 1},
48{0,1,0, D3D12_DESCRIPTOR_RANGE_TYPE_CBV, 2}});
49return rsc.Generate(m_device.Get(), true);
50
51*/
