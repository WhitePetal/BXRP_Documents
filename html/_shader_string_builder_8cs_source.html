126 public void AppendLines(string lines)
127 {
128 if (string.IsNullOrEmpty(lines))
129 return;
130
131 int startSearchIndex = 0;
132 int indexOfNextBreak = lines.IndexOfAny(LineSeparators);
133 while (indexOfNextBreak >= 0)
134 {
135 AppendLine(lines, startSearchIndex, indexOfNextBreak - startSearchIndex);
136 startSearchIndex = indexOfNextBreak + 1;
137 indexOfNextBreak = lines.IndexOfAny(LineSeparators, startSearchIndex);
138 }
139
140 if (startSearchIndex < lines.Length)
141 {
142 AppendLine(lines, startSearchIndex, lines.Length - startSearchIndex);
143 }
144 }
