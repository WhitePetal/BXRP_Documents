249 void SortEntries(List<NodeEntry> nodeEntries)
250 {
251 // Sort the entries lexicographically by group then title with the requirement that items always comes before sub-groups in the same group.
252 // Example result:
253 // - Art/BlendMode
254 // - Art/Adjustments/ColorBalance
255 // - Art/Adjustments/Contrast
256 nodeEntries.Sort((entry1, entry2) =>
257 {
258 for (var i = 0; i < entry1.title.Length; i++)
259 {
260 if (i >= entry2.title.Length)
261 return 1;
262 var value = entry1.title[i].CompareTo(entry2.title[i]);
263 if (value != 0)
264 {
265 // Make sure that leaves go before nodes
266 if (entry1.title.Length != entry2.title.Length && (i == entry1.title.Length - 1 || i == entry2.title.Length - 1))
267 {
268 //once nodes are sorted, sort slot entries by slot order instead of alphebetically
269 var alphaOrder = entry1.title.Length < entry2.title.Length ? -1 : 1;
270 var slotOrder = entry1.compatibleSlotId.CompareTo(entry2.compatibleSlotId);
271 return alphaOrder.CompareTo(slotOrder);
272 }
273
274 return value;
275 }
276 }
277 return 0;
278 });
279 }

282 {

330 public Searcher LoadSearchWindow()
331 {
332 if (regenerateEntries)
333 {
334 GenerateNodeEntries();
335 regenerateEntries = false;
336 }
337
338 // create empty root for searcher tree
339 var root = new List<SearcherItem>();
340 var dummyEntry = new NodeEntry();
341
342 foreach (var nodeEntry in currentNodeEntries)
343 {
344 SearcherItem item = null;
345 SearcherItem parent = null;
346 for (int i = 0; i < nodeEntry.title.Length; i++)
347 {
348 var pathEntry = nodeEntry.title[i];
349 List<SearcherItem> children = parent != null ? parent.Children : root;
350 item = children.Find(x => x.Name == pathEntry);
351
352 if (item == null)
353 {
354 //if we have slot entries and are at a leaf, add the slot name to the entry title
355 if (nodeEntry.compatibleSlotId != -1 && i == nodeEntry.title.Length - 1)
356 item = new SearchNodeItem(pathEntry + ": " + nodeEntry.slotName, nodeEntry, nodeEntry.node.synonyms);
357 //if we don't have slot entries and are at a leaf, add userdata to the entry
358 else if (nodeEntry.compatibleSlotId == -1 && i == nodeEntry.title.Length - 1)
359 item = new SearchNodeItem(pathEntry, nodeEntry, nodeEntry.node.synonyms);
360 //if we aren't a leaf, don't add user data
361 else
362 item = new SearchNodeItem(pathEntry, dummyEntry, null);
363
364 if (parent != null)
365 {
366 parent.AddChild(item);
367 }
368 else
369 {
370 children.Add(item);
371 }
372 }
373
374 parent = item;
375
376 if (parent.Depth == 0 && !root.Contains(parent))
377 root.Add(parent);
378 }
379 }
380
381 var nodeDatabase = SearcherDatabase.Create(root, string.Empty, false);
382
383 return new Searcher(nodeDatabase, new SearchWindowAdapter("Create Node"));
384 }

386 public bool OnSearcherSelectEntry(SearcherItem entry, Vector2 screenMousePosition)
387 {
388 if (entry == null || (entry as SearchNodeItem).NodeGUID.node == null)
389 return true;
390
391 var nodeEntry = (entry as SearchNodeItem).NodeGUID;
392
393 if (nodeEntry.node is PropertyNode propNode)
394 if (propNode.property is MultiJsonInternal.UnknownGeometryPropertyType)
395 return true;
396
397 var node = CopyNodeForGraph(nodeEntry.node);
398
399 var windowRoot = m_EditorWindow.rootVisualElement;
400 var windowMousePosition = windowRoot.ChangeCoordinatesTo(windowRoot.parent, screenMousePosition); //- m_EditorWindow.position.position);
401 var graphMousePosition = m_GraphView.contentViewContainer.WorldToLocal(windowMousePosition);
402
403 m_Graph.owner.RegisterCompleteObjectUndo("Add " + node.name);
404
405 if (node is BlockNode blockNode)
406 {
407 if (!(target is ContextView contextView))
408 return true;
409
410 // ensure custom blocks have a unique name provided the existing context.
411 if (blockNode.isCustomBlock)
412 {
413 HashSet<string> usedNames = new HashSet<string>();
414 foreach (var other in contextView.contextData.blocks) usedNames.Add(other.value.descriptor.displayName);
415 blockNode.customName = GraphUtil.SanitizeName(usedNames, "{0}_{1}", blockNode.descriptor.displayName);
416 }
417 // Test against all current BlockNodes in the Context
418 // Never allow duplicate BlockNodes
419 else if (contextView.contextData.blocks.Where(x => x.value.name == blockNode.name).FirstOrDefault().value != null)
420 {
421 return true;
422 }
423
424 // Insert block to Data
425 blockNode.owner = m_Graph;
426 int index = contextView.GetInsertionIndex(screenMousePosition);
427 m_Graph.AddBlock(blockNode, contextView.contextData, index);
428 return true;
429 }
430
431 var drawState = node.drawState;
432 drawState.position = new Rect(graphMousePosition, Vector2.zero);
433 node.drawState = drawState;
434 m_Graph.AddNode(node);
435
436 if (connectedPort != null)
437 {
438 var connectedSlot = connectedPort.slot;
439 var connectedSlotReference = connectedSlot.owner.GetSlotReference(connectedSlot.id);
440 var compatibleSlotReference = node.GetSlotReference(nodeEntry.compatibleSlotId);
441
442 var fromReference = connectedSlot.isOutputSlot ? connectedSlotReference : compatibleSlotReference;
443 var toReference = connectedSlot.isOutputSlot ? compatibleSlotReference : connectedSlotReference;
444 m_Graph.Connect(fromReference, toReference);
445
446 nodeNeedsRepositioning = true;
447 targetSlotReference = compatibleSlotReference;
448 targetPosition = graphMousePosition;
449 }
450
451 return true;
452 }
