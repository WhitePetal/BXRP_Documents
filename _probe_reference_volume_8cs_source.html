576 public Bounds CalculateAABB()
577 {
578 Vector3 min = new Vector3(float.MaxValue, float.MaxValue, float.MaxValue);
579 Vector3 max = new Vector3(float.MinValue, float.MinValue, float.MinValue);
580
581 for (int x = 0; x < 2; ++x)
582 {
583 for (int y = 0; y < 2; ++y)
584 {
585 for (int z = 0; z < 2; ++z)
586 {
587 Vector3 dir = new Vector3(x, y, z);
588 Vector3 pt = corner + X * dir.x + Y * dir.y + Z * dir.z;
589 min = Vector3.Min(min, pt);
590 max = Vector3.Max(max, pt);
591 }
592 }
593 }
594
595 return new Bounds((min + max) * 0.5f, max - min);
596 }
597
598 public void CalculateCenterAndSize(out Vector3 center, out Vector3 size)
599 {
600 size = new Vector3(X.magnitude, Y.magnitude, Z.magnitude);
601 center = corner + 0.5f * X + 0.5f * Y + 0.5f * Z;
602 }
603
604 public void Transform(Matrix4x4 trs)
605 {
606 corner = trs.MultiplyPoint(corner);
607 X = trs.MultiplyVector(X);
608 Z = trs.MultiplyVector(Y);
609 Z = trs.MultiplyVector(Z);
610 }

1288 // This one is internal for baking purpose only.
1289 // Calling this from "outside" will not properly update Loaded/ToBeloadedCells arrays and thus will break the state of streaming.
1290 internal bool LoadCell(Cell cell, bool ignoreErrorLog = false)
1291 {
1292 // First try to allocate pool memory. This is what is most likely to fail.
1293 if (ReservePoolChunks(cell.desc.bricksCount, cell.poolInfo.chunkList, ignoreErrorLog))
1294 {
1295 int indirectionBufferEntries = cell.indexInfo.indirectionEntryInfo.Length;
1296
1297 var indexInfo = cell.indexInfo;
1298
1299 for (int entry = 0; entry < indirectionBufferEntries; ++entry)
1300 {
1301 // TODO: remove, this is for migration
1302 if (!cell.indexInfo.indirectionEntryInfo[entry].hasMinMax)
1303 {
1304 if (cell.data.bricks.IsCreated)
1305 ComputeEntryMinMax(ref cell.indexInfo.indirectionEntryInfo[entry], cell.data.bricks);
1306 else
1307 {
1308 int entrySize = CellSize(GetEntrySubdivLevel());
1309 cell.indexInfo.indirectionEntryInfo[entry].minBrickPos = Vector3Int.zero;
1310 }
1311 }
1312
1313 int brickCountAtResForEntry = GetNumberOfBricksAtSubdiv(cell.indexInfo.indirectionEntryInfo[entry]);
1314 indexInfo.updateInfo.entriesInfo[entry].numberOfChunks = m_Index.GetNumberOfChunks(brickCountAtResForEntry);
1315 }
1316
1317 bool canAllocatedCell = m_Index.FindSlotsForEntries(ref indexInfo.updateInfo.entriesInfo);
1318 if (canAllocatedCell)
1319 {
1320 bool scenarioValid = cell.UpdateCellScenarioData(lightingScenario, otherScenario);
1321
1322 bool successfulReserve = m_Index.ReserveChunks(indexInfo.updateInfo.entriesInfo, ignoreErrorLog);
1323 Debug.Assert(successfulReserve);
1324
1325 for (int entry = 0; entry < indirectionBufferEntries; ++entry)
1326 {
1327 indexInfo.updateInfo.entriesInfo[entry].minValidBrickIndexForCellAtMaxRes = indexInfo.indirectionEntryInfo[entry].minBrickPos;
1328 indexInfo.updateInfo.entriesInfo[entry].maxValidBrickIndexForCellAtMaxResPlusOne = indexInfo.indirectionEntryInfo[entry].maxBrickPosPlusOne;
1329 indexInfo.updateInfo.entriesInfo[entry].entryPositionInBricksAtMaxRes = indexInfo.indirectionEntryInfo[entry].positionInBricks;
1330 indexInfo.updateInfo.entriesInfo[entry].minSubdivInCell = indexInfo.indirectionEntryInfo[entry].minSubdiv;
1331 indexInfo.updateInfo.entriesInfo[entry].hasOnlyBiggerBricks = indexInfo.indirectionEntryInfo[entry].hasOnlyBiggerBricks;
1332 }
1333 cell.loaded = true;
1334
1335 // Copy proper data inside index buffers and pool textures or kick off streaming request.
1336 if (scenarioValid)
1337 AddBricks(cell);
1338
1339 minLoadedCellPos = Vector3Int.Min(minLoadedCellPos, cell.desc.position);
1340 maxLoadedCellPos = Vector3Int.Max(maxLoadedCellPos, cell.desc.position);
1341
1342 ClearDebugData();
1343 return true;
1344 }
1345 else
1346 {
1347 // Index allocation failed, we need to release the pool chunks.
1348 ReleasePoolChunks(cell.poolInfo.chunkList);
1349 // We know we should have the space (test done in TryLoadCell above) so it's because of fragmentation.
1350 StartIndexDefragmentation();
1351 }
1352
1353 return false;
1354 }
1355
1356 return false;
1357 }

1545 internal void ComputeEntryMinMax(ref IndirectionEntryInfo entryInfo, ReadOnlySpan<Brick> bricks)
1546 {
1547 int entrySize = CellSize(GetEntrySubdivLevel());
1548 Vector3Int entry_min = entryInfo.positionInBricks;
1549 Vector3Int entry_max = entryInfo.positionInBricks + new Vector3Int(entrySize, entrySize, entrySize);
1550
1551 if (entryInfo.hasOnlyBiggerBricks)
1552 {
1553 entryInfo.minBrickPos = entry_min;
1554 entryInfo.maxBrickPosPlusOne = entry_max;
1555 }
1556 else
1557 {
1558 entryInfo.minBrickPos = entryInfo.maxBrickPosPlusOne = Vector3Int.zero;
1559
1560 bool initialized = false;
1561 for (int i = 0; i < bricks.Length; ++i)
1562 {
1563 int brickSize = ProbeReferenceVolume.CellSize(bricks[i].subdivisionLevel);
1564 var brickMin = bricks[i].position;
1565 var brickMax = bricks[i].position + new Vector3Int(brickSize, brickSize, brickSize);
1566 if (!ProbeBrickIndex.BrickOverlapEntry(brickMin, brickMax, entry_min, entry_max))
1567 continue;
1568
1569 // Bricsk can be bigger than entries !
1570 brickMin = Vector3Int.Max(brickMin, entry_min);
1571 brickMax = Vector3Int.Min(brickMax, entry_max);
1572
1573 if (initialized)
1574 {
1575 entryInfo.minBrickPos = Vector3Int.Min(brickMin, entryInfo.minBrickPos);
1576 entryInfo.maxBrickPosPlusOne = Vector3Int.Max(brickMax, entryInfo.maxBrickPosPlusOne);
1577 }
1578 else
1579 {
1580 entryInfo.minBrickPos = brickMin;
1581 entryInfo.maxBrickPosPlusOne = brickMax;
1582 initialized = true;
1583 }
1584 }
1585 }
1586
1587 entryInfo.minBrickPos = entryInfo.minBrickPos - entry_min;
1588 entryInfo.maxBrickPosPlusOne = Vector3Int.one + entryInfo.maxBrickPosPlusOne - entry_min;
1589 entryInfo.hasMinMax = transform;
1590 }
