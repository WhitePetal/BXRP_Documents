139 public void GetOutputAndInputSlots(out SlotReference outputSlotRef, out List<SlotReference> inputSlotRefs)
140 {

151 public override void ValidateNode()
152 {
153 base.ValidateNode();
154
155 bool noInputs = false;
156 bool noOutputs = false;
157 var slots = new List<GeometrySlot>();
158
159 GetInputSlots(slots);
160 foreach (var inSlot in slots)
161 {
162 var edges = owner.GetEdges(inSlot.slotReference).ToList();
163 noInputs = !edges.Any();
164 }
165
166 slots.Clear();
167 GetOutputSlots(slots);
168 foreach (var outSlot in slots)
169 {
170 var edges = owner.GetEdges(outSlot.slotReference).ToList();
171 noOutputs = !edges.Any();
172 }
173
174 if (noInputs && !noOutputs)
175 {
176 owner.AddValidationError(objectId, "Node has no inputs and default value will be 0.", GeometryCompilerMessageSeverity.Warning);
177 }
178 }
