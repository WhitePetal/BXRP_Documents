55 private const int NB_SKY_PRECOMPUTED_DIRECTIONS = 255;
56 private static Vector3[] k_SkyDirections;
57
58 private static Vector3[] GenerateSkyDirections()
59 {
60 var skyDirections = new Vector3[NB_SKY_PRECOMPUTED_DIRECTIONS];
61
62 float sqrtNBpoints = Mathf.Sqrt((float)(NB_SKY_PRECOMPUTED_DIRECTIONS));
63 float phi = 0.0f;
64 float phiMax = 0.0f;
65 float thetaMax = 0.0f;
66
67 // Spiral based sampling on sphere
68 // See http://web.archive.org/web/20120331125729/http://www.math.niu.edu/~rusin/known-math/97/spherefaq
69 // http://www.math.vanderbilt.edu/saffeb/texts/161.pdf
70 for(int i = 0; i < NB_SKY_PRECOMPUTED_DIRECTIONS; ++i)
71 {
72 // theta from 0 to PI
73 // phi from 0 to 2PI
74 float h = -1.0f + (2.0f * i) / (NB_SKY_PRECOMPUTED_DIRECTIONS - 1.0f);
75 float theta = Mathf.Acos(h);
76 if (i == NB_SKY_PRECOMPUTED_DIRECTIONS - 1 || i == 0)
77 phi = 0.0f;
78 else
79 phi = phi * 3.6f / sqrtNBpoints * 1.0f / (Mathf.Sqrt(1.0f - h * h));
80
81 Vector3 pointOnSphere = new Vector3(Mathf.Sin(theta) * Mathf.Cos(phi), Mathf.Sin(theta) * Mathf.Sin(phi), Mathf.Cos(theta));
82
83 pointOnSphere.Normalize();
84 skyDirections[i] = pointOnSphere;
85
86 phiMax = Mathf.Max(phiMax, phi);
87 thetaMax = Mathf.Max(thetaMax, theta);
88 }
89
90 return skyDirections;
91 }

167 private static List<uint3> ComputeMask(uint validityMask)
168 {
169 List<uint3> samples = new();
170
171 // Cube sample
172 if(validityMask == 0 || validityMask == 255)
173 {
174 samples.Add(1);
175 return samples;
176 }
177
178 // track which probes are sampled
179 uint samplingMask = 0;
180
181 // Find face sample
182 for(int i = 0; i < 6; ++i)
183 {
184 int axis = i / 2;
185 uint face = BuildFace(axis, i % 2);
186 if((validityMask & face) == face) // all face is valid, sample it
187 {
188 uint3 offset = 0;
189 offset[axis] = (i % 2) == 0 ? 0u : 2u;
190 offset[(axis + 1) % 3] = 1;
191 offset[(axis + 2) % 3] = 1;
192
193 samples.Add(offset);
194 samplingMask = face;
195 break;
196 }
197 }
198
199 // Find edge samples
200 while (true)
201 {
202 if (!TryGetEdge(validityMask, samplingMask, out uint edge, out uint3 offset))
203 break;
204
205 samples.Add(offset);
206 samplingMask |= edge;
207 }
208
209 // Find single probes samples
210 for(int i = 0; i < 8; ++i)
211 {
212 if (((1 << i) & (validityMask & ~samplingMask)) == 0)
213 continue;
214 samples.Add(2 * GetSampleOffset((uint)i));
215 samplingMask |= (uint)(1 << i);
216 }
217
218 return samples;
219 }

221 private static uint PackSamplingDir(uint val)
222 {
223 // On a single axis there is up to 2 probes. A face or edge sample needs to sample in between the probes
224 // We encode 0 as sample first probe, 1 as sample between probe, 2 as sample second probe (2 bits)
225 // For faster decoding, we use a third bit that reduces ALU in shader
226 return /* 2 bits */ (val << 1) | /* 1 bits */ ((~val & 2) >> 1);
227 }
228
229 private static uint InvalidSampleMask()
230 {
231 // This is a special code that results in no sampling in shader without any additional ALU
232 return 2 | (2 << 3) | (2 << 6);
233 }
234
235 private static uint ComputeAntiLeakData(uint validityMask)
236 {
237 // This may generate more than 3 samples, but we limit to 3
238 var samples = ComputeMask(validityMask);
239 uint mask = 0;
240
241 for(int i = 0; i < 3; ++i)
242 {
243 uint sampleMask;
244 if (i < samples.Count)
245 sampleMask = PackSamplingDir(samples[i].x) | (PackSamplingDir(samples[i].y) << 3) | (PackSamplingDir(samples[i].z) << 6);
246 else
247 sampleMask = InvalidSampleMask();
248
249 // 32bits - 9bits per samples (up to 3 samples)
250 // Each sample encodes sampling on each axis (3axis * 3bits)
251 // See PackSamplingDir for axis encoding
252 mask |= sampleMask << (9 * i);
253 }
254
255 return mask;
256 }
