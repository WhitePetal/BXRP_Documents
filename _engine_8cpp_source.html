239void Engine::UpdateBufferResource(
240 ComPtr<ID3D12GraphicsCommandList2> commandList,
241 ID3D12Resource** pDestinationResource,
242 ID3D12Resource** pIntermediatedResource,
243 size_t numElements, size_t elementSize, const void* bufferData,
244 D3D12_RESOURCE_FLAGS flags
245)
246{
247 auto device = Application::Get().GetDevice();
248 size_t bufferSize = numElements * elementSize;
249
250 // Create a committed resource for the CPU resource in a default heap
251 // default heap 的内存是 GPU内存, CPU 无法直接写入
252 CD3DX12_HEAP_PROPERTIES defaultHeapProperties(D3D12_HEAP_TYPE_DEFAULT);
253 CD3DX12_RESOURCE_DESC bufferDesc = CD3DX12_RESOURCE_DESC::Buffer(bufferSize, flags);
254 ThrowIfFaild(device->CreateCommittedResource(
255 &defaultHeapProperties,
256 D3D12_HEAP_FLAG_NONE,
257 &bufferDesc,
258 D3D12_RESOURCE_STATE_COMMON,
259 nullptr,
260 IID_PPV_ARGS(pDestinationResource)
261 ));
262
263 // Create an committted resource for the upload.
264 // upload heap 的内存是 系统内存，CPU可以写入
265 // 之后通过 commandlist 发起指令让 GPU 从这里复制数据
266 if (bufferData)
267 {
268 CD3DX12_HEAP_PROPERTIES uploadHeapProperties(D3D12_HEAP_TYPE_UPLOAD);
269 ThrowIfFaild(device->CreateCommittedResource(
270 &uploadHeapProperties,
271 D3D12_HEAP_FLAG_NONE,
272 &bufferDesc,
273 D3D12_RESOURCE_STATE_GENERIC_READ,
274 nullptr,
275 IID_PPV_ARGS(pIntermediatedResource)
276 ));
277
278 D3D12_SUBRESOURCE_DATA subresourceData = {};
279 subresourceData.pData = bufferData;
280 subresourceData.RowPitch = bufferSize;
281 subresourceData.SlicePitch = subresourceData.RowPitch;
282
283 UpdateSubresources(commandList.Get(), *pDestinationResource, *pIntermediatedResource, 0, 0, 1, &subresourceData);
284 }
285}

298void Engine::ResizeDepthBuffer(int width, int height)
299{
300 if (m_ContentLoaded)
301 {
302 // Flush any GPU commands that might be referencing the depth buffer.
303 Application::Get().Flush();
304
305 width = std::max(1, width);
306 height = std::max(1, height);
307
308 auto device = Application::Get().GetDevice();
309 // Resize screen dependent reoures.
310 // Create a depth buffer.
311 D3D12_CLEAR_VALUE optimizedClearValue = {};
312 optimizedClearValue.Format = DXGI_FORMAT_D32_FLOAT;
313 optimizedClearValue.DepthStencil = { 1.0f, 0 };
314
315 CD3DX12_HEAP_PROPERTIES heapProerties(D3D12_HEAP_TYPE_DEFAULT);
316 CD3DX12_RESOURCE_DESC tex2DResDesc = CD3DX12_RESOURCE_DESC::Tex2D(DXGI_FORMAT_D32_FLOAT, width, height,
317 1, 0, 1, 0, D3D12_RESOURCE_FLAG_ALLOW_DEPTH_STENCIL);
318 ThrowIfFaild(device->CreateCommittedResource(
319 &heapProerties,
320 D3D12_HEAP_FLAG_NONE,
321 &tex2DResDesc,
322 D3D12_RESOURCE_STATE_DEPTH_WRITE,
323 &optimizedClearValue,
324 IID_PPV_ARGS(&m_DepthBuffer)
325 ));
326
327 // Update the depth-stencil view
328 D3D12_DEPTH_STENCIL_VIEW_DESC dsv = {};
329 dsv.Format = DXGI_FORMAT_D32_FLOAT;
330 dsv.ViewDimension = D3D12_DSV_DIMENSION_TEXTURE2D;
331 dsv.Texture2D.MipSlice = 0;
332 dsv.Flags = D3D12_DSV_FLAG_NONE;
333
334 device->CreateDepthStencilView(m_DepthBuffer.Get(), &dsv,
335 m_DSVHeap->GetCPUDescriptorHandleForHeapStart());
336 }
337}
