235 public void SetOverrideActiveState(ActiveState overrideState, bool updateConnections = true)
236 {
237 if (m_ActiveState == overrideState)
238 {
239 return;
240 }
241
242 m_ActiveState = overrideState;
243 switch (m_ActiveState)
244 {
245 case ActiveState.Implicit:
246 if (updateConnections)
247 {
248 NodeUtils.ReevaluateActivityOfConnectedNodes(this);
249 }
250 break;
251 case ActiveState.ExplicitInactive:
252 if (m_IsActive == false)
253 {
254 break;
255 }
256 else
257 {
258 m_IsActive = false;
259 Dirty(ModificationScope.Node);
260 if (updateConnections)
261 {
262 NodeUtils.ReevaluateActivityOfConnectedNodes(this);
263 }
264 break;
265 }
266 case ActiveState.ExplicitActive:
267 if (m_IsActive == true)
268 {
269 break;
270 }
271 else
272 {
273 m_IsActive = true;
274 Dirty(ModificationScope.Node);
275 if (updateConnections)
276 {
277 NodeUtils.ReevaluateActivityOfConnectedNodes(this);
278 }
279 break;
280 }
281 }

502 public static ConcreteSlotValueType ConvertDynamicVectorInputTypeToConcrete(IEnumerable<ConcreteSlotValueType> inputTypes)
503 {
504 var concreteSlotValueTypes = inputTypes as IList<ConcreteSlotValueType> ?? inputTypes.ToList();
505
506 var inputTypesDistinct = concreteSlotValueTypes.Distinct().ToList();
507 switch (inputTypesDistinct.Count)
508 {
509 case 0:
510 // nothing connected -- use Vec1 by default
511 return ConcreteSlotValueType.Vector1;
512 case 1:
513 if (SlotValueHelper.AreCompatible(SlotValueType.DynamicVector, inputTypesDistinct.First()))
514 {
515 if (inputTypesDistinct.First() == ConcreteSlotValueType.Boolean)
516 return ConcreteSlotValueType.Vector1;
517 return inputTypesDistinct.First();
518 }
519 break;
520 default:
521 // find the 'minumum' channel width excluding 1 as it can promote
522 inputTypesDistinct.RemoveAll(x => (x == ConcreteSlotValueType.Vector1) || (x == ConcreteSlotValueType.Boolean));
523 var ordered = inputTypesDistinct.OrderByDescending(x => x);
524 if (ordered.Any())
525 {
526 var first = ordered.FirstOrDefault();
527 return first;
528 }
529 break;
530 }
531 return ConcreteSlotValueType.Vector1;
532 }

534 public static ConcreteSlotValueType ConvertDynamicMatrixInputTypeToConcrete(IEnumerable<ConcreteSlotValueType> inputTypes)
535 {
536 var concreteSlotValueTypes = inputTypes as IList<ConcreteSlotValueType> ?? inputTypes.ToList();
537
538 var inputTypesDistinct = concreteSlotValueTypes.Distinct().ToList();
539 switch (inputTypesDistinct.Count)
540 {
541 case 0:
542 return ConcreteSlotValueType.Matrix2;
543 case 1:
544 return inputTypesDistinct.FirstOrDefault();
545 default:
546 var ordered = inputTypesDistinct.OrderByDescending(x => x);
547 if (ordered.Any())
548 return ordered.FirstOrDefault();
549 break;
550 }
551 return ConcreteSlotValueType.Matrix2;

556 // evaluate ALL the precisions...
557 public virtual void UpdatePrecision(List<GeometrySlot> inputSlots)
558 {
559 // first let's reduce from precision ==> graph precision
560 if (precision == Precision.Inherit)
561 {
562 // inherit means calculate it automatically based on inputs
563
564 // If no inputs were found use the precision of the Graph
565 if (inputSlots.Count == 0)
566 {
567 graphPrecision = GraphPrecision.Graph;
568 }
569 else
570 {
571 int curGraphPrecision = (int)GraphPrecision.Half;
572 foreach (var inputSlot in inputSlots)
573 {
574 // If input port doesn't have an edge use the Graph's precision for that input
575 var edges = owner?.GetEdges(inputSlot.slotReference).ToList();
576 if (!edges.Any())
577 {
578 // disconnected inputs use graph precision
579 curGraphPrecision = Math.Min(curGraphPrecision, (int)GraphPrecision.Graph);
580 }
581 else
582 {
583 var outputSlotRef = edges[0].outputSlot;
584 var outputNode = outputSlotRef.node;
585 curGraphPrecision = Math.Min(curGraphPrecision, (int)outputNode.graphPrecision);
586 }
587 }
588 graphPrecision = (GraphPrecision)curGraphPrecision;
589 }
590 }
591 else
592 {
593 // not inherited, just use the node's selected precision
594 graphPrecision = precision.ToGraphPrecision(GraphPrecision.Graph);
595 }
596
597 // calculate the concrete precision, with fall-back to the graph concrete precision
598 concretePrecision = graphPrecision.ToConcrete(owner.graphDefaultConcretePrecision);
599 }

812 public GeometrySlot AddSlot(GeometrySlot slot, bool attemptToModifyExistingInstance = true)
813 {
814 if (slot == null)
815 {
816 throw new ArgumentException($"Trying to add null slot to node {this}");
817 }
818 GeometrySlot foundSlot = FindSlot<GeometrySlot>(slot.id);
819
820 if (slot == foundSlot)
821 return foundSlot;
822
823 // Try to keep the existing instance to avoid unnecessary changes to file
824 if (attemptToModifyExistingInstance && foundSlot != null && slot.GetType() == foundSlot.GetType())
825 {
826 foundSlot.displayName = slot.RawDisplayName();
827 foundSlot.CopyDefaultValue(slot);
828 return foundSlot;
829 }
830
831 // keep the same ordering by replacing the first match, if it exists
832 int firstIndex = m_Slots.FindIndex(s => s.value.id == slot.id);
833 if (firstIndex >= 0)
834 {
835 m_Slots[firstIndex] = slot;
836
837 // remove additional matches to get rid of unused duplicates
838 m_Slots.RemoveAllFromRange(s => s.value.id == slot.id, firstIndex + 1, m_Slots.Count - (firstIndex + 1));
839 }
840 else
841 m_Slots.Add(slot);
842

858 public void RemoveSlot(int slotId)
859 {
860 // Remove edges that use this slot
861 // no owner can happen after creation
862 // but before added to graph
863 if (owner != null)
864 {

866 owner.RemoveEdges(edges.ToArray());
867 }
868
869 //remove slots

872 OnSlotsChanged();

893 public bool SetSlotOrder(List<int> desiredOrderSlotIds)
894 {
895 bool changed = false;
896 int writeIndex = 0;
897 for (int orderIndex = 0; orderIndex < desiredOrderSlotIds.Count; orderIndex++)
898 {
899 var id = desiredOrderSlotIds[orderIndex];
900 var matchIndex = m_Slots.FindIndex(s => s.value.id == id);
901 if (matchIndex < 0)
902 {
903 // no matching slot with that id.. skip it
904 }
905 else
906 {
907 if (writeIndex != matchIndex)
908 {
909 // swap the matching slot into position
910 var slot = m_Slots[matchIndex];
911 m_Slots[matchIndex] = m_Slots[writeIndex];
912 m_Slots[writeIndex] = slot;
913 changed = true;
914 }
915 writeIndex++;
916 }
917 }
918 return changed;
