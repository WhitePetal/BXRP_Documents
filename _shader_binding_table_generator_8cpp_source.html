79//
80// Compute the size of the SBT based on the set of programs and hit groups it contains
81uint32_t ShaderBindingTableGenerator::ComputeSBTSize()
82{
83 // Size of a program identifier
84 m_progIdSize = D3D12_SHADER_IDENTIFIER_SIZE_IN_BYTES;
85 // Compute the entry size of each program type depending on the maximum number of parameters in
86 // each category
87 m_rayGenEntrySize = GetEntrySize(m_rayGen);
88 m_missEntrySize = GetEntrySize(m_miss);
89 m_hitGroupEntrySize = GetEntrySize(m_hitGroup);
90
91 // The total SBT size is the sum of the entries for ray generation, miss and hit groups, aligned
92 // on 256 bytes
93 uint32_t sbtSize = ROUND_UP(m_rayGenEntrySize * static_cast<UINT>(m_rayGen.size()) +
94 m_missEntrySize * static_cast<UINT>(m_miss.size()) +
95 m_hitGroupEntrySize * static_cast<UINT>(m_hitGroup.size()),
96 256);
97 return sbtSize;
98}

100//--------------------------------------------------------------------------------------------------
101//
102// Build the SBT and store it into sbtBuffer, which has to be pre-allocated on the upload heap.
103// Access to the raytracing pipeline object is required to fetch program identifiers using their
104// names
105void ShaderBindingTableGenerator::Generate(ID3D12Resource* sbtBuffer,
106 ID3D12StateObjectProperties* raytracingPipeline)
107{
108 // Map the SBT
109 uint8_t* pData;
110 HRESULT hr = sbtBuffer->Map(0, nullptr, reinterpret_cast<void**>(&pData));
111 if (FAILED(hr))
112 {
113 throw std::logic_error("Could not map the shader binding table");
114 }
115 // Copy the shader identifiers followed by their resource pointers or root constants: first the
116 // ray generation, then the miss shaders, and finally the set of hit groups
117 uint32_t offset = 0;
118
119 offset = CopyShaderData(raytracingPipeline, pData, m_rayGen, m_rayGenEntrySize);
120 pData += offset;
121
122 offset = CopyShaderData(raytracingPipeline, pData, m_miss, m_missEntrySize);
123 pData += offset;
124
125 offset = CopyShaderData(raytracingPipeline, pData, m_hitGroup, m_hitGroupEntrySize);
126
127 // Unmap the SBT
128 sbtBuffer->Unmap(0, nullptr);
129}

198//--------------------------------------------------------------------------------------------------
199//
200// For each entry, copy the shader identifier followed by its resource pointers and/or root
201// constants in outputData, with a stride in bytes of entrySize, and returns the size in bytes
202// actually written to outputData.
203uint32_t ShaderBindingTableGenerator::CopyShaderData(
204 ID3D12StateObjectProperties* raytracingPipeline, uint8_t* outputData,
205 const std::vector<SBTEntry>& shaders, uint32_t entrySize)
206{
207 uint8_t* pData = outputData;
208 for (const auto& shader : shaders)
209 {
210 // Get the shader identifier, and check whether that identifier is known
211 void* id = raytracingPipeline->GetShaderIdentifier(shader.m_entryPoint.c_str());
212 if (!id)
213 {
214 std::wstring errMsg(std::wstring(L"Unknown shader identifier used in the SBT: ") +
215 shader.m_entryPoint);
216 throw std::logic_error(std::string(errMsg.begin(), errMsg.end()));
217 }
218 // Copy the shader identifier
219 memcpy(pData, id, m_progIdSize);
220 // Copy all its resources pointers or values in bulk
221 memcpy(pData + m_progIdSize, shader.m_inputData.data(), shader.m_inputData.size() * 8);
222
223 pData += entrySize;
224 }
225 // Return the number of bytes actually written to the output buffer
226 return static_cast<uint32_t>(shaders.size()) * entrySize;
227}

229//--------------------------------------------------------------------------------------------------
230//
231// Compute the size of the SBT entries for a set of entries, which is determined by the maximum
232// number of parameters of their root signature
233uint32_t ShaderBindingTableGenerator::GetEntrySize(const std::vector<SBTEntry>& entries)
234{
235 // Find the maximum number of parameters used by a single entry
236 size_t maxArgs = 0;
237 for (const auto& shader : entries)
238 {
239 maxArgs = max(maxArgs, shader.m_inputData.size());
240 }
241 // A SBT entry is made of a program ID and a set of parameters, taking 8 bytes each. Those
242 // parameters can either be 8-bytes pointers, or 4-bytes constants
243 uint32_t entrySize = m_progIdSize + 8 * static_cast<uint32_t>(maxArgs);
244
245 // The entries of the shader binding table must be 16-bytes-aligned
246 entrySize = ROUND_UP(entrySize, D3D12_RAYTRACING_SHADER_TABLE_BYTE_ALIGNMENT);
247
248 return entrySize;
249}
